```javascript
// Dynamic micro-frontend loading
class MicroFrontendLoader {
  constructor() {
    this.loadedMicroFrontends = new Map();
    this.errorBoundaries = new Map();
  }
  
  async loadMicroFrontend(name, fallbackComponent = null) {
    try {
      if (this.loadedMicroFrontends.has(name)) {
        return this.loadedMicroFrontends.get(name);
      }
      
      // Dynamic import with error handling
      const module = await import(name);
      const Component = module.default || module;
      
      // Wrap with error boundary
      const WrappedComponent = this.wrapWithErrorBoundary(
        Component, 
        name, 
        fallbackComponent
      );
      
      this.loadedMicroFrontends.set(name, WrappedComponent);
      return WrappedComponent;
      
    } catch (error) {
      console.error(`Failed to load micro-frontend: ${name}`, error);
      
      // Fallback strategy
      return fallbackComponent || this.getDefaultFallback(name);
    }
  }
  
  wrapWithErrorBoundary(Component, name, fallback) {
    return class MicroFrontendErrorBoundary extends React.Component {
      constructor(props) {
        super(props);
        this.state = { hasError: false, error: null };
      }
      
      static getDerivedStateFromError(error) {
        return { hasError: true, error };
      }
      
      componentDidCatch(error, errorInfo) {
        // Log error to monitoring service
        this.logErrorToService({
          microFrontend: name,
          error: error.message,
          stack: error.stack,
          errorInfo
        });
      }
      
      logErrorToService(errorData) {
        // Send to error tracking service (e.g., Sentry)
        if (window.Sentry) {
          window.Sentry.captureException(errorData.error, {
            tags: { microFrontend: errorData.microFrontend },
            extra: errorData
          });
        }
      }
      
      render() {
        if (this.state.hasError) {
          return fallback || (
            <div className="micro-frontend-error">
              <h3>Something went wrong with {name}</h3>
              <button onClick={() => this.setState({ hasError: false })}>
                Retry
              </button>
            </div>
          );
        }
        
        return <Component {...this.props} />;
      }
    };
  }
  
  getDefaultFallback(name) {
    return () => (
      <div className="micro-frontend-fallback">
        <div className="loading-skeleton">
          <p>Loading {name}...</p>
        </div>
      </div>
    );
  }

// Cross-micro-frontend communication
class MicroFrontendEventBus {
  constructor() {
    this.events = new Map();
    this.globalState = new Map();
  }
  
  // Event-based communication
  subscribe(eventName, callback, microFrontendId) {
    if (!this.events.has(eventName)) {
      this.events.set(eventName, new Set());
    }
    
    const subscription = { callback, microFrontendId, id: Date.now() };
    this.events.get(eventName).add(subscription);
    
    // Return unsubscribe function
    return () => {
      this.events.get(eventName).delete(subscription);
    };
  }
  
  publish(eventName, data, sourceId) {
    if (this.events.has(eventName)) {
      this.events.get(eventName).forEach(subscription => {
        // Prevent self-notification
        if (subscription.microFrontendId !== sourceId) {
          try {
            subscription.callback(data, sourceId);
          } catch (error) {
            console.error(`Error in event handler for ${eventName}:`, error);
          }
        }
      });
    }
  }
  
  // Shared state management
  setSharedState(key, value, sourceId) {
    const previousValue = this.globalState.get(key);
    this.globalState.set(key, value);
    
    // Notify subscribers of state change
    this.publish(`state:${key}:changed`, {
      key,
      value,
      previousValue,
      timestamp: Date.now()
    }, sourceId);
  }
  
  getSharedState(key) {
    return this.globalState.get(key);
  }
  
  subscribeToState(key, callback, microFrontendId) {
    return this.subscribe(`state:${key}:changed`, callback, microFrontendId);
  }
}
```

```javascript
// Observable State Manager
class ObservableState {
  constructor(initialState = {}) {
    this.state = { ...initialState };
    this.observers = new Map();
    this.middleware = [];
    this.history = [];
    this.maxHistorySize = 50;
  }
  
  // Subscribe to state changes
  subscribe(selector, callback, options = {}) {
    const observerId = this.generateObserverId();
    const observer = {
      id: observerId,
      selector: selector || ((state) => state),
      callback,
      options: {
        immediate: false,
        deep: true,
        ...options
      },
      lastValue: undefined
    };
    
    this.observers.set(observerId, observer);
    
    // Call immediately if requested
    if (observer.options.immediate) {
      const currentValue = observer.selector(this.state);
      observer.lastValue = this.deepClone(currentValue);
      observer.callback(currentValue, undefined, this.state);
    }
    
    // Return unsubscribe function
    return () => {
      this.observers.delete(observerId);
    };
  }
  
  // Update state and notify observers
  setState(updater, actionType = 'UPDATE') {
    const previousState = this.deepClone(this.state);
    
    // Apply state update
    if (typeof updater === 'function') {
      this.state = { ...this.state, ...updater(this.state) };
    } else {
      this.state = { ...this.state, ...updater };
    }
    
    // Apply middleware
    this.applyMiddleware({
      type: actionType,
      previousState,
      currentState: this.state,
      timestamp: Date.now()
    });
    
    // Add to history
    this.addToHistory({
      action: actionType,
      previousState,
      currentState: this.deepClone(this.state),
      timestamp: Date.now()
    });
    
    // Notify observers
    this.notifyObservers(previousState);
  }
  
  notifyObservers(previousState) {
    this.observers.forEach(observer => {
      try {
        const currentValue = observer.selector(this.state);
        const previousValue = observer.selector(previousState);
        
        // Check if value actually changed
        if (this.hasValueChanged(currentValue, observer.lastValue, observer.options.deep)) {
          observer.lastValue = this.deepClone(currentValue);
          observer.callback(currentValue, previousValue, this.state);
        }
      } catch (error) {
        console.error('Error in observer callback:', error);
      }
    });
  }
  
  hasValueChanged(current, previous, deep = true) {
    if (!deep) {
      return current !== previous;
    }
    
    return JSON.stringify(current) !== JSON.stringify(previous);
  }
  
  // Middleware support
  use(middleware) {
    this.middleware.push(middleware);
  }
  
  applyMiddleware(action) {
    this.middleware.forEach(middleware => {
      try {
        middleware(action, this);
      } catch (error) {
        console.error('Middleware error:', error);
      }
    });
  }
  
  // History management
  addToHistory(entry) {
    this.history.push(entry);
    if (this.history.length > this.maxHistorySize) {
      this.history.shift();
    }
  }
  
  getHistory() {
    return [...this.history];
  }
  
  // Utility methods
  getState() {
    return this.deepClone(this.state);
  }
  
  deepClone(obj) {
    return JSON.parse(JSON.stringify(obj));
  }
  
  generateObserverId() {
    return `observer_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
}

// React Hook for Observable State
function useObservableState(store, selector, options) {
  const [state, setState] = useState(() => {
    const initialValue = selector ? selector(store.getState()) : store.getState();
    return initialValue;
  });
  
  useEffect(() => {
    const unsubscribe = store.subscribe(
      selector,
      (newValue) => {
        setState(newValue);
      },
      options
    );
    
    return unsubscribe;
  }, [store, selector, options]);
  
  return state;
}

class CompositeStore {
  constructor(stores) {
    this.stores = stores;
    this.combinedObservers = new Map();
  }
  
  // Subscribe to multiple stores
  subscribeToMultiple(selectors, callback) {
    const unsubscribers = [];
    const values = {};
    
    Object.keys(selectors).forEach(storeName => {
      const store = this.stores[storeName];
      const selector = selectors[storeName];
      
      const unsubscribe = store.subscribe(selector, (newValue) => {
        values[storeName] = newValue;
        callback(values);
      });
      
      unsubscribers.push(unsubscribe);
    });
    
    return () => {
      unsubscribers.forEach(unsub => unsub());
    };
  }
}
```

```javascript
// HOC for authentication logic
function withAuthentication(WrappedComponent, options = {}) {
  const {
    redirectTo = '/login',
    requiredRoles = [],
    loadingComponent: LoadingComponent = DefaultLoader,
    unauthorizedComponent: UnauthorizedComponent = DefaultUnauthorized
  } = options;
  
  return function AuthenticatedComponent(props) {
    const [authState, setAuthState] = useState({
      isLoading: true,
      isAuthenticated: false,
      user: null,
      error: null
    });
    
    useEffect(() => {
      const checkAuthentication = async () => {
        try {
          const token = localStorage.getItem('authToken');
          if (!token) {
            setAuthState({
              isLoading: false,
              isAuthenticated: false,
              user: null,
              error: null
            });
            return;
          }
          
          const response = await fetch('/api/auth/verify', {
            headers: { Authorization: `Bearer ${token}` }
          });
          
          if (response.ok) {
            const user = await response.json();
            
            // Check role requirements
            const hasRequiredRole = requiredRoles.length === 0 || 
              requiredRoles.some(role => user.roles.includes(role));
            
            setAuthState({
              isLoading: false,
              isAuthenticated: true,
              user,
              hasRequiredRole,
              error: null
            });
          } else {
            throw new Error('Authentication failed');
          }
        } catch (error) {
          setAuthState({
            isLoading: false,
            isAuthenticated: false,
            user: null,
            error: error.message
          });
          
          // Redirect to login
          window.location.href = redirectTo;
        }
      };
      
      checkAuthentication();
    }, []);
    
    // Loading state
    if (authState.isLoading) {
      return <LoadingComponent />;
    }
    
    // Not authenticated
    if (!authState.isAuthenticated) {
      return null; // Will redirect
    }
    
    // Insufficient permissions
    if (requiredRoles.length > 0 && !authState.hasRequiredRole) {
      return <UnauthorizedComponent requiredRoles={requiredRoles} />;
    }
    
    // Render wrapped component with auth props
    return (
      <WrappedComponent 
        {...props} 
        user={authState.user}
        isAuthenticated={authState.isAuthenticated}
      />
    );
  };
}

// HOC for data fetching
function withDataFetching(WrappedComponent, config) {
  const {
    endpoint,
    dependencies = [],
    transform = (data) => data,
    errorComponent: ErrorComponent = DefaultError,
    loadingComponent: LoadingComponent = DefaultLoader
  } = config;
  
  return function DataFetchingComponent(props) {
    const [state, setState] = useState({
      data: null,
      loading: true,
      error: null
    });
    
    const fetchData = useCallback(async () => {
      setState(prev => ({ ...prev, loading: true, error: null }));
      
      try {
        const url = typeof endpoint === 'function' ? endpoint(props) : endpoint;
        const response = await fetch(url);
        
        if (!response.ok) {
          throw new Error(`HTTP ${response.status}: ${response.statusText}`);
        }
        
        const rawData = await response.json();
        const transformedData = transform(rawData);
        
        setState({
          data: transformedData,
          loading: false,
          error: null
        });
      } catch (error) {
        setState({
          data: null,
          loading: false,
          error: error.message
        });
      }
    }, [props, ...dependencies]);
    
    useEffect(() => {
      fetchData();
    }, [fetchData]);
    
    if (state.loading) {
      return <LoadingComponent />;
    }
    
    if (state.error) {
      return <ErrorComponent error={state.error} onRetry={fetchData} />;
    }
    
    return (
      <WrappedComponent 
        {...props} 
        data={state.data}
        refetch={fetchData}
        loading={state.loading}
      />
    );
  };
}
```

// Render Props for form state management
class FormState extends Component {
  constructor(props) {
    super(props);
    this.state = {
      values: props.initialValues || {},
      errors: {},
      touched: {},
      isSubmitting: false,
      isValid: true
    };
  }
  
  validateField = (name, value) => {
    const { validationSchema } = this.props;
    if (!validationSchema || !validationSchema[name]) {
      return null;
    }
    
    const validator = validationSchema[name];
    try {
      if (typeof validator === 'function') {
        return validator(value, this.state.values);
      }
      
      // Support for validation objects
      if (validator.required && (!value || value.trim() === '')) {
        return validator.message || `${name} is required`;
      }
      
      if (validator.minLength && value.length < validator.minLength) {
        return validator.message || `${name} must be at least ${validator.minLength} characters`;
      }
      
      if (validator.pattern && !validator.pattern.test(value)) {
        return validator.message || `${name} format is invalid`;
      }
      
      return null;
    } catch (error) {
      return error.message;
    }
  };
  
  handleChange = (name, value) => {
    const error = this.validateField(name, value);
    
    this.setState(prevState => {
      const newValues = { ...prevState.values, [name]: value };
      const newErrors = { ...prevState.errors };
      
      if (error) {
        newErrors[name] = error;
      } else {
        delete newErrors[name];
      }
      
      const isValid = Object.keys(newErrors).length === 0;
      
      return {
        values: newValues,
        errors: newErrors,
        isValid
      };
    });
  };
  
  handleBlur = (name) => {
    this.setState(prevState => ({
      touched: { ...prevState.touched, [name]: true }
    }));
  };
  
  handleSubmit = async (onSubmit) => {
    // Mark all fields as touched
    const touched = {};
    Object.keys(this.state.values).forEach(key => {
      touched[key] = true;
    });
    
    this.setState({ touched, isSubmitting: true });
    
    // Validate all fields
    const errors = {};
    Object.keys(this.state.values).forEach(name => {
      const error = this.validateField(name, this.state.values[name]);
      if (error) {
        errors[name] = error;
      }
    });
    
    if (Object.keys(errors).length > 0) {
      this.setState({ errors, isSubmitting: false, isValid: false });
      return;
    }
    
    try {
      await onSubmit(this.state.values);
      this.setState({ isSubmitting: false });
    } catch (error) {
      this.setState({ 
        isSubmitting: false,
        errors: { submit: error.message }
      });
    }
  };
  
  reset = () => {
    this.setState({
      values: this.props.initialValues || {},
      errors: {},
      touched: {},
      isSubmitting: false,
      isValid: true
    });
  };
  
  render() {
    const { children } = this.props;
    const { values, errors, touched, isSubmitting, isValid } = this.state;
    
    return children({
      values,
      errors,
      touched,
      isSubmitting,
      isValid,
      handleChange: this.handleChange,
      handleBlur: this.handleBlur,
      handleSubmit: this.handleSubmit,
      reset: this.reset,
      getFieldProps: (name) => ({
        value: values[name] || '',
        onChange: (e) => this.handleChange(name, e.target.value),
        onBlur: () => this.handleBlur(name),
        error: touched[name] && errors[name]
      })
    });
  }
}

// Render Props for data fetching with caching
const DataProvider = ({ children, endpoint, cacheKey, dependencies = [] }) => {
  const [state, setState] = useState({
    data: null,
    loading: true,
    error: null,
    lastFetch: null
  });
  
  const cache = useRef(new Map());
  const abortController = useRef(null);
  
  const fetchData = useCallback(async (force = false) => {
    // Check cache first
    if (!force && cacheKey && cache.current.has(cacheKey)) {
      const cachedData = cache.current.get(cacheKey);
      const cacheAge = Date.now() - cachedData.timestamp;
      
      // Use cache if less than 5 minutes old
      if (cacheAge < 5 * 60 * 1000) {
        setState({
          data: cachedData.data,
          loading: false,
          error: null,
          lastFetch: cachedData.timestamp
        });
        return;
      }
    }
    
    // Cancel previous request
    if (abortController.current) {
      abortController.current.abort();
    }
    
    abortController.current = new AbortController();
    setState(prev => ({ ...prev, loading: true, error: null }));
    
    try {
      const response = await fetch(endpoint, {
        signal: abortController.current.signal
      });
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      
      const data = await response.json();
      const timestamp = Date.now();
      
      // Cache the result
      if (cacheKey) {
        cache.current.set(cacheKey, { data, timestamp });
      }
      
      setState({
        data,
        loading: false,
        error: null,
        lastFetch: timestamp
      });
    } catch (error) {
      if (error.name !== 'AbortError') {
        setState({
          data: null,
          loading: false,
          error: error.message,
          lastFetch: null
        });
      }
    }
  }, [endpoint, cacheKey, ...dependencies]);
  
  useEffect(() => {
    fetchData();
    
    return () => {
      if (abortController.current) {
        abortController.current.abort();
      }
    };
  }, [fetchData]);
  
  return children({
    ...state,
    refetch: () => fetchData(true),
    clearCache: () => {
      if (cacheKey) {
        cache.current.delete(cacheKey);
      }
    }
  });
};

## 5. 面试技巧与建议 - Interview Tips and Suggestions

### 展示架构思维的要点

1. **系统性思考 (Systematic Thinking)**
   - 从整体到局部分析问题
   - 考虑可扩展性、可维护性、性能等多个维度
   - 权衡不同方案的优缺点

2. **设计原则 (Design Principles)**
   - 单一职责原则 (Single Responsibility Principle)
   - 开闭原则 (Open/Closed Principle)
   - 依赖倒置原则 (Dependency Inversion Principle)
   - 接口隔离原则 (Interface Segregation Principle)

3. **架构决策 (Architecture Decisions)**
   - 解释为什么选择某种架构模式
   - 讨论技术选型的考虑因素
   - 分析业务需求对架构的影响

### 常用表达句式

**描述架构设计：**
- "I designed a modular architecture that separates concerns..."
- "The system follows a layered architecture pattern where..."
- "I implemented a micro-frontend approach to enable..."
- "The component hierarchy is structured to promote reusability..."

**解释设计模式：**
- "I used the Observer pattern to handle state changes..."
- "The Factory pattern helps us create components dynamically..."
- "I implemented a Singleton to ensure single instance..."
- "The Strategy pattern allows us to switch algorithms..."

**讨论技术选型：**
- "I chose this approach because it provides better scalability..."
- "This pattern fits our use case because..."
- "The trade-off here is between performance and maintainability..."
- "This solution addresses the specific requirements of..."

**分析性能影响：**
- "This architecture reduces bundle size by..."
- "The lazy loading strategy improves initial load time..."
- "Code splitting helps us achieve better performance..."
- "The caching strategy reduces server load and improves UX..."

### 避免的误区

1. **过度设计 (Over-engineering)**
   - 不要为了使用设计模式而使用
   - 避免过度抽象导致代码复杂
   - 根据实际需求选择合适的架构

2. **忽视性能 (Ignoring Performance)**
   - 考虑架构决策对性能的影响
   - 平衡代码组织和运行效率
   - 关注用户体验和开发体验

3. **缺乏扩展性考虑 (Lack of Scalability Consideration)**
   - 考虑未来功能扩展的可能性
   - 设计灵活的接口和抽象层
   - 预留扩展点和配置选项

4. **忽视团队协作 (Ignoring Team Collaboration)**
   - 考虑代码的可读性和可维护性
   - 建立清晰的代码规范和文档
   - 设计易于理解和使用的API

### 深入讨论的话题

1. **架构演进 (Architecture Evolution)**
   - 如何从单体应用迁移到微前端
   - 渐进式重构的策略和实践
   - 技术债务的管理和偿还

2. **跨团队协作 (Cross-team Collaboration)**
   - 如何设计团队间的接口规范
   - 共享组件库的设计和维护
   - 版本管理和发布策略

3. **质量保证 (Quality Assurance)**
   - 架构层面的测试策略
   - 代码质量监控和度量
   - 持续集成和部署流程

4. **技术选型 (Technology Selection)**
   - 评估新技术的标准和流程
   - 技术栈统一vs多样化的权衡
   - 开源vs自研的决策考虑
