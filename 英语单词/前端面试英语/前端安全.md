```javascript
function useCSRFToken() {
  const [csrfToken, setCSRFToken] = useState(null);

    fetchCSRFToken();

// Axios interceptor for automatic CSRF token inclusion
import axios from 'axios';

class APIClient {
  constructor() {
    this.csrfToken = null;
    this.setupInterceptors();
  }
  
  async initializeCSRFToken() {
    try {
      const response = await axios.get('/api/csrf-token');
      this.csrfToken = response.data.csrfToken;
    } catch (error) {
      console.error('Failed to initialize CSRF token:', error);
    }
  }
  
  setupInterceptors() {
    // Request interceptor to add CSRF token
    axios.interceptors.request.use(
      (config) => {
        // Add CSRF token to state-changing requests
        if (['post', 'put', 'patch', 'delete'].includes(config.method)) {
          if (this.csrfToken) {
            config.headers['X-CSRF-Token'] = this.csrfToken;
          }
        }
        return config;
      },
      (error) => Promise.reject(error)
    );
    
    // Response interceptor to handle CSRF token refresh
    axios.interceptors.response.use(
      (response) => response,
      async (error) => {
        if (error.response?.status === 403 && 
            error.response?.data?.code === 'INVALID_CSRF_TOKEN') {
          // Refresh CSRF token and retry
          await this.initializeCSRFToken();
          
          // Retry the original request
          const originalRequest = error.config;
          originalRequest.headers['X-CSRF-Token'] = this.csrfToken;
          return axios.request(originalRequest);
        }
        return Promise.reject(error);
      }
    );
  }
}

const apiClient = new APIClient();
```

```javascript
// JWT utility with security best practices
import jwt from 'jsonwebtoken';
import crypto from 'crypto';

class JWTManager {
  constructor() {
    this.accessTokenSecret = process.env.JWT_ACCESS_SECRET;
    this.refreshTokenSecret = process.env.JWT_REFRESH_SECRET;
    this.accessTokenExpiry = '15m'; // Short-lived access tokens
    this.refreshTokenExpiry = '7d'; // Longer-lived refresh tokens
  }
  
  generateTokenPair(payload) {
    // Generate access token
    const accessToken = jwt.sign(
      {
        ...payload,
        type: 'access',
        jti: crypto.randomUUID() // Unique token ID
      },
      this.accessTokenSecret,
      {
        expiresIn: this.accessTokenExpiry,
        issuer: 'your-app-name',
        audience: 'your-app-users'
      }
    );
    
    // Generate refresh token
    const refreshToken = jwt.sign(
      {
        userId: payload.userId,
        type: 'refresh',
        jti: crypto.randomUUID()
      },
      this.refreshTokenSecret,
      {
        expiresIn: this.refreshTokenExpiry,
        issuer: 'your-app-name',
        audience: 'your-app-users'
      }
    );
    
    return { accessToken, refreshToken };
  }
  
  verifyAccessToken(token) {
    try {
      const decoded = jwt.verify(token, this.accessTokenSecret, {
        issuer: 'your-app-name',
        audience: 'your-app-users'
      });
      
      if (decoded.type !== 'access') {
        throw new Error('Invalid token type');
      }
      
      return decoded;
    } catch (error) {
      throw new Error('Invalid access token');
    }
  }
  
  verifyRefreshToken(token) {
    try {
      const decoded = jwt.verify(token, this.refreshTokenSecret, {
        issuer: 'your-app-name',
        audience: 'your-app-users'
      });
      
      if (decoded.type !== 'refresh') {
        throw new Error('Invalid token type');
      }
      
      return decoded;
    } catch (error) {
      throw new Error('Invalid refresh token');
    }
  }
}
```

**Secure Token Storage**:
```javascript
// Secure token storage in React
class SecureTokenStorage {
  constructor() {
    this.accessTokenKey = 'access_token';
    this.refreshTokenKey = 'refresh_token';
  }
  
  // Store access token in memory (most secure)
  setAccessToken(token) {
    // Store in memory variable (lost on page refresh)
    window.__accessToken = token;
    
    // Also store in sessionStorage as backup
    sessionStorage.setItem(this.accessTokenKey, token);
  }
  
  getAccessToken() {
    // Try memory first, then sessionStorage
    return window.__accessToken || 
           sessionStorage.getItem(this.accessTokenKey);
  }
  
  // Store refresh token in httpOnly cookie (server-side)
  setRefreshToken(token) {
    // This should be done server-side
    document.cookie = `${this.refreshTokenKey}=${token}; ` +
      'HttpOnly; Secure; SameSite=Strict; Max-Age=604800'; // 7 days
  }
  
  clearTokens() {
    // Clear memory
    delete window.__accessToken;
    
    // Clear sessionStorage
    sessionStorage.removeItem(this.accessTokenKey);
    
    // Clear refresh token cookie (server-side endpoint)
    fetch('/api/auth/logout', {
      method: 'POST',
      credentials: 'include'
    });
  }
  
  // Check if token is expired
  isTokenExpired(token) {
    if (!token) return true;
    
    try {
      const decoded = jwt.decode(token);
      const currentTime = Date.now() / 1000;
      return decoded.exp < currentTime;
    } catch (error) {
      return true;
    }
  }
}

const tokenStorage = new SecureTokenStorage();
```

**Authentication Context with Auto-Refresh**:
```javascript
// React authentication context
import React, { createContext, useContext, useEffect, useState } from 'react';

const AuthContext = createContext();

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [accessToken, setAccessToken] = useState(null);
  
  useEffect(() => {
    initializeAuth();
  }, []);
  
  const initializeAuth = async () => {
    try {
      const storedToken = tokenStorage.getAccessToken();
      
      if (storedToken && !tokenStorage.isTokenExpired(storedToken)) {
        // Token is valid, verify with server
        await verifyToken(storedToken);
      } else {
        // Try to refresh token
        await refreshAccessToken();
      }
    } catch (error) {
      console.error('Auth initialization failed:', error);
      logout();
    } finally {
      setLoading(false);
    }
  };
  
  const verifyToken = async (token) => {
    const response = await fetch('/api/auth/verify', {
      headers: {
        'Authorization': `Bearer ${token}`
      },
      credentials: 'include'
    });
    
    if (response.ok) {
      const userData = await response.json();
      setUser(userData.user);
      setAccessToken(token);
      tokenStorage.setAccessToken(token);
    } else {
      throw new Error('Token verification failed');
    }
  };
  
  const refreshAccessToken = async () => {
    const response = await fetch('/api/auth/refresh', {
      method: 'POST',
      credentials: 'include' // Include refresh token cookie
    });
    
    if (response.ok) {
      const data = await response.json();
      setAccessToken(data.accessToken);
      setUser(data.user);
      tokenStorage.setAccessToken(data.accessToken);
      
      // Schedule next refresh
      scheduleTokenRefresh(data.accessToken);
    } else {
      throw new Error('Token refresh failed');
    }
  };
  
  const scheduleTokenRefresh = (token) => {
    const decoded = jwt.decode(token);
    const expiresIn = decoded.exp * 1000 - Date.now();
    const refreshTime = expiresIn - 60000; // Refresh 1 minute before expiry
    
    if (refreshTime > 0) {
      setTimeout(() => {
        refreshAccessToken().catch(() => logout());
      }, refreshTime);
    }
  };
  
  const login = async (credentials) => {
    const response = await fetch('/api/auth/login', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(credentials),
      credentials: 'include'
    });
    
    if (response.ok) {
      const data = await response.json();
      setUser(data.user);
      setAccessToken(data.accessToken);
      tokenStorage.setAccessToken(data.accessToken);
      scheduleTokenRefresh(data.accessToken);
    } else {
      const error = await response.json();
      throw new Error(error.message);
    }
  };
  
  const logout = () => {
    setUser(null);
    setAccessToken(null);
    tokenStorage.clearTokens();
  };
  
  const value = {
    user,
    accessToken,
    loading,
    login,
    logout,
    refreshAccessToken
  };
  
  return (
    <AuthContext.Provider value={value}>
      {children}
    </AuthContext.Provider>
  );
}

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth must be used within AuthProvider');
  }
  return context;
};
```

```javascript
    permissions: Object.values(PERMISSIONS)
  }

// Permission checking hook
function usePermissions() {
  const { user } = useAuth();
  
  const hasPermission = (permission) => {
    if (!user || !user.role) return false;
    
    const userRole = ROLES[user.role.toUpperCase()];
    return userRole?.permissions.includes(permission) || false;
  };
  
  const hasAnyPermission = (permissions) => {
    return permissions.some(permission => hasPermission(permission));
  };
  
  const hasAllPermissions = (permissions) => {
    return permissions.every(permission => hasPermission(permission));
  };
  
  return {
    hasPermission,
    hasAnyPermission,
    hasAllPermissions
  };
}

// Protected route component
function ProtectedRoute({ children, requiredPermission, fallback }) {
  const { hasPermission } = usePermissions();
  const { loading } = useAuth();
  
  if (loading) {
    return <LoadingSpinner />;
  }
  
  if (!hasPermission(requiredPermission)) {
    return fallback || <AccessDenied />;
  }
  
  return children;
}
```

```javascript
// Personal Identifiable Information (PII) protection
class PIIProtection {
  constructor() {
    this.sensitiveFields = [
      'ssn', 'socialSecurityNumber',
      'creditCard', 'bankAccount',
      'passport', 'driverLicense',
      'email', 'phone', 'address'
    ];
  }
  
  // Mask sensitive data for display
  maskSensitiveData(data, fieldName) {
    if (!data) return data;
    
    switch (fieldName) {
      case 'email':
        const [username, domain] = data.split('@');
        return `${username.slice(0, 2)}***@${domain}`;
        
      case 'phone':
        return data.replace(/(\d{3})(\d{3})(\d{4})/, '($1) ***-$3');
        
      case 'creditCard':
        return data.replace(/(\d{4})(\d{4})(\d{4})(\d{4})/, '**** **** **** $4');
        
      case 'ssn':
        return data.replace(/(\d{3})(\d{2})(\d{4})/, '***-**-$3');
        
      default:
        return data.length > 4 ? 
          `${data.slice(0, 2)}***${data.slice(-2)}` : 
          '***';
    }
  }
  
  // Remove PII from objects before logging
  sanitizeForLogging(obj) {
    const sanitized = { ...obj };
    
    const sanitizeRecursive = (item, path = '') => {
      if (typeof item === 'object' && item !== null) {
        for (const [key, value] of Object.entries(item)) {
          const fullPath = path ? `${path}.${key}` : key;
          
          if (this.isSensitiveField(key)) {
            item[key] = '[REDACTED]';
          } else if (typeof value === 'object') {
            sanitizeRecursive(value, fullPath);
          }
        }
      }
    };
    
    sanitizeRecursive(sanitized);
    return sanitized;
  }
  
  isSensitiveField(fieldName) {
    return this.sensitiveFields.some(sensitive => 
      fieldName.toLowerCase().includes(sensitive.toLowerCase())
    );
  }
  
  // Validate data before processing
  validateDataProcessing(data, purpose) {
    const allowedPurposes = [
      'authentication',
      'user-profile',
      'billing',
      'support'
    ];
    
    if (!allowedPurposes.includes(purpose)) {
      throw new Error('Unauthorized data processing purpose');
    }
    
    // Check for unnecessary PII collection
    const containsPII = Object.keys(data).some(key => 
      this.isSensitiveField(key)
    );
    
    if (containsPII && purpose === 'analytics') {
      console.warn('PII detected in analytics data');
      return this.sanitizeForLogging(data);
    }
    
    return data;
  }
}
```

// Data lifecycle management
class DataLifecycleManager {
  constructor() {
    this.retentionPolicies = {
      'user-session': 24 * 60 * 60 * 1000, // 24 hours
      'user-preferences': 365 * 24 * 60 * 60 * 1000, // 1 year
      'analytics': 90 * 24 * 60 * 60 * 1000, // 90 days
      'audit-logs': 7 * 365 * 24 * 60 * 60 * 1000 // 7 years
    };
  }
  
  storeDataWithRetention(key, data, category) {
    const retentionPeriod = this.retentionPolicies[category];
    if (!retentionPeriod) {
      throw new Error(`Unknown data category: ${category}`);
    }
    
    const dataWithMetadata = {
      data,
      category,
      createdAt: Date.now(),
      expiresAt: Date.now() + retentionPeriod
    };
    
    secureStorage.setItem(key, dataWithMetadata);
  }
  
  getDataIfValid(key) {
    const storedData = secureStorage.getItem(key);
    if (!storedData) return null;
    
    if (Date.now() > storedData.expiresAt) {
      // Data has expired, remove it
      secureStorage.removeItem(key);
      return null;
    }
    
    return storedData.data;
  }
  
  cleanupExpiredData() {
    // This should run periodically
    const allKeys = Object.keys(localStorage);
    
    allKeys.forEach(key => {
      try {
        const data = secureStorage.getItem(key);
        if (data && data.expiresAt && Date.now() > data.expiresAt) {
          secureStorage.removeItem(key);
          console.log(`Removed expired data: ${key}`);
        }
      } catch (error) {
        // Skip invalid data
      }
    });
  }
  
  // Minimize data collection
  collectMinimalData(fullData, purpose) {
    const dataMinimizationRules = {
      'authentication': ['email', 'password'],
      'user-profile': ['name', 'email', 'preferences'],
      'analytics': ['userId', 'action', 'timestamp'],
      'support': ['userId', 'issue', 'contactInfo']
    };
    
    const allowedFields = dataMinimizationRules[purpose];
    if (!allowedFields) {
      throw new Error(`Unknown purpose: ${purpose}`);
    }
    
    const minimalData = {};
    allowedFields.forEach(field => {
      if (fullData.hasOwnProperty(field)) {
        minimalData[field] = fullData[field];
      }
    });
    
    return minimalData;
  }
}

## 6. 面试技巧和建议

### 展示安全意识的要点
1. **Security-first mindset** - 安全优先思维
   - "I always consider security implications from the design phase"
   - "Security is not an afterthought but a fundamental requirement"

2. **Defense in depth** - 纵深防御
   - "I implement multiple layers of security controls"
   - "No single security measure is sufficient on its own"

3. **Principle of least privilege** - 最小权限原则
   - "Users should only have access to what they absolutely need"
   - "I regularly review and audit access permissions"

4. **Security by design** - 安全设计
   - "I integrate security considerations into the development lifecycle"
   - "Threat modeling helps identify potential vulnerabilities early"

### 常用表达句式
- "To mitigate this risk, I would implement..."
- "This approach provides protection against..."
- "The security implications of this decision include..."
- "I would recommend conducting a security review before..."
- "This vulnerability could be exploited by..."
- "The attack vector for this threat is..."
- "To ensure compliance with security standards..."
- "The security trade-offs we need to consider are..."

### 避免的误区
1. **过度自信** - 不要声称100%安全
2. **忽视用户体验** - 平衡安全性和可用性
3. **单一防护** - 避免依赖单一安全措施
4. **忽视更新** - 强调持续的安全维护
5. **缺乏实例** - 提供具体的实现案例
