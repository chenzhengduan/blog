// Custom hook for form management
function useForm(initialValues, validationSchema) {
  const [values, setValues] = useState(initialValues);
  const [errors, setErrors] = useState({});
  const [touched, setTouched] = useState({});
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  const validateField = useCallback((name, value) => {
    if (!validationSchema[name]) return null;
    
    const validator = validationSchema[name];
    if (validator.required && !value) {
      return validator.message || `${name} is required`;
    }
    
    if (validator.pattern && !validator.pattern.test(value)) {
      return validator.message || `Invalid ${name} format`;
    }
    
    if (validator.minLength && value.length < validator.minLength) {
      return `${name} must be at least ${validator.minLength} characters`;
    }
    
    return null;
  }, [validationSchema]);
  
  const handleChange = useCallback((name, value) => {
    setValues(prev => ({ ...prev, [name]: value }));
    
    // Validate on change if field was touched
    if (touched[name]) {
      const error = validateField(name, value);
      setErrors(prev => ({
        ...prev,
        [name]: error
      }));
    }
  }, [touched, validateField]);
  
  const handleBlur = useCallback((name) => {
    setTouched(prev => ({ ...prev, [name]: true }));
    
    // Validate on blur
    const error = validateField(name, values[name]);
    setErrors(prev => ({
      ...prev,
      [name]: error
    }));
  }, [values, validateField]);
  
  const handleSubmit = useCallback(async (onSubmit) => {
    // Mark all fields as touched
    const allTouched = Object.keys(values).reduce((acc, key) => {
      acc[key] = true;
      return acc;
    }, {});
    setTouched(allTouched);
    
    // Validate all fields
    const allErrors = {};
    Object.keys(values).forEach(name => {
      const error = validateField(name, values[name]);
      if (error) allErrors[name] = error;
    });
    
    setErrors(allErrors);
    
    if (Object.keys(allErrors).length > 0) {
      return;
    }
    
    setIsSubmitting(true);
    try {
      await onSubmit(values);
    } catch (error) {
      setErrors({ submit: error.message });
    } finally {
      setIsSubmitting(false);
    }
  }, [values, validateField]);
  
  const reset = useCallback(() => {
    setValues(initialValues);
    setErrors({});
    setTouched({});
    setIsSubmitting(false);
  }, [initialValues]);
  
  return {
    values,
    errors,
    touched,
    isSubmitting,
    handleChange,
    handleBlur,
    handleSubmit,
    reset,
    isValid: Object.keys(errors).length === 0
  };
}

// Custom hook for API data fetching with caching
function useApi(endpoint, options = {}) {
  const {
    dependencies = [],
    transform = (data) => data,
    enabled = true,
    cacheTime = 5 * 60 * 1000, // 5 minutes
    staleTime = 1 * 60 * 1000   // 1 minute
  } = options;
  
  const [state, setState] = useState({
    data: null,
    loading: true,
    error: null,
    lastFetch: null
  });
  
  const cache = useRef(new Map());
  const abortController = useRef(null);
  
  const fetchData = useCallback(async (force = false) => {
    const cacheKey = JSON.stringify({ endpoint, dependencies });
    
    // Check cache
    if (!force && cache.current.has(cacheKey)) {
      const cached = cache.current.get(cacheKey);
      const age = Date.now() - cached.timestamp;
      
      if (age < staleTime) {
        setState({
          data: cached.data,
          loading: false,
          error: null,
          lastFetch: cached.timestamp
        });
        return;
      }
    }
    
    // Cancel previous request
    if (abortController.current) {
      abortController.current.abort();
    }
    
    abortController.current = new AbortController();
    setState(prev => ({ ...prev, loading: true, error: null }));
    
    try {
      const response = await fetch(endpoint, {
        signal: abortController.current.signal,
        ...options.fetchOptions
      });
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      
      const rawData = await response.json();
      const data = transform(rawData);
      const timestamp = Date.now();
      
      // Update cache
      cache.current.set(cacheKey, { data, timestamp });
      
      // Clean old cache entries
      for (const [key, value] of cache.current.entries()) {
        if (timestamp - value.timestamp > cacheTime) {
          cache.current.delete(key);
        }
      }
      
      setState({
        data,
        loading: false,
        error: null,
        lastFetch: timestamp
      });
    } catch (error) {
      if (error.name !== 'AbortError') {
        setState({
          data: null,
          loading: false,
          error: error.message,
          lastFetch: null
        });
      }
    }
  }, [endpoint, JSON.stringify(dependencies), transform, staleTime, cacheTime]);
  
  useEffect(() => {
    if (enabled) {
      fetchData();
    }
    
    return () => {
      if (abortController.current) {
        abortController.current.abort();
      }
    };
  }, [fetchData, enabled]);
  
  const refetch = useCallback(() => fetchData(true), [fetchData]);
  
  return {
    ...state,
    refetch,
    isStale: state.lastFetch && (Date.now() - state.lastFetch) > staleTime
  };
}

// Composable for API data fetching
import { ref, computed, watch, onUnmounted } from 'vue';

export function useApi(url, options = {}) {
  const data = ref(null);
  const loading = ref(false);
  const error = ref(null);
  const abortController = ref(null);
  
  const {
    immediate = true,
    transform = (data) => data,
    onSuccess,
    onError
  } = options;
  
  const execute = async (customUrl) => {
    const requestUrl = customUrl || url.value || url;
    
    if (!requestUrl) return;
    
    // Cancel previous request
    if (abortController.value) {
      abortController.value.abort();
    }
    
    abortController.value = new AbortController();
    loading.value = true;
    error.value = null;
    
    try {
      const response = await fetch(requestUrl, {
        signal: abortController.value.signal,
        ...options.fetchOptions
      });
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      
      const result = await response.json();
      data.value = transform(result);
      
      if (onSuccess) {
        onSuccess(data.value);
      }
    } catch (err) {
      if (err.name !== 'AbortError') {
        error.value = err.message;
        if (onError) {
          onError(err);
        }
      }
    } finally {
      loading.value = false;
    }
  };
  
  // Watch URL changes
  if (typeof url === 'object' && url.value !== undefined) {
    watch(url, execute, { immediate });
  } else if (immediate) {
    execute();
  }
  
  // Cleanup on unmount
  onUnmounted(() => {
    if (abortController.value) {
      abortController.value.abort();
    }
  });
  
  return {
    data: readonly(data),
    loading: readonly(loading),
    error: readonly(error),
    execute,
    refresh: () => execute()
  };
}

// Composable for form validation
export function useForm(initialValues, validationRules) {
  const values = reactive({ ...initialValues });
  const errors = ref({});
  const touched = ref({});
  const isSubmitting = ref(false);
  
  const validateField = (field, value) => {
    const rules = validationRules[field];
    if (!rules) return null;
    
    for (const rule of rules) {
      if (rule.required && (!value || value.toString().trim() === '')) {
        return rule.message || `${field} is required`;
      }
      
      if (rule.minLength && value.length < rule.minLength) {
        return rule.message || `${field} must be at least ${rule.minLength} characters`;
      }
      
      if (rule.pattern && !rule.pattern.test(value)) {
        return rule.message || `Invalid ${field} format`;
      }
      
      if (rule.validator && !rule.validator(value, values)) {
        return rule.message || `${field} is invalid`;
      }
    }
    
    return null;
  };
  
  const validateForm = () => {
    const newErrors = {};
    
    Object.keys(validationRules).forEach(field => {
      const error = validateField(field, values[field]);
      if (error) {
        newErrors[field] = error;
      }
    });
    
    errors.value = newErrors;
    return Object.keys(newErrors).length === 0;
  };
  
  const handleChange = (field, value) => {
    values[field] = value;
    
    // Validate if field was touched
    if (touched.value[field]) {
      const error = validateField(field, value);
      if (error) {
        errors.value[field] = error;
      } else {
        delete errors.value[field];
      }
    }
  };
  
  const handleBlur = (field) => {
    touched.value[field] = true;
    const error = validateField(field, values[field]);
    if (error) {
      errors.value[field] = error;
    } else {
      delete errors.value[field];
    }
  };
  
  const handleSubmit = async (onSubmit) => {
    // Mark all fields as touched
    Object.keys(validationRules).forEach(field => {
      touched.value[field] = true;
    });
    
    if (!validateForm()) {
      return;
    }
    
    isSubmitting.value = true;
    
    try {
      await onSubmit(toRaw(values));
    } catch (error) {
      errors.value.submit = error.message;
    } finally {
      isSubmitting.value = false;
    }
  };
  
  const reset = () => {
    Object.assign(values, initialValues);
    errors.value = {};
    touched.value = {};
    isSubmitting.value = false;
  };
  
  const isValid = computed(() => Object.keys(errors.value).length === 0);
  
  return {
    values,
    errors: readonly(errors),
    touched: readonly(touched),
    isSubmitting: readonly(isSubmitting),
    isValid,
    handleChange,
    handleBlur,
    handleSubmit,
    reset,
    validateForm
  };
}

// Core HTTP service
import { Injectable } from '@angular/core';
import { HttpClient, HttpErrorResponse, HttpParams } from '@angular/common/http';
import { Observable, throwError, BehaviorSubject } from 'rxjs';
import { catchError, retry, map, finalize } from 'rxjs/operators';

export interface ApiResponse<T> {
  data: T;
  message: string;
  success: boolean;
}

export interface PaginatedResponse<T> extends ApiResponse<T[]> {
  pagination: {
    page: number;
    limit: number;
    total: number;
    totalPages: number;
  };
}

@Injectable({
  providedIn: 'root'
})
export class ApiService {
  private readonly baseUrl = '/api';
  private loadingSubject = new BehaviorSubject<boolean>(false);
  
  public loading$ = this.loadingSubject.asObservable();
  
  constructor(private http: HttpClient) {}
  
  private handleError(error: HttpErrorResponse): Observable<never> {
    let errorMessage = 'An unknown error occurred';
    
    if (error.error instanceof ErrorEvent) {
      // Client-side error
      errorMessage = error.error.message;
    } else {
      // Server-side error
      errorMessage = error.error?.message || `HTTP ${error.status}: ${error.statusText}`;
    }
    
    console.error('API Error:', errorMessage);
    return throwError(() => new Error(errorMessage));
  }
  
  private setLoading(loading: boolean): void {
    this.loadingSubject.next(loading);
  }
  
  get<T>(endpoint: string, params?: any): Observable<ApiResponse<T>> {
    this.setLoading(true);
    
    let httpParams = new HttpParams();
    if (params) {
      Object.keys(params).forEach(key => {
        if (params[key] !== null && params[key] !== undefined) {
          httpParams = httpParams.set(key, params[key].toString());
        }
      });
    }
    
    return this.http.get<ApiResponse<T>>(`${this.baseUrl}${endpoint}`, { params: httpParams })
      .pipe(
        retry(2),
        catchError(this.handleError),
        finalize(() => this.setLoading(false))
      );
  }
  
  post<T>(endpoint: string, data: any): Observable<ApiResponse<T>> {
    this.setLoading(true);
    
    return this.http.post<ApiResponse<T>>(`${this.baseUrl}${endpoint}`, data)
      .pipe(
        catchError(this.handleError),
        finalize(() => this.setLoading(false))
      );
  }
  
  put<T>(endpoint: string, data: any): Observable<ApiResponse<T>> {
    this.setLoading(true);
    
    return this.http.put<ApiResponse<T>>(`${this.baseUrl}${endpoint}`, data)
      .pipe(
        catchError(this.handleError),
        finalize(() => this.setLoading(false))
      );
  }
  
  delete<T>(endpoint: string): Observable<ApiResponse<T>> {
    this.setLoading(true);
    
    return this.http.delete<ApiResponse<T>>(`${this.baseUrl}${endpoint}`)
      .pipe(
        catchError(this.handleError),
        finalize(() => this.setLoading(false))
      );
  }
  
  getPaginated<T>(endpoint: string, page: number = 1, limit: number = 10, filters?: any): Observable<PaginatedResponse<T>> {
    const params = { page: page.toString(), limit: limit.toString(), ...filters };
    return this.get<T[]>(endpoint, params) as Observable<PaginatedResponse<T>>;
  }
}

    const token = localStorage.getItem('authToken');

          this.currentUserSubject.next(response.data);
        },
        error: (error) => {

          this.logout();
        }
      });
    }
  }

    return this.apiService.get<User>('/auth/me');
  }

          this.usersSubject.next(response.data);
          return response;
        })
      );

    return this.apiService.get<User>(`/users/${id}`);

    return this.apiService.post<User>('/users', userData)
      .pipe(
        map(response => {
          if (response.success) {
            const currentUsers = this.usersSubject.value;
            this.usersSubject.next([...currentUsers, response.data]);
            this.notificationService.showSuccess('User created successfully');
          }
          return response;
        })
      );
  }
  
  updateUser(id: string, userData: UpdateUserRequest): Observable<ApiResponse<User>> {
    return this.apiService.put<User>(`/users/${id}`, userData)
      .pipe(
        map(response => {
          if (response.success) {
            const currentUsers = this.usersSubject.value;
            const updatedUsers = currentUsers.map(user => 
              user.id === id ? response.data : user
            );
            this.usersSubject.next(updatedUsers);
            
            // Update current user if it's the same user
            const currentUser = this.currentUserSubject.value;
            if (currentUser && currentUser.id === id) {
              this.currentUserSubject.next(response.data);
            }
            
            this.notificationService.showSuccess('User updated successfully');
          }
          return response;
        })
      );
  }
  
  deleteUser(id: string): Observable<ApiResponse<any>> {
    return this.apiService.delete(`/users/${id}`)
      .pipe(
        map(response => {
          if (response.success) {
            const currentUsers = this.usersSubject.value;
            const filteredUsers = currentUsers.filter(user => user.id !== id);
            this.usersSubject.next(filteredUsers);
            this.notificationService.showSuccess('User deleted successfully');
          }
          return response;
        })
      );
  }
  
  login(email: string, password: string): Observable<ApiResponse<{ user: User; token: string }>> {
    return this.apiService.post<{ user: User; token: string }>('/auth/login', { email, password })
      .pipe(
        map(response => {
          if (response.success) {
            localStorage.setItem('authToken', response.data.token);
            this.currentUserSubject.next(response.data.user);
            this.notificationService.showSuccess('Login successful');
          }
          return response;
        })
      );
  }
  
  logout(): void {
    localStorage.removeItem('authToken');
    this.currentUserSubject.next(null);
    this.usersSubject.next([]);
    this.notificationService.showInfo('Logged out successfully');
  }
  
  isAuthenticated(): boolean {
    return !!this.currentUserSubject.value;
  }
  
  hasRole(role: string): boolean {
    const currentUser = this.currentUserSubject.value;
    return currentUser ? currentUser.role === role : false;
  }
}

  }

## 5. 面试技巧与建议 - Interview Tips and Suggestions

### 展示框架掌握程度的要点

1. **深度理解核心概念 (Deep Understanding of Core Concepts)**
   - 解释框架的核心原理和设计思想
   - 讨论框架解决的具体问题
   - 对比不同框架的优缺点

2. **实际项目经验 (Real Project Experience)**
   - 分享具体的项目实施经验
   - 讨论遇到的挑战和解决方案
   - 展示性能优化和最佳实践

3. **生态系统了解 (Ecosystem Knowledge)**
   - 熟悉相关工具和库
   - 了解社区最佳实践
   - 跟上技术发展趋势

### 常用表达句式

**描述框架特性：**
- "React's virtual DOM provides efficient rendering by..."
- "Vue's reactivity system automatically tracks dependencies..."
- "Angular's dependency injection enables loose coupling..."
- "The component-based architecture promotes reusability..."

**解释技术选型：**
- "I chose React because it offers better performance for..."
- "Vue was the right choice for this project due to..."
- "Angular's enterprise features made it suitable for..."
- "The learning curve and team expertise influenced our decision..."

**讨论最佳实践：**
- "I follow the principle of keeping components small and focused..."
- "State management should be centralized for complex applications..."
- "I implement proper error boundaries to handle failures gracefully..."
- "Performance optimization includes memoization and lazy loading..."

**分析性能影响：**
- "This approach reduces bundle size by implementing code splitting..."
- "The virtual scrolling technique improves performance for large lists..."
- "Memoization prevents unnecessary re-renders in React components..."
- "Server-side rendering improves initial page load time..."

### 避免的误区

1. **过度比较框架 (Over-comparing Frameworks)**
   - 避免贬低其他框架
   - 客观分析各框架的适用场景
   - 承认每个框架都有其优势

2. **忽视基础知识 (Ignoring Fundamentals)**
   - 不要只关注框架特性
   - 理解底层JavaScript原理
   - 掌握Web标准和浏览器API

3. **缺乏实践经验 (Lack of Practical Experience)**
   - 避免只有理论知识
   - 准备具体的代码示例
   - 分享真实的项目经验

4. **忽视性能考虑 (Ignoring Performance Considerations)**
   - 考虑框架对性能的影响
   - 了解优化策略和工具
   - 关注用户体验指标

### 深入讨论的话题

1. **框架演进 (Framework Evolution)**
   - 框架版本升级策略
   - 新特性的采用和迁移
   - 向后兼容性考虑

2. **团队协作 (Team Collaboration)**
   - 代码规范和最佳实践
   - 组件库和设计系统
   - 知识分享和培训

3. **性能优化 (Performance Optimization)**
   - 渲染性能优化策略
   - 打包和加载优化
   - 监控和分析工具

4. **测试策略 (Testing Strategy)**
   - 单元测试和集成测试
   - 组件测试最佳实践
   - 端到端测试工具
