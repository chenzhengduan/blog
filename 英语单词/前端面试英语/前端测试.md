// test-utils.js
import React from 'react';
import { render } from '@testing-library/react';
import { BrowserRouter } from 'react-router-dom';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { UserContext } from '../contexts/UserContext';
import { ThemeProvider } from '../contexts/ThemeContext';

// Create a custom render function that includes providers
export function renderWithProviders(
  ui,
  {
    preloadedState = {},
    route = '/',
    user = null,
    theme = 'light',
    ...renderOptions
  } = {}
) {
  const queryClient = new QueryClient({
    defaultOptions: {
      queries: {
        retry: false,
      },
    },
  });
  
  const userContextValue = {
    user,
    updateUser: jest.fn(),
    loading: false,
    ...preloadedState.user
  };
  
  const themeContextValue = {
    theme,
    toggleTheme: jest.fn(),
    ...preloadedState.theme
  };
  
  function Wrapper({ children }) {
    return (
      <BrowserRouter>
        <QueryClientProvider client={queryClient}>
          <ThemeProvider value={themeContextValue}>
            <UserContext.Provider value={userContextValue}>
              {children}
            </UserContext.Provider>
          </ThemeProvider>
        </QueryClientProvider>
      </BrowserRouter>
    );
  }
  
  // Set initial route
  window.history.pushState({}, 'Test page', route);
  
  return {
    ...render(ui, { wrapper: Wrapper, ...renderOptions }),
    queryClient,
    userContextValue,
    themeContextValue
  };
}

// Custom matchers
export const customMatchers = {
  toBeAccessible: (received) => {
    // Check for basic accessibility requirements
    const hasAriaLabel = received.hasAttribute('aria-label') || 
                        received.hasAttribute('aria-labelledby');
    const hasRole = received.hasAttribute('role');
    const isInteractive = ['button', 'link', 'input'].includes(received.tagName.toLowerCase());
    
    if (isInteractive && !hasAriaLabel && !hasRole) {
      return {
        message: () => `Expected element to have aria-label, aria-labelledby, or role attribute`,
        pass: false
      };
    }
    
    return {
      message: () => `Expected element to not be accessible`,
      pass: true
    };
  }
};

    fireEvent.change(searchInput, { target: { value: 'a' } });
    fireEvent.change(searchInput, { target: { value: 'ap' } });
    fireEvent.change(searchInput, { target: { value: 'app' } });
    
    // Should not call onSearch immediately
    expect(mockOnSearch).not.toHaveBeenCalled();
    
    // Wait for debounce delay
    await waitFor(() => {
      expect(mockOnSearch).toHaveBeenCalledWith('app');
    }, { timeout: 1000 });
    
    // Should only be called once after debounce
    expect(mockOnSearch).toHaveBeenCalledTimes(1);
  });
});

### 避免的误区 - Common Pitfalls to Avoid

1. **过度测试** - Over-testing
   - 不要说："I test everything to achieve 100% coverage"
   - 应该说："I focus on testing critical paths and high-risk areas effectively"

2. **忽视测试维护** - Ignoring test maintenance
   - 不要说："Once tests are written, they don't need much attention"
   - 应该说："I regularly review and update tests to keep them relevant and reliable"

3. **只关注工具** - Focusing only on tools
   - 不要说："I use Cypress because it's popular"
   - 应该说："I chose Cypress because it provides reliable browser automation for our E2E testing needs"

4. **忽视性能影响** - Ignoring performance impact
   - 不要说："Test speed doesn't matter as long as they pass"
   - 应该说："I optimize test performance to maintain fast feedback loops in our development process"
