# 对比排课优化方案 - 复制组件方案详细分析

## 📋 背景

### 当前问题
1. **性能问题**: 20学员 + 20老师 = 40次API调用
2. **本迭代需求**: 对比视图需要支持时段切换（上午/下午/晚上）
3. **耦合严重**: `TimeCanlendarView.vue` 中有大量 `isCompare` 判断（20+ 处）

### 迭代规划
- ✅ **本迭代 (你负责)**:
  - 对比视图支持时段/时刻切换
  - 请求优化 (40次 → 2次)
  - 复制组件 + 清理代码
- ⏸️ **等待依赖**: 范围模式组件（别人开发中）
- ⏭️ **下个迭代**: 对比视图支持范围模式

### 对比视图的特殊需求
- ✅ 永远是**指定对象模式** (assign=1)
- ✅ 需要支持**时刻/时段/范围**三种模式切换（范围模式下个迭代）
- ✅ 永远是**周视图** (不需要 月/季度 等)
- ✅ 数据由**父组件统一加载**后传入
- 🔜 本迭代先支持**时刻和时段**，范围模式等组件开发完再接入

---

## 🎯 方案决策：复制一份组件

### ✅ 优点
1. **彻底解耦**: 改造互不影响，降低回归测试风险
2. **代码简化**: 移除大量条件判断，逻辑更清晰
3. **独立演进**: 时段切换等新功能不污染主视图
4. **性能优化**: 统一请求40次变2次

### ⚠️ 挑战
1. **代码重复**: ~4500行组件需要复制
2. **同步修复**: 公共Bug需要两处修改
3. **初期成本**: 清理、改造、测试工作量大

### 💡 解决方案
- **提取 Composables**: 公共逻辑只维护一份
- **明确分工**: 核心逻辑在 Composables，界面逻辑在组件

#### 2.1 compareCanlendar.vue 批量加载
```typescript
// 学员数据管理
const studentsDataMap = ref<Map<string, any>>(new Map())
const studentsLoading = ref(false)

// 老师数据管理
const teachersDataMap = ref<Map<string, any>>(new Map())
const teachersLoading = ref(false)

// 批量加载学员（1次API）
async function loadAllStudentsData() {
  if (selectedStudents.value.length === 0) return
  
  studentsLoading.value = true
  try {
    const startOfWeek = dayjs(internalSearchParams.value.currentWeek).startOf('week').format('YYYY-MM-DD')
    const endOfWeek = dayjs(internalSearchParams.value.currentWeek).endOf('week').format('YYYY-MM-DD')
    
    const res = await queryCalendarStudent({
      IDList: selectedStudents.value.map(s => s.ID),  // 所有学员ID
      StartDate: startOfWeek,
      EndDate: endOfWeek,
      PageSize: 99999,
      PageIndex: 1,
    })
    
    // 按学员ID分发
    studentsDataMap.value.clear()
    res.Data.forEach((studentData: any) => {
      studentsDataMap.value.set(studentData.StudentID, studentData)
    })
  } catch (error) {
    ElMessage.error('加载学员课表失败')
  } finally {
    studentsLoading.value = false
  }
}

// 批量加载老师（1次API）
async function loadAllTeachersData() {
  if (selectedTeachers.value.length === 0) return
  
  teachersLoading.value = true
  try {
    const startOfWeek = dayjs(internalSearchParams.value.currentWeek).startOf('week').format('YYYY-MM-DD')
    const endOfWeek = dayjs(internalSearchParams.value.currentWeek).endOf('week').format('YYYY-MM-DD')
    
    const res = await queryCalendarTeacher({
      IDList: selectedTeachers.value.map(t => t.ID),  // 所有老师ID
      StartDate: startOfWeek,
      EndDate: endOfWeek,
      PageSize: 99999,
      PageIndex: 1,
    })
    
    // 按老师ID分发
    teachersDataMap.value.clear()
    res.Data.forEach((teacherData: any) => {
      teachersDataMap.value.set(teacherData.TeacherID, teacherData)
    })
  } catch (error) {
    ElMessage.error('加载老师课表失败')
  } finally {
    teachersLoading.value = false
  }
}
```

```typescript
// composables/useTimeCalendarCore.ts
export function useTimeCalendarCore() {
  // 计算课程在网格中的位置
  const calculateCoursePosition = (course: any, rowHeight: number) => {
    const [startHour, startMinute] = course.StartTime.split(':').map(Number)
    const [endHour, endMinute] = course.EndTime.split(':').map(Number)
    
    const startDecimal = startHour + startMinute / 60
    const endDecimal = endHour + endMinute / 60
    
    const top = (startDecimal - 6) * rowHeight  // 假设从6点开始
    const height = (endDecimal - startDecimal) * rowHeight
    
    return { top, height }
  }
  
  // 获取星期几
  const getDayOfWeek = (date: string, startOfWeek: string) => {
    const daysDiff = dayjs(date).diff(dayjs(startOfWeek), 'day')
    return daysDiff
  }
  
  // 格式化时间范围
  const formatTimeRange = (startTime: string, endTime: string) => {
    return `${startTime}-${endTime}`
  }
  
  return {
    calculateCoursePosition,
    getDayOfWeek,
    formatTimeRange
  }
}

// composables/useHolidayData.ts
export function useHolidayData(currentWeek: Ref<string>) {
  const holidays = ref([])
  
  const loadHolidays = async () => {
    const startOfWeek = dayjs(currentWeek.value).startOf('week').format('YYYY-MM-DD')
    const endOfWeek = dayjs(currentWeek.value).endOf('week').format('YYYY-MM-DD')
    
    try {
      const res = await queryHoliday({ sdate: startOfWeek, edate: endOfWeek })
      holidays.value = (res?.Data?.Data || []).filter((h: any) => {
        const k = dayjs(h.Date).format('YYYY-MM-DD')
        return k && k !== 'Invalid Date'
      })
    } catch (error) {
      console.error('加载节假日失败:', error)
    }
  }
  
  const isHoliday = (date: string) => {
    return holidays.value.some((h: any) => 
      dayjs(h.Date).format('YYYY-MM-DD') === date
    )
  }
  
  return { holidays, loadHolidays, isHoliday }
}

// composables/useCourseConflict.ts
export function useCourseConflict() {
  const checkConflict = (course1: any, course2: any) => {
    if (course1.Date !== course2.Date) return false
    
    const start1 = dayjs(`${course1.Date} ${course1.StartTime}`)
    const end1 = dayjs(`${course1.Date} ${course1.EndTime}`)
    const start2 = dayjs(`${course2.Date} ${course2.StartTime}`)
    const end2 = dayjs(`${course2.Date} ${course2.EndTime}`)
    
    return start1.isBefore(end2) && start2.isBefore(end1)
  }
  
  return { checkConflict }
}
```

## 🚨 风险与注意事项

### 风险1: 时段切换与范围模式冲突
