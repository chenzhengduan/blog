##### 数据加载重构

**1. 新增外部数据处理函数**
```typescript
function processExternalData() {
  if (!props.externalData) {
    internalCourses.value = []
    loading.value = false
    return
  }

  loading.value = true
  try {
    const startOfWeek = dayjs(props.currentWeek).startOf('week').format('YYYY-MM-DD')
    
    if (props.externalData.CourseList && Array.isArray(props.externalData.CourseList)) {
      // 转换外部数据为内部格式
      internalCourses.value = props.externalData.CourseList.map(item => ({
        ...item,
        day: getDayOfWeek(
          dayjs(item.StartTime).format('YYYY-MM-DD'),
          startOfWeek
        ),
        StartTime: dayjs(item.StartTime).format('HH:mm'),
        EndTime: dayjs(item.EndTime).format('HH:mm'),
        Date: dayjs(item.StartTime).format('YYYY-MM-DD'),
        type: 'course'
      }))
    } else {
      internalCourses.value = []
    }
  } finally {
    loading.value = false
  }
}
```

**学员批量加载**
```typescript
// 🚀 批量加载学员数据（1次API调用获取所有学员数据）
const loadAllStudentsData = async () => {
  if (selectedStudents.value.length === 0) {
    studentsDataMap.value.clear()
    return
  }

  batchLoadingStudents.value = true
  try {
    const startDate = dayjs(currentWeek.value).startOf('week').format('YYYY-MM-DD')
    const endDate = dayjs(currentWeek.value).endOf('week').format('YYYY-MM-DD')

    // 🔥 关键：IDList 传入所有学员ID，一次性获取
    const res = await queryCalendarStudent({
      IDList: selectedStudents.value.map(s => s.ID),  // [1, 2, 3, ..., 20]
      StartDate: startDate,
      EndDate: endDate,
      CampusIDList: internalSearchParams.value.CampusIDList,
      PageSize: 99999
    })

    // 清空旧数据
    studentsDataMap.value.clear()

    // 按 StudentID 分组数据
    if (res && Array.isArray(res)) {
      res.forEach(item => {
        const studentId = item.StudentID
        if (!studentsDataMap.value.has(studentId)) {
          studentsDataMap.value.set(studentId, { CourseList: [] })
        }
        studentsDataMap.value.get(studentId).CourseList.push(item)
      })
    }

    console.log('✅ 批量加载学员数据完成:', studentsDataMap.value.size, '个学员')
  } catch (error) {
    console.error('❌ 批量加载学员数据失败:', error)
    ElMessage.error('加载学员数据失败')
  } finally {
    batchLoadingStudents.value = false
  }
}
```
