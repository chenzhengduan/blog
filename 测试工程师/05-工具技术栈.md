### 1. Web自动化测试

#### 1.1 Selenium
**简介**：最流行的Web自动化测试框架

**技术栈**：
- **Selenium WebDriver**: 核心驱动
- **Selenium Grid**: 分布式执行
- **Selenium IDE**: 录制回放工具

**支持语言**：Java、Python、C#、JavaScript、Ruby

**实战示例**：
```python
# Python + Selenium 示例
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

class WebTestFramework:
    def __init__(self, browser="chrome"):
        if browser == "chrome":
            self.driver = webdriver.Chrome()
        elif browser == "firefox":
            self.driver = webdriver.Firefox()
        
        self.wait = WebDriverWait(self.driver, 10)
    
    def open_page(self, url):
        self.driver.get(url)
    
    def find_element(self, locator):
        return self.wait.until(EC.presence_of_element_located(locator))
    
    def click_element(self, locator):
        element = self.find_element(locator)
        element.click()
    
    def input_text(self, locator, text):
        element = self.find_element(locator)
        element.clear()
        element.send_keys(text)
    
    def get_text(self, locator):
        element = self.find_element(locator)
        return element.text
    
    def close(self):
        self.driver.quit()

# 使用示例
test = WebTestFramework()
test.open_page("https://example.com/login")
test.input_text((By.ID, "username"), "testuser")
test.input_text((By.ID, "password"), "password123")
test.click_element((By.ID, "login-btn"))
assert "Dashboard" in test.driver.title
test.close()
```

**最佳实践**：
```python
# 页面对象模式 (POM) 示例
class LoginPage:
    def __init__(self, driver):
        self.driver = driver
        self.username_input = (By.ID, "username")
        self.password_input = (By.ID, "password")
        self.login_button = (By.ID, "login-btn")
        self.error_message = (By.CLASS_NAME, "error")
    
    def login(self, username, password):
        self.driver.find_element(*self.username_input).send_keys(username)
        self.driver.find_element(*self.password_input).send_keys(password)
        self.driver.find_element(*self.login_button).click()
    
    def get_error_message(self):
        return self.driver.find_element(*self.error_message).text

# 测试用例
def test_invalid_login():
    login_page = LoginPage(driver)
    login_page.login("invalid", "invalid")
    assert "Invalid credentials" in login_page.get_error_message()
```

```scala
import io.gatling.core.Predef._
import io.gatling.http.Predef._
import scala.concurrent.duration._

class APILoadTest extends Simulation {
  
  val httpProtocol = http
    .baseUrl("https://api.example.com")
    .acceptHeader("application/json")
    .contentTypeHeader("application/json")
  
  val scn = scenario("API Load Test")
    .exec(
      http("Login")
        .post("/login")
        .body(StringBody("""{"username":"test","password":"123"}"""))
        .check(status.is(200))
        .check(jsonPath("$.token").saveAs("authToken"))
    )
    .pause(1)
    .exec(
      http("Get User Data")
        .get("/users")
        .header("Authorization", "Bearer ${authToken}")
        .check(status.is(200))
    )
  
  setUp(
    scn.inject(
      rampUsers(100) during (60 seconds)
    )
  ).protocols(httpProtocol)
}
```

```javascript
// K6 测试脚本示例
import http from 'k6/http';
import { check, sleep } from 'k6';

export let options = {
  stages: [
    { duration: '2m', target: 100 }, // 2分钟内增加到100用户
    { duration: '5m', target: 100 }, // 保持100用户5分钟
    { duration: '2m', target: 0 },   // 2分钟内减少到0用户
  ],
};

export default function() {
  // 登录请求
  let loginResponse = http.post('https://api.example.com/login', {
    username: 'testuser',
    password: 'password123'
  });
  
  check(loginResponse, {
    'login status is 200': (r) => r.status === 200,
    'login response time < 500ms': (r) => r.timings.duration < 500,
  });
  
  let authToken = loginResponse.json('token');
  
  // 业务请求
  let headers = { 'Authorization': `Bearer ${authToken}` };
  let response = http.get('https://api.example.com/users', { headers });
  
  check(response, {
    'status is 200': (r) => r.status === 200,
    'response time < 200ms': (r) => r.timings.duration < 200,
  });
  
  sleep(1);
}
```
