#### 挑战2：并发测试问题
**问题描述**：
- 多用户同时下单时库存计算错误
- 支付过程中的并发问题
- 数据库锁竞争

**解决方案**：
```python
# 并发测试脚本
import threading
import time
import requests
from concurrent.futures import ThreadPoolExecutor

class ConcurrencyTest:
    
    def __init__(self, base_url, num_threads=10):
        self.base_url = base_url
        self.num_threads = num_threads
        self.results = []
        self.lock = threading.Lock()
    
    def concurrent_order_test(self):
        """并发下单测试"""
        
        def place_order(thread_id):
            """单个线程下单操作"""
            try:
                # 获取认证token
                token = self.get_auth_token(f"user_{thread_id}")
                
                # 创建订单
                order_data = {
                    "items": [
                        {
                            "product_id": 1001,  # 同一个商品
                            "quantity": 1,
                            "price": 99.99
                        }
                    ],
                    "shipping_address": {
                        "street": f"Address {thread_id}",
                        "city": "Test City",
                        "postal_code": "12345"
                    }
                }
                
                headers = {
                    "Authorization": f"Bearer {token}",
                    "Content-Type": "application/json"
                }
                
                start_time = time.time()
                response = requests.post(
                    f"{self.base_url}/orders",
                    json=order_data,
                    headers=headers
                )
                end_time = time.time()
                
                with self.lock:
                    self.results.append({
                        "thread_id": thread_id,
                        "status_code": response.status_code,
                        "response_time": end_time - start_time,
                        "response_data": response.json() if response.status_code == 201 else None,
                        "error": response.text if response.status_code != 201 else None
                    })
                    
            except Exception as e:
                with self.lock:
                    self.results.append({
                        "thread_id": thread_id,
                        "error": str(e)
                    })
        
        # 使用线程池执行并发测试
        with ThreadPoolExecutor(max_workers=self.num_threads) as executor:
            futures = [
                executor.submit(place_order, i) 
                for i in range(self.num_threads)
            ]
            
            # 等待所有线程完成
            for future in futures:
                future.result()
        
        return self.analyze_results()
    
    def analyze_results(self):
        """分析测试结果"""
        successful_orders = [r for r in self.results if r.get("status_code") == 201]
        failed_orders = [r for r in self.results if r.get("status_code") != 201]
        
        analysis = {
            "total_requests": len(self.results),
            "successful_orders": len(successful_orders),
            "failed_orders": len(failed_orders),
            "success_rate": len(successful_orders) / len(self.results) * 100,
            "average_response_time": sum(r.get("response_time", 0) for r in successful_orders) / len(successful_orders) if successful_orders else 0,
            "errors": [r.get("error") for r in failed_orders if r.get("error")]
        }
        
        return analysis

# 使用示例
if __name__ == "__main__":
    test = ConcurrencyTest("https://api.example-ecommerce.com", num_threads=20)
    results = test.concurrent_order_test()
    
    print(f"并发测试结果:")
    print(f"总请求数: {results['total_requests']}")
    print(f"成功订单: {results['successful_orders']}")
    print(f"失败订单: {results['failed_orders']}")
    print(f"成功率: {results['success_rate']:.2f}%")
    print(f"平均响应时间: {results['average_response_time']:.3f}秒")

# 数据一致性测试框架
class DataConsistencyTest:
    
    def __init__(self, db_config):
        self.db = DatabaseConnection(db_config)
        self.tolerance = 0.01  # 金额误差容忍度
    
    def test_account_balance_consistency(self):
        """测试账户余额一致性"""
        
        # 获取所有账户
        accounts = self.db.query("SELECT account_id, balance FROM accounts")
        
        for account in accounts:
            account_id = account['account_id']
            current_balance = account['balance']
            
            # 计算交易记录的余额
            transactions = self.db.query("""
                SELECT transaction_type, amount 
                FROM transactions 
                WHERE account_id = %s 
                ORDER BY created_at
            """, (account_id,))
            
            calculated_balance = 0
            for trans in transactions:
                if trans['transaction_type'] == 'credit':
                    calculated_balance += trans['amount']
                elif trans['transaction_type'] == 'debit':
                    calculated_balance -= trans['amount']
            
            # 验证余额一致性
            difference = abs(current_balance - calculated_balance)
            assert difference <= self.tolerance, \
                f"账户 {account_id} 余额不一致: 当前余额 {current_balance}, 计算余额 {calculated_balance}"
    
    def test_transaction_double_entry(self):
        """测试复式记账一致性"""
        
        # 获取所有转账交易
        transfers = self.db.query("""
            SELECT transfer_id, from_account, to_account, amount 
            FROM transfers 
            WHERE status = 'completed'
        """)
        
        for transfer in transfers:
            transfer_id = transfer['transfer_id']
            from_account = transfer['from_account']
            to_account = transfer['to_account']
            amount = transfer['amount']
            
            # 检查借方记录
            debit_record = self.db.query("""
                SELECT amount FROM transactions 
                WHERE account_id = %s AND reference_id = %s AND transaction_type = 'debit'
            """, (from_account, transfer_id))
            
            # 检查贷方记录
            credit_record = self.db.query("""
                SELECT amount FROM transactions 
                WHERE account_id = %s AND reference_id = %s AND transaction_type = 'credit'
            """, (to_account, transfer_id))
            
            assert len(debit_record) == 1, f"转账 {transfer_id} 缺少借方记录"
            assert len(credit_record) == 1, f"转账 {transfer_id} 缺少贷方记录"
            assert debit_record[0]['amount'] == amount, f"转账 {transfer_id} 借方金额不匹配"
            assert credit_record[0]['amount'] == amount, f"转账 {transfer_id} 贷方金额不匹配"
```

# 安全性测试
class SecurityTest:
    
    def test_sql_injection_protection(self):
        """测试SQL注入防护"""
        
        malicious_inputs = [
            "'; DROP TABLE accounts; --",
            "' OR '1'='1",
            "'; UPDATE accounts SET balance = 999999 WHERE account_id = '1'; --",
            "' UNION SELECT password FROM users --"
        ]
        
        for malicious_input in malicious_inputs:
            response = requests.post(
                "https://api.bank.com/accounts/search",
                json={"account_name": malicious_input},
                headers=self.get_auth_headers()
            )
            
            # 验证系统没有被SQL注入攻击
            assert response.status_code != 200 or "error" in response.json()
            
            # 验证数据库没有被修改
            account_count = self.db.query("SELECT COUNT(*) FROM accounts")[0]['count']
            assert account_count == self.expected_account_count
    
    def test_authorization_control(self):
        """测试权限控制"""
        
        # 普通用户尝试访问管理员功能
        user_token = self.get_user_token("regular_user")
        
        admin_endpoints = [
            "/admin/users",
            "/admin/transactions/all",
            "/admin/system/config"
        ]
        
        for endpoint in admin_endpoints:
            response = requests.get(
                f"https://api.bank.com{endpoint}",
                headers={"Authorization": f"Bearer {user_token}"}
            )
            
            assert response.status_code == 403, \
                f"普通用户不应该能访问管理员接口: {endpoint}"
    
    def test_sensitive_data_encryption(self):
        """测试敏感数据加密"""
        
        # 检查数据库中的敏感数据是否加密
        users = self.db.query("SELECT password, id_number, phone FROM users LIMIT 10")
        
        for user in users:
            # 密码应该是哈希值，不是明文
            assert len(user['password']) >= 60, "密码应该是哈希值"

        for service in expected_services:
            assert service in actual_services, f"缺少服务: {service}"
        
        # 验证链路性能
        total_duration = trace_data["spans"][0]["duration"]
        assert total_duration < 5000000, f"请求耗时过长: {total_duration/1000}ms"  # 5秒
    
    def get_trace_from_jaeger(self, trace_id):
        """从Jaeger获取链路数据"""
        jaeger_url = f"http://jaeger:16686/api/traces/{trace_id}"
        response = requests.get(jaeger_url)
        return response.json()["data"][0]
