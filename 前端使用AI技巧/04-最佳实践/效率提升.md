// 模块化开发助手
class ModularDevelopmentAssistant {
  private modules: Map<string, ModuleDefinition> = new Map()
  private dependencies: Map<string, string[]> = new Map()
  
  // 定义模块
  defineModule(module: ModuleDefinition): void {
    this.modules.set(module.name, module)
    this.dependencies.set(module.name, module.dependencies || [])
  }
  
  // 生成模块开发计划
  generateDevelopmentPlan(): DevelopmentPlan {
    const sortedModules = this.topologicalSort()
    const plan: DevelopmentPlan = {
      phases: [],
      totalEstimate: 0,
      parallelTasks: []
    }
    
    // 按依赖关系分组
    const phases = this.groupByDependencyLevel(sortedModules)
    
    phases.forEach((moduleNames, level) => {
      const phaseModules = moduleNames.map(name => this.modules.get(name)!)
      const phaseEstimate = phaseModules.reduce((sum, mod) => sum + mod.estimatedHours, 0)
      
      plan.phases.push({
        level,
        modules: phaseModules,
        estimatedHours: phaseEstimate,
        canRunInParallel: moduleNames.length > 1
      })
      
      plan.totalEstimate += phaseEstimate
    })
    
    return plan
  }
  
  // 生成模块AI提示词
  generateModulePrompt(moduleName: string): string {
    const module = this.modules.get(moduleName)
    if (!module) throw new Error(`Module ${moduleName} not found`)
    
    const dependencies = this.dependencies.get(moduleName) || []
    const dependencyModules = dependencies.map(dep => this.modules.get(dep)!)
    
    return `
# ${module.name}模块开发

## 模块信息
- 模块名称: ${module.name}
- 功能描述: ${module.description}
- 预估工时: ${module.estimatedHours}小时
- 复杂度: ${module.complexity}

## 依赖模块
${dependencyModules.map(dep => `- ${dep.name}: ${dep.description}`).join('\n')}

## 技术要求
- 技术栈: ${module.techStack.join(', ')}
- 设计模式: ${module.patterns.join(', ')}
- 性能要求: ${module.performance}

## 接口定义
\`\`\`typescript
${module.interfaces}
\`\`\`

## 实现要求
${module.requirements.map(req => `- ${req}`).join('\n')}

## 测试要求
- 单元测试覆盖率: 80%+
- 集成测试: 核心功能
- 性能测试: 关键路径

请生成完整的模块实现，包括:
1. 核心功能代码
2. 类型定义
3. 单元测试
4. 使用文档
    `
  }
  
  private topologicalSort(): string[] {
    const visited = new Set<string>()
    const result: string[] = []
    
    const visit = (moduleName: string) => {
      if (visited.has(moduleName)) return
      visited.add(moduleName)
      
      const deps = this.dependencies.get(moduleName) || []
      deps.forEach(dep => visit(dep))
      
      result.push(moduleName)
    }
    
    Array.from(this.modules.keys()).forEach(visit)
    return result
  }
  
  private groupByDependencyLevel(sortedModules: string[]): Map<number, string[]> {
    const levels = new Map<number, string[]>()
    const moduleLevel = new Map<string, number>()
    
    sortedModules.forEach(moduleName => {
      const deps = this.dependencies.get(moduleName) || []
      const maxDepLevel = deps.length > 0 
        ? Math.max(...deps.map(dep => moduleLevel.get(dep) || 0))
        : -1
      
      const level = maxDepLevel + 1
      moduleLevel.set(moduleName, level)
      
      if (!levels.has(level)) {
        levels.set(level, [])
      }
      levels.get(level)!.push(moduleName)
    })
    
    return levels
  }
}

// 持续改进循环实现
class ContinuousImprovementCycle {
  private metrics: PerformanceMetrics
  private feedback: FeedbackCollector
  private optimizer: CodeOptimizer
  
  constructor() {
    this.metrics = new PerformanceMetrics()
    this.feedback = new FeedbackCollector()
    this.optimizer = new CodeOptimizer()
  }
  
  // 执行改进循环
  async executeCycle(codebase: Codebase): Promise<ImprovementResult> {
    // 1. 测量当前性能
    const currentMetrics = await this.metrics.measure(codebase)
    
    // 2. 收集反馈
    const feedback = await this.feedback.collect()
    
    // 3. 识别改进点
    const improvements = this.identifyImprovements(currentMetrics, feedback)
    
    // 4. 生成优化方案
    const optimizations = await this.optimizer.generateOptimizations(improvements)
    
    // 5. 应用优化
    const optimizedCodebase = await this.applyOptimizations(codebase, optimizations)
    
    // 6. 验证改进效果
    const newMetrics = await this.metrics.measure(optimizedCodebase)
    
    return {
      before: currentMetrics,
      after: newMetrics,
      improvements: this.calculateImprovements(currentMetrics, newMetrics),
      optimizations
    }
  }
  
  private identifyImprovements(
    metrics: PerformanceMetrics, 
    feedback: UserFeedback[]
  ): ImprovementOpportunity[] {
    const opportunities: ImprovementOpportunity[] = []
    
    // 性能瓶颈识别
    if (metrics.loadTime > 3000) {
      opportunities.push({
        type: 'performance',
        priority: 'high',
        description: '页面加载时间过长',
        target: 'reduce-load-time',
        expectedImprovement: '50%'
      })
    }
    
    // 代码质量问题
    if (metrics.codeComplexity > 10) {
      opportunities.push({
        type: 'maintainability',
        priority: 'medium',
        description: '代码复杂度过高',
        target: 'reduce-complexity',
        expectedImprovement: '30%'
      })
    }
    
    // 用户体验问题
    const uxIssues = feedback.filter(f => f.type === 'ux' && f.severity === 'high')
    if (uxIssues.length > 0) {
      opportunities.push({
        type: 'user-experience',
        priority: 'high',
        description: '用户体验问题',
        target: 'improve-ux',
        expectedImprovement: '40%'
      })
    }
    
    return opportunities.sort((a, b) => {
      const priorityOrder = { high: 3, medium: 2, low: 1 }
      return priorityOrder[b.priority] - priorityOrder[a.priority]
    })
  }
  
  private async applyOptimizations(
    codebase: Codebase, 
    optimizations: Optimization[]
  ): Promise<Codebase> {
    let optimizedCodebase = { ...codebase }
    
    for (const optimization of optimizations) {
      try {
        optimizedCodebase = await this.applyOptimization(optimizedCodebase, optimization)
        
        // 验证优化后的代码
        const isValid = await this.validateOptimization(optimizedCodebase, optimization)
        if (!isValid) {
          console.warn(`优化 ${optimization.name} 验证失败，回滚更改`)
          optimizedCodebase = codebase // 回滚
        }
      } catch (error) {
        console.error(`应用优化 ${optimization.name} 失败:`, error)
      }
    }
    
    return optimizedCodebase
  }
  
  private calculateImprovements(
    before: PerformanceMetrics, 
    after: PerformanceMetrics
  ): ImprovementSummary {
    return {
      loadTime: {
        before: before.loadTime,
        after: after.loadTime,
        improvement: ((before.loadTime - after.loadTime) / before.loadTime * 100).toFixed(1) + '%'
      },
      bundleSize: {
        before: before.bundleSize,
        after: after.bundleSize,
        improvement: ((before.bundleSize - after.bundleSize) / before.bundleSize * 100).toFixed(1) + '%'
      },
      codeComplexity: {
        before: before.codeComplexity,
        after: after.codeComplexity,
        improvement: ((before.codeComplexity - after.codeComplexity) / before.codeComplexity * 100).toFixed(1) + '%'
      }
    }
  }
}

class DevelopmentEfficiencyMonitor {
  private activities: DevelopmentActivity[] = []
  private aiUsage: AIUsageRecord[] = []
  private metrics: EfficiencyMetrics
  
  constructor() {
    this.metrics = new EfficiencyMetrics()
  }
  
  // 记录开发活动
  recordActivity(activity: DevelopmentActivity): void {
    activity.timestamp = new Date()
    this.activities.push(activity)
    
    // 实时计算效率指标
    this.updateMetrics()
  }
  
  // 记录AI使用情况
  recordAIUsage(usage: AIUsageRecord): void {
    usage.timestamp = new Date()
    this.aiUsage.push(usage)
  }
  
  // 生成效率报告
  generateEfficiencyReport(timeRange: TimeRange): EfficiencyReport {
    const filteredActivities = this.filterActivitiesByTime(timeRange)
    const filteredAIUsage = this.filterAIUsageByTime(timeRange)
    
    return {
      timeRange,
      summary: this.calculateSummary(filteredActivities, filteredAIUsage),
      productivity: this.calculateProductivity(filteredActivities),
      aiImpact: this.calculateAIImpact(filteredAIUsage),
      recommendations: this.generateRecommendations(filteredActivities, filteredAIUsage)
    }
  }
  
  private calculateSummary(
    activities: DevelopmentActivity[], 
    aiUsage: AIUsageRecord[]
  ): EfficiencySummary {
    const totalTime = activities.reduce((sum, activity) => sum + activity.duration, 0)
    const codingTime = activities
      .filter(a => a.type === 'coding')
      .reduce((sum, activity) => sum + activity.duration, 0)
    
    const aiAssistedTime = aiUsage.reduce((sum, usage) => sum + usage.timeSaved, 0)
    
    return {
      totalDevelopmentTime: totalTime,
      actualCodingTime: codingTime,
      aiAssistedTime,
      efficiencyGain: (aiAssistedTime / totalTime * 100).toFixed(1) + '%',
      featuresCompleted: activities.filter(a => a.type === 'feature-completion').length,
      bugsFixed: activities.filter(a => a.type === 'bug-fix').length
    }
  }
  
  private calculateProductivity(activities: DevelopmentActivity[]): ProductivityMetrics {
    const codingActivities = activities.filter(a => a.type === 'coding')
    const totalLines = codingActivities.reduce((sum, a) => sum + (a.linesOfCode || 0), 0)
    const totalTime = codingActivities.reduce((sum, a) => sum + a.duration, 0)
    
    return {
      linesPerHour: totalTime > 0 ? (totalLines / (totalTime / 3600)).toFixed(1) : '0',
      featuresPerDay: this.calculateFeaturesPerDay(activities),
      averageTaskTime: this.calculateAverageTaskTime(activities),
      codeQualityScore: this.calculateCodeQualityScore(activities)
    }
  }
  
  private calculateAIImpact(aiUsage: AIUsageRecord[]): AIImpactMetrics {
    const totalTimeSaved = aiUsage.reduce((sum, usage) => sum + usage.timeSaved, 0)
    const totalRequests = aiUsage.length
    const successfulRequests = aiUsage.filter(u => u.successful).length
    
    return {
      totalTimeSaved: `${(totalTimeSaved / 3600).toFixed(1)} hours`,
      averageTimeSavedPerRequest: `${(totalTimeSaved / totalRequests / 60).toFixed(1)} minutes`,
      successRate: `${(successfulRequests / totalRequests * 100).toFixed(1)}%`,
      mostUsedFeatures: this.getMostUsedAIFeatures(aiUsage),
      efficiencyImprovement: this.calculateEfficiencyImprovement(aiUsage)
    }
  }
  
  private generateRecommendations(
    activities: DevelopmentActivity[], 
    aiUsage: AIUsageRecord[]
  ): string[] {
    const recommendations: string[] = []
    
    // 基于活动分析的建议
    const debuggingTime = activities
      .filter(a => a.type === 'debugging')
      .reduce((sum, a) => sum + a.duration, 0)
    
    const totalTime = activities.reduce((sum, a) => sum + a.duration, 0)
    
    if (debuggingTime / totalTime > 0.3) {
      recommendations.push('调试时间占比过高，建议增加单元测试和代码审查')
    }
    
    // 基于AI使用的建议
    const lowSuccessRateFeatures = this.getLowSuccessRateFeatures(aiUsage)
    if (lowSuccessRateFeatures.length > 0) {
      recommendations.push(`以下AI功能成功率较低，建议优化提示词: ${lowSuccessRateFeatures.join(', ')}`)
    }
    
    // 效率提升建议
    const underutilizedFeatures = this.getUnderutilizedAIFeatures(aiUsage)
    if (underutilizedFeatures.length > 0) {
      recommendations.push(`建议更多使用以下AI功能: ${underutilizedFeatures.join(', ')}`)
    }
    
    return recommendations
  }
  
  // 实时效率提醒
  checkEfficiencyAlerts(): EfficiencyAlert[] {
    const alerts: EfficiencyAlert[] = []
    const recentActivities = this.getRecentActivities(60) // 最近1小时
    
    // 检查是否长时间在同一任务上
    const currentTask = recentActivities[recentActivities.length - 1]
    if (currentTask && currentTask.duration > 7200) { // 超过2小时
      alerts.push({
        type: 'long-task',
        severity: 'warning',
        message: '当前任务耗时较长，建议考虑分解任务或寻求帮助',
        suggestion: '使用AI助手分析问题或将任务分解为更小的子任务'
      })
    }
    
    // 检查AI使用频率
    const recentAIUsage = this.getRecentAIUsage(60)
    if (recentAIUsage.length === 0 && recentActivities.length > 0) {
      alerts.push({
        type: 'low-ai-usage',
        severity: 'info',
        message: '最近没有使用AI助手，可能错过了提升效率的机会',
        suggestion: '考虑使用AI助手进行代码生成、调试或优化'
      })
    }
    
    return alerts
  }
}

class PersonalEfficiencyOptimizer {
