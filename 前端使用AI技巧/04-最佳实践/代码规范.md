### 2. React组件规范

```tsx
// ✅ 标准React组件结构

import React, { useState, useCallback, useMemo } from 'react'
import { Button, Input, message } from 'antd'
import { UserProfile, CreateUserRequest } from '@/types/user'
import { useCreateUser } from '@/hooks/useUser'
import { validateEmail } from '@/utils/validation'
import styles from './UserForm.module.css'

// 1. Props接口定义
interface UserFormProps {
  /** 初始用户数据 */
  initialData?: Partial<UserProfile>
  /** 提交成功回调 */
  onSuccess?: (user: UserProfile) => void
  /** 提交失败回调 */
  onError?: (error: Error) => void
  /** 是否显示取消按钮 */
  showCancel?: boolean
  /** 取消按钮回调 */
  onCancel?: () => void
}

// 2. 组件实现
export const UserForm: React.FC<UserFormProps> = ({
  initialData,
  onSuccess,
  onError,
  showCancel = false,
  onCancel
}) => {
  // 3. 状态定义
  const [formData, setFormData] = useState<CreateUserRequest>({
    name: initialData?.name || '',
    email: initialData?.email || '',
    avatar: initialData?.avatar
  })
  
  const [errors, setErrors] = useState<Record<string, string>>({})
  
  // 4. 自定义Hook使用
  const { createUser, loading } = useCreateUser({
    onSuccess: (user) => {
      message.success('用户创建成功')
      onSuccess?.(user)
    },
    onError: (error) => {
      message.error('用户创建失败')
      onError?.(error)
    }
  })
  
  // 5. 计算属性
  const isFormValid = useMemo(() => {
    return formData.name.trim() && 
           formData.email.trim() && 
           validateEmail(formData.email) &&
           Object.keys(errors).length === 0
  }, [formData, errors])
  
  // 6. 事件处理函数
  const handleInputChange = useCallback((field: keyof CreateUserRequest) => (
    event: React.ChangeEvent<HTMLInputElement>
  ) => {
    const value = event.target.value
    setFormData(prev => ({ ...prev, [field]: value }))
    
    // 清除对应字段的错误
    if (errors[field]) {
      setErrors(prev => {
        const newErrors = { ...prev }
        delete newErrors[field]
        return newErrors
      })
    }
  }, [errors])
  
  const validateForm = useCallback((): boolean => {
    const newErrors: Record<string, string> = {}
    
    if (!formData.name.trim()) {
      newErrors.name = '姓名不能为空'
    }
    
    if (!formData.email.trim()) {
      newErrors.email = '邮箱不能为空'
    } else if (!validateEmail(formData.email)) {
      newErrors.email = '邮箱格式不正确'
    }
    
    setErrors(newErrors)
    return Object.keys(newErrors).length === 0
  }, [formData])
  
  const handleSubmit = useCallback(async () => {
    if (!validateForm()) return
    
    await createUser({
      name: formData.name.trim(),
      email: formData.email.trim(),
      avatar: formData.avatar
    })
  }, [formData, validateForm, createUser])
  
  // 7. 渲染函数
  return (
    <form className={styles.userForm} onSubmit={(e) => e.preventDefault()}>
      <div className={styles.formGroup}>
        <label htmlFor="name" className={styles.label}>
          姓名 <span className={styles.required}>*</span>
        </label>
        <Input
          id="name"
          value={formData.name}
          onChange={handleInputChange('name')}
          placeholder="请输入姓名"
          status={errors.name ? 'error' : ''}
          aria-describedby={errors.name ? 'name-error' : undefined}
        />
        {errors.name && (
          <span id="name-error" className={styles.errorText} role="alert">
            {errors.name}
          </span>
        )}
      </div>
      
      <div className={styles.formGroup}>
        <label htmlFor="email" className={styles.label}>
          邮箱 <span className={styles.required}>*</span>
        </label>
        <Input
          id="email"
          type="email"
          value={formData.email}
          onChange={handleInputChange('email')}
          placeholder="请输入邮箱"
          status={errors.email ? 'error' : ''}
          aria-describedby={errors.email ? 'email-error' : undefined}
        />
        {errors.email && (
          <span id="email-error" className={styles.errorText} role="alert">
            {errors.email}
          </span>
        )}
      </div>
      
      <div className={styles.buttonGroup}>
        {showCancel && (
          <Button onClick={onCancel} disabled={loading}>
            取消
          </Button>
        )}
        <Button
          type="primary"
          onClick={handleSubmit}
          loading={loading}
          disabled={!isFormValid || loading}
        >
          保存
        </Button>
      </div>
    </form>
  )
}

// 8. 默认Props（如果需要）
UserForm.defaultProps = {
  showCancel: false
}

// 9. 显示名称（用于调试）
UserForm.displayName = 'UserForm'
```

### 3. 自定义Hook规范

```typescript
// ✅ 标准自定义Hook结构

import { useState, useEffect, useCallback, useRef } from 'react'
import { userApi } from '@/api/user'
import { UserProfile, CreateUserRequest } from '@/types/user'

// 1. Hook选项接口
interface UseUserOptions {
  /** 是否立即获取数据 */
  immediate?: boolean
  /** 错误回调 */
  onError?: (error: Error) => void
  /** 成功回调 */
  onSuccess?: (data: UserProfile[]) => void
}

// 2. Hook返回值接口
interface UseUserReturn {
  /** 用户列表数据 */
  users: UserProfile[]
  /** 加载状态 */
  loading: boolean
  /** 错误信息 */
  error: Error | null
  /** 刷新数据 */
  refresh: () => Promise<void>
  /** 创建用户 */
  createUser: (data: CreateUserRequest) => Promise<UserProfile>
  /** 更新用户 */
  updateUser: (id: string, data: Partial<UserProfile>) => Promise<UserProfile>
  /** 删除用户 */
  deleteUser: (id: string) => Promise<void>
}

// 3. Hook实现
export const useUser = (options: UseUserOptions = {}): UseUserReturn => {
  const { immediate = true, onError, onSuccess } = options
  
  // 4. 状态定义
  const [users, setUsers] = useState<UserProfile[]>([])
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<Error | null>(null)
  
  // 5. Ref用于避免闭包问题
  const optionsRef = useRef(options)
  optionsRef.current = options
  
  // 6. 获取数据函数
  const fetchUsers = useCallback(async () => {
    try {
      setLoading(true)
      setError(null)
      
      const data = await userApi.getUsers()
      setUsers(data)
      
      optionsRef.current.onSuccess?.(data)
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Unknown error')
      setError(error)
      optionsRef.current.onError?.(error)
    } finally {
      setLoading(false)
    }
  }, [])
  
  // 7. CRUD操作
  const createUser = useCallback(async (data: CreateUserRequest): Promise<UserProfile> => {
    try {
      setLoading(true)
      const newUser = await userApi.createUser(data)
      setUsers(prev => [...prev, newUser])
      return newUser
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Create failed')
      setError(error)
      throw error
    } finally {
      setLoading(false)
    }
  }, [])
  
  const updateUser = useCallback(async (
    id: string, 
    data: Partial<UserProfile>
  ): Promise<UserProfile> => {
    try {
      setLoading(true)
      const updatedUser = await userApi.updateUser(id, data)
      setUsers(prev => prev.map(user => 
        user.id === id ? updatedUser : user
      ))
      return updatedUser
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Update failed')
      setError(error)
      throw error
    } finally {
      setLoading(false)
    }
  }, [])
  
  const deleteUser = useCallback(async (id: string): Promise<void> => {
    try {
      setLoading(true)
      await userApi.deleteUser(id)
      setUsers(prev => prev.filter(user => user.id !== id))
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Delete failed')
      setError(error)
      throw error
    } finally {
      setLoading(false)
    }
  }, [])
  
  // 8. 副作用
  useEffect(() => {
    if (immediate) {
      fetchUsers()
    }
  }, [immediate, fetchUsers])
  
  // 9. 返回值
  return {
    users,
    loading,
    error,
    refresh: fetchUsers,
    createUser,
    updateUser,
    deleteUser
  }
}

// 10. 使用示例和文档
/**
 * 用户管理Hook
 * 
 * @example
 * ```tsx
 * const { users, loading, createUser } = useUser({
 *   immediate: true,
 *   onError: (error) => message.error(error.message)
 * })
 * ```
 */
```

/**
 * 用户管理组件
 * 
 * 提供用户的增删改查功能，支持批量操作和搜索过滤

 * @example
 * ```tsx
 * <UserManager
 *   initialUsers={users}

 *   onUserUpdate={handleUpdate}
 *   permissions={['create', 'update', 'delete']}
 * />
 * ```

 * @author AI Assistant
 * @since 1.0.0
 */
export interface UserManagerProps {
  /** 初始用户列表 */
  initialUsers?: UserProfile[]
  
  /** 
   * 用户创建回调
   * @param user 新创建的用户信息
   */
  onUserCreate?: (user: UserProfile) => void
  
  /** 
   * 用户更新回调
   * @param user 更新后的用户信息
   * @param changes 变更的字段
   */
  onUserUpdate?: (user: UserProfile, changes: Partial<UserProfile>) => void
  
  /** 
   * 用户权限配置
   * @default ['read']
   */
  permissions?: Array<'create' | 'read' | 'update' | 'delete'>
}

/**
 * 格式化货币金额
 * 
 * 将数字格式化为指定货币格式，支持多种货币和地区设置
 * 
 * @param amount - 金额数值
 * @param currency - 货币代码，如 'CNY', 'USD'
 * @param locale - 地区设置，如 'zh-CN', 'en-US'
 * @returns 格式化后的货币字符串
 * 
 * @throws {Error} 当amount不是有效数字时抛出错误
 * 
 * @example
 * ```typescript
 * formatCurrency(1234.56) // '¥1,234.56'
 * formatCurrency(1234.56, 'USD', 'en-US') // '$1,234.56'
 * ```
 */
export function formatCurrency(
  amount: number,
  currency: string = 'CNY',
  locale: string = 'zh-CN'
): string {
  // 参数验证
  if (typeof amount !== 'number' || isNaN(amount)) {
    throw new Error('Amount must be a valid number')
  }
  
  // 使用Intl.NumberFormat进行格式化
  return new Intl.NumberFormat(locale, {
    style: 'currency',
    currency: currency,
    minimumFractionDigits: 2,
    maximumFractionDigits: 2
  }).format(amount)
}
```

// scripts/quality-check.ts
import { execSync } from 'child_process'
import fs from 'fs'
import path from 'path'

interface QualityMetrics {
  eslintErrors: number
  eslintWarnings: number
  testCoverage: number
  typeErrors: number
  duplicateCode: number
}

class QualityChecker {
  private projectRoot: string
  
  constructor(projectRoot: string) {
    this.projectRoot = projectRoot
  }
  
  async checkQuality(): Promise<QualityMetrics> {
    console.log('🔍 开始代码质量检查...')
    
    const metrics: QualityMetrics = {
      eslintErrors: await this.checkESLint(),
      eslintWarnings: 0,
      testCoverage: await this.checkTestCoverage(),
      typeErrors: await this.checkTypeScript(),
      duplicateCode: await this.checkDuplicateCode()
    }
    
    this.generateReport(metrics)
    return metrics
  }
  
  private async checkESLint(): Promise<number> {
    try {
      const result = execSync('npx eslint src --format json', {
        cwd: this.projectRoot,
        encoding: 'utf8'
      })
      
      const eslintResults = JSON.parse(result)
      return eslintResults.reduce((total: number, file: any) => {
        return total + file.errorCount
      }, 0)
    } catch (error) {
      console.error('ESLint检查失败:', error)
      return -1
    }
  }
  
  private async checkTestCoverage(): Promise<number> {
    try {
      execSync('npm run test:coverage', {
        cwd: this.projectRoot,
        stdio: 'pipe'
      })
      
      const coveragePath = path.join(this.projectRoot, 'coverage/coverage-summary.json')
      if (fs.existsSync(coveragePath)) {
        const coverage = JSON.parse(fs.readFileSync(coveragePath, 'utf8'))
        return coverage.total.lines.pct
      }
      
      return 0
    } catch (error) {
      console.error('测试覆盖率检查失败:', error)
      return 0
    }
  }
  
  private async checkTypeScript(): Promise<number> {
    try {
      const result = execSync('npx tsc --noEmit --pretty false', {
        cwd: this.projectRoot,
        encoding: 'utf8'
      })
      
      // 如果没有错误，tsc不会输出任何内容
      return result.trim() ? result.split('\n').length : 0
    } catch (error) {
      // tsc在有错误时会抛出异常
      const output = (error as any).stdout || ''
      return output.trim() ? output.split('\n').length : 0
    }
  }
  
  private async checkDuplicateCode(): Promise<number> {
    // 简化的重复代码检查
    // 实际项目中可以使用jscpd等工具
    return 0
  }
  
  private generateReport(metrics: QualityMetrics): void {
    const report = `
# 代码质量报告

生成时间: ${new Date().toLocaleString()}

## 📊 质量指标

- ESLint错误: ${metrics.eslintErrors}
- 测试覆盖率: ${metrics.testCoverage.toFixed(1)}%
- TypeScript错误: ${metrics.typeErrors}
- 重复代码: ${metrics.duplicateCode}

## 📈 质量评级

${this.getQualityGrade(metrics)}

## 🎯 改进建议

${this.getImprovementSuggestions(metrics)}
    `.trim()
    
    fs.writeFileSync(
      path.join(this.projectRoot, 'quality-report.md'),
      report
    )
    
    console.log('📋 质量报告已生成: quality-report.md')
  }
  
  private getQualityGrade(metrics: QualityMetrics): string {
    const score = this.calculateQualityScore(metrics)
    
    if (score >= 90) return '🟢 优秀 (A)'
    if (score >= 80) return '🟡 良好 (B)'
    if (score >= 70) return '🟠 一般 (C)'
    return '🔴 需要改进 (D)'
  }
  
  private calculateQualityScore(metrics: QualityMetrics): number {
    let score = 100
    
    // ESLint错误扣分
    score -= metrics.eslintErrors * 5
    
    // 测试覆盖率加分
    score += (metrics.testCoverage - 80) * 0.5
    
    // TypeScript错误扣分
    score -= metrics.typeErrors * 3
    
    return Math.max(0, Math.min(100, score))
  }
  
  private getImprovementSuggestions(metrics: QualityMetrics): string {
    const suggestions: string[] = []
    
    if (metrics.eslintErrors > 0) {
      suggestions.push('- 修复ESLint错误，提高代码规范性')
    }
    
    if (metrics.testCoverage < 80) {
      suggestions.push('- 增加测试用例，提高测试覆盖率')
    }
    
    if (metrics.typeErrors > 0) {
      suggestions.push('- 修复TypeScript类型错误')
    }
    
    return suggestions.length > 0 
      ? suggestions.join('\n')
      : '- 代码质量良好，继续保持！'
  }
}
