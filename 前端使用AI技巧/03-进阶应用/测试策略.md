// utils/format.ts
export function formatCurrency(amount: number, currency = 'CNY', locale = 'zh-CN'): string {
  if (typeof amount !== 'number' || isNaN(amount)) {
    throw new Error('Amount must be a valid number')
  }
  
  return new Intl.NumberFormat(locale, {
    style: 'currency',
    currency: currency,
    minimumFractionDigits: 2,
    maximumFractionDigits: 2
  }).format(amount)
}

export function debounce<T extends (...args: any[]) => any>(
  func: T,
  wait: number
): (...args: Parameters<T>) => void {
  let timeout: NodeJS.Timeout
  
  return function executedFunction(...args: Parameters<T>) {
    const later = () => {
      clearTimeout(timeout)
      func(...args)
    }
    
    clearTimeout(timeout)
    timeout = setTimeout(later, wait)
  }
}

<script setup lang="ts">
import { ref, reactive, watch } from 'vue'
import { ElMessage } from 'element-plus'

interface UserFormProps {
  initialData?: { name: string; email: string }
  loading?: boolean
}

interface UserFormEmits {
  submit: [data: { name: string; email: string }]
}

const props = withDefaults(defineProps<UserFormProps>(), {
  loading: false
})

const emit = defineEmits<UserFormEmits>()

const name = ref(props.initialData?.name || '')
const email = ref(props.initialData?.email || '')
const errors = reactive<{ name?: string; email?: string }>({})

watch(() => props.initialData, (newData) => {
  if (newData) {
    name.value = newData.name || ''
    email.value = newData.email || ''
  }
}, { immediate: true })

const validateForm = () => {
  const newErrors: { name?: string; email?: string } = {}
  
  if (!name.value.trim()) {
    newErrors.name = '姓名不能为空'
  }
  
  if (!email.value.trim()) {
    newErrors.email = '邮箱不能为空'
  } else if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email.value)) {
    newErrors.email = '邮箱格式不正确'
  }
  
  Object.assign(errors, newErrors)
  return Object.keys(newErrors).length === 0
}

const handleSubmit = async () => {
  if (!validateForm()) return
  
  try {
    emit('submit', { 
      name: name.value.trim(), 
      email: email.value.trim() 
    })
    ElMessage.success('保存成功')
  } catch (error) {
    ElMessage.error('保存失败，请重试')
  }
}
</script>

<style scoped>
.user-form {
      <div className="form-item">
        <label>姓名</label>
        <Input
          value={name}
          onChange={(e) => setName(e.target.value)}
          placeholder="请输入姓名"
          status={errors.name ? 'error' : ''}
        />
        {errors.name && <span className="error-text">{errors.name}</span>}
      </div>
      
      <div className="form-item">
        <label>邮箱</label>
        <Input

          status={errors.email ? 'error' : ''}
        />
        {errors.email && <span className="error-text">{errors.email}</span>}
      </div>
      
      <Button 
        type="primary" 
        onClick={handleSubmit}
        loading={loading}
        disabled={loading}
      >

  )
}

// scripts/test-quality.ts
import fs from 'fs'
import path from 'path'

interface CoverageReport {
  total: {
    lines: { pct: number }
    functions: { pct: number }
    statements: { pct: number }
    branches: { pct: number }
  }
}

interface TestQualityMetrics {
  coverage: CoverageReport['total']
  testFiles: number
  totalTests: number
  testToCodeRatio: number
  avgTestsPerFile: number
}

class TestQualityAnalyzer {
  private projectRoot: string
  
  constructor(projectRoot: string) {
    this.projectRoot = projectRoot
  }
  
  async analyzeTestQuality(): Promise<TestQualityMetrics> {
    const coverage = this.getCoverageReport()
    const testStats = this.getTestStatistics()
    const codeStats = this.getCodeStatistics()
    
    return {
      coverage: coverage.total,
      testFiles: testStats.files,
      totalTests: testStats.tests,
      testToCodeRatio: testStats.files / codeStats.files,
      avgTestsPerFile: testStats.tests / testStats.files
    }
  }
  
  private getCoverageReport(): CoverageReport {
    const coveragePath = path.join(this.projectRoot, 'coverage/coverage-summary.json')
    
    if (!fs.existsSync(coveragePath)) {
      throw new Error('Coverage report not found. Run tests with coverage first.')
    }
    
    return JSON.parse(fs.readFileSync(coveragePath, 'utf8'))
  }
  
  private getTestStatistics() {
    const testFiles = this.findFiles('**/*.test.{ts,tsx,js,jsx}')
    let totalTests = 0
    
    testFiles.forEach(file => {
      const content = fs.readFileSync(file, 'utf8')
      const testMatches = content.match(/\b(it|test)\s*\(/g)
      totalTests += testMatches ? testMatches.length : 0
    })
    
    return {
      files: testFiles.length,
      tests: totalTests
    }
  }
  
  private getCodeStatistics() {
    const codeFiles = this.findFiles('src/**/*.{ts,tsx,js,jsx}')
      .filter(file => !file.includes('.test.') && !file.includes('.spec.'))
    
    return {
      files: codeFiles.length
    }
  }
  
  private findFiles(pattern: string): string[] {
    // 简化的文件查找实现
    // 实际项目中可以使用 glob 库
    return []
  }
  
  generateQualityReport(metrics: TestQualityMetrics): string {
    const { coverage, testFiles, totalTests, testToCodeRatio, avgTestsPerFile } = metrics
    
    return `
# 测试质量报告

## 📊 覆盖率指标
- 行覆盖率: ${coverage.lines.pct}%
- 函数覆盖率: ${coverage.functions.pct}%
- 语句覆盖率: ${coverage.statements.pct}%
- 分支覆盖率: ${coverage.branches.pct}%

## 🧪 测试统计
- 测试文件数: ${testFiles}
- 测试用例总数: ${totalTests}
- 测试代码比: ${(testToCodeRatio * 100).toFixed(1)}%
- 平均每文件测试数: ${avgTestsPerFile.toFixed(1)}

## 📈 质量评估
${this.getQualityAssessment(metrics)}

## 🎯 改进建议
${this.getImprovementSuggestions(metrics)}
    `.trim()
  }
  
  private getQualityAssessment(metrics: TestQualityMetrics): string {
    const { coverage } = metrics
    const avgCoverage = (coverage.lines.pct + coverage.functions.pct + coverage.statements.pct + coverage.branches.pct) / 4
    
    if (avgCoverage >= 90) return '🟢 优秀 - 测试覆盖率很高'
    if (avgCoverage >= 80) return '🟡 良好 - 测试覆盖率达标'
    if (avgCoverage >= 70) return '🟠 一般 - 测试覆盖率需要提升'
    return '🔴 较差 - 测试覆盖率严重不足'
  }
  
  private getImprovementSuggestions(metrics: TestQualityMetrics): string {
    const suggestions: string[] = []
    
    if (metrics.coverage.branches.pct < 80) {
      suggestions.push('- 增加边界条件和异常情况的测试')
    }
    
    if (metrics.testToCodeRatio < 0.5) {
      suggestions.push('- 为更多组件和函数添加测试')
    }
    
    if (metrics.avgTestsPerFile < 3) {
      suggestions.push('- 增加每个模块的测试用例数量')
    }
    
    return suggestions.length > 0 ? suggestions.join('\n') : '- 当前测试质量良好，继续保持'
  }
}

    console.log(report)
