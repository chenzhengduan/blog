# 性能分析报告

## 🔍 发现的性能问题

### 1. 频繁的过滤计算
**问题**: 每次输入变化都会触发完整的用户列表过滤
**影响**: 当用户列表很大时，会造成明显的输入延迟

### 2. 缺少防抖处理
**问题**: 用户快速输入时会触发多次不必要的过滤操作
**影响**: 浪费计算资源，影响输入体验

### 3. 未使用虚拟化
**问题**: 大量DOM节点同时渲染
**影响**: 长列表渲染性能差，滚动不流畅

### 4. 缺少记忆化优化
**问题**: 子组件可能因为父组件重渲染而不必要地重新渲染
**影响**: 整体渲染性能下降

```javascript
// workers/filterWorker.js
self.onmessage = function(e) {
  const { users, searchTerm, filters } = e.data
  
  const filteredUsers = users.filter(user => {
    const matchesSearch = !searchTerm || 
      user.name.toLowerCase().includes(searchTerm.toLowerCase())
    const matchesDepartment = !filters.department || 
      user.department === filters.department
    const matchesStatus = !filters.status || 
      user.status === filters.status
```

import { useDebounceFn } from '@vueuse/core'
import UserListItems from './UserListItems.vue'

interface User {
  id: string
  name: string
  email: string
  avatar: string
  department: string
  status: string

interface Props {
  users: User[]
  filters: {
    department?: string
    status?: string
  }

const emit = defineEmits<{
  userSelect: [user: User]

const searchTerm = ref('')
const filteredUsers = ref<User[]>([])

// 防抖处理搜索输入

  searchTerm.value = value
}, 300)

watch(searchInput, (newValue) => {
  debouncedSearch(newValue)

  // 创建Web Worker
  worker = new Worker('/workers/filterWorker.js')
  
  worker.onmessage = (e) => {
    filteredUsers.value = e.data
  }
})

onUnmounted(() => {
  worker?.terminate()
})

// 监听数据变化并处理过滤
watch(
  [() => props.users, searchTerm, () => props.filters],
  ([users, search, filters]) => {
    if (worker && users.length > 1000) {
      // 大数据量时使用Worker
      worker.postMessage({
        users,
        searchTerm: search,
        filters
      })
    } else {
      // 小数据量直接在主线程处理
      filteredUsers.value = users.filter(user => {
        const matchesSearch = !search || 
          user.name.toLowerCase().includes(search.toLowerCase())
        const matchesDepartment = !filters.department || 
          user.department === filters.department
        const matchesStatus = !filters.status || 
          user.status === filters.status
          
        return matchesSearch && matchesDepartment && matchesStatus
      })
    }
  },
  { immediate: true }
)

  emit('userSelect', user)
}

```typescript
// composables/useIntelligentPreload.ts
import { watch } from 'vue'
import { useRoute } from 'vue-router'

// 预加载映射表
const preloadMap: Record<string, string[]> = {
  '/': ['/users', '/reports'], // 在首页时预加载用户和报告页面
  '/users': ['/users/create', '/users/edit'], // 在用户列表时预加载创建和编辑页面
  '/reports': ['/reports/analytics'] // 在报告页面时预加载分析页面
}

export function useIntelligentPreload() {
  const route = useRoute()
  
  watch(
    () => route.path,
    (currentPath) => {
      const preloadPaths = preloadMap[currentPath]
      
      if (preloadPaths) {
        // 延迟预加载，避免影响当前页面性能
        setTimeout(() => {
          preloadPaths.forEach(path => {
            // 预加载路由组件
            import(`@/pages${path}.vue`).catch(() => {
              // 静默处理预加载失败
            })
          })
        }, 2000) // 2秒后开始预加载
      }
    },
    { immediate: true }
  )
}

// composables/useLinkPreload.ts
export function useLinkPreload() {
  const preloadLink = (href: string) => {
    const link = document.createElement('link')
    link.rel = 'prefetch'
    link.href = href
    document.head.appendChild(link)
  }
  
  const handleMouseEnter = (href: string) => {
    preloadLink(href)
  }
  
  return { handleMouseEnter }
}
```

  className?: string
  placeholder?: string
  sizes?: string
  priority?: boolean

  sizes: '(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw',
  priority: false

const imageError = ref(false)

  imgRef,
  {
    threshold: 0.1,
    rootMargin: '50px'
  }
)

// 生成响应式图片URL
const generateResponsiveUrl = (originalSrc: string, width: number) => {
  // 假设使用图片CDN服务
  return `${originalSrc}?w=${width}&q=80&f=webp`
}

// 生成srcSet
const generateSrcSet = (originalSrc: string) => {
  const widths = [320, 640, 768, 1024, 1280, 1920]
  return widths
    .map(w => `${generateResponsiveUrl(originalSrc, w)} ${w}w`)
    .join(', ')
}

// 监听视口变化并加载图片
watch(
  [isInView, () => props.priority, () => props.src],
  ([inView, priority, src]) => {
    if (!priority && inView && imageSrc.value === props.placeholder) {
      isLoading.value = true
      
      // 预加载图片
      const img = new Image()
      img.onload = () => {
        imageSrc.value = src
        isLoading.value = false
      }
      img.onerror = () => {
        imageError.value = true
        isLoading.value = false
      }
      img.src = src
    }
  },
  { immediate: true }
)

  imageError.value = true
  imageSrc.value = '/images/error-placeholder.svg'

```typescript
// composables/useImagePreload.ts
import { ref, watch } from 'vue'

export function useImagePreload(urls: string[]) {
  const loadedImages = ref<Set<string>>(new Set())
  
  watch(
    () => urls,
    (newUrls) => {
      const preloadImages = newUrls.map(url => {
        return new Promise<string>((resolve, reject) => {
          const img = new Image()
          img.onload = () => resolve(url)
          img.onerror = reject
          img.src = url
        })
      })
      
      Promise.allSettled(preloadImages).then(results => {
        const loaded = new Set<string>()
        results.forEach((result, index) => {
          if (result.status === 'fulfilled') {
            loaded.add(newUrls[index])
          }
        })
        loadedImages.value = loaded
      })
    },
    { immediate: true, deep: true }
  )
  
  return loadedImages
}
```

```typescript
// composables/useTouchOptimization.ts
import { ref } from 'vue'

export function useTouchOptimization() {
  const touchStart = ref<{ x: number; y: number; time: number } | null>(null)
  
  const handleTouchStart = (e: TouchEvent) => {
    const touch = e.touches[0]
    touchStart.value = {
      x: touch.clientX,
      y: touch.clientY,
      time: Date.now()
    }
  }
  
  const handleTouchEnd = (e: TouchEvent, onClick: () => void) => {
    if (!touchStart.value) return
    
    const touch = e.changedTouches[0]
    const deltaX = Math.abs(touch.clientX - touchStart.value.x)
    const deltaY = Math.abs(touch.clientY - touchStart.value.y)
    const deltaTime = Date.now() - touchStart.value.time
    
    // 判断是否为点击（而非滑动）
    if (deltaX < 10 && deltaY < 10 && deltaTime < 500) {
      onClick()
    }
    
    touchStart.value = null
  }
  
  return { handleTouchStart, handleTouchEnd }
}
```

```typescript
// composables/useScrollOptimization.ts
import { ref, onMounted, onUnmounted } from 'vue'
import { useThrottleFn } from '@vueuse/core'

export function useScrollOptimization() {
  const scrollElementRef = ref<HTMLDivElement>()
  const isScrolling = ref(false)
  let scrollTimeout: NodeJS.Timeout
  
  // 节流的滚动处理
  const handleScroll = useThrottleFn((e: Event) => {
    if (!isScrolling.value) {
      isScrolling.value = true
      // 滚动开始时的优化
      document.body.style.pointerEvents = 'none'
    }
    
    // 清除之前的定时器
    if (scrollTimeout) {
      clearTimeout(scrollTimeout)
    }
    
    // 滚动结束后恢复
    scrollTimeout = setTimeout(() => {
      isScrolling.value = false
      document.body.style.pointerEvents = 'auto'
    }, 150)
  }, 16) // 约60fps
  
  onMounted(() => {
    const element = scrollElementRef.value
    if (!element) return
    
    element.addEventListener('scroll', handleScroll, { passive: true })
  })
  
  onUnmounted(() => {
    const element = scrollElementRef.value
    if (element) {
      element.removeEventListener('scroll', handleScroll)
    }
    if (scrollTimeout) {
      clearTimeout(scrollTimeout)
    }
  })
  
  return {
    scrollElementRef,
    isScrolling
  }
}
```

### 1. 性能指标收集

```typescript
// 性能监控工具
class PerformanceMonitor {
  private metrics: Map<string, number> = new Map()
  private observers: PerformanceObserver[] = []
  
  constructor() {
    this.initObservers()
  }
  
  private initObservers() {
    // 监控核心Web指标
    if ('PerformanceObserver' in window) {
      // LCP (Largest Contentful Paint)
      const lcpObserver = new PerformanceObserver((list) => {
        const entries = list.getEntries()
        const lastEntry = entries[entries.length - 1]
        this.metrics.set('LCP', lastEntry.startTime)
      })
      lcpObserver.observe({ entryTypes: ['largest-contentful-paint'] })
      this.observers.push(lcpObserver)
      
      // FID (First Input Delay)
      const fidObserver = new PerformanceObserver((list) => {
        const entries = list.getEntries()
        entries.forEach((entry: any) => {
          this.metrics.set('FID', entry.processingStart - entry.startTime)
        })
      })
      fidObserver.observe({ entryTypes: ['first-input'] })
      this.observers.push(fidObserver)
      
      // CLS (Cumulative Layout Shift)
      let clsValue = 0
      const clsObserver = new PerformanceObserver((list) => {
        const entries = list.getEntries()
        entries.forEach((entry: any) => {
          if (!entry.hadRecentInput) {
            clsValue += entry.value
            this.metrics.set('CLS', clsValue)
          }
        })
      })
      clsObserver.observe({ entryTypes: ['layout-shift'] })
      this.observers.push(clsObserver)
    }
  }
  
  // 测量自定义指标
  measureCustomMetric(name: string, fn: () => void) {
    const start = performance.now()
    fn()
    const end = performance.now()
    this.metrics.set(name, end - start)
  }
  
  // 获取页面加载指标
  getPageLoadMetrics() {
    const navigation = performance.getEntriesByType('navigation')[0] as PerformanceNavigationTiming
    
    return {
      // DNS查询时间
      dnsLookup: navigation.domainLookupEnd - navigation.domainLookupStart,
      // TCP连接时间
      tcpConnect: navigation.connectEnd - navigation.connectStart,
      // 请求响应时间
      request: navigation.responseEnd - navigation.requestStart,
      // DOM解析时间
      domParse: navigation.domContentLoadedEventEnd - navigation.responseEnd,
      // 资源加载时间
      resourceLoad: navigation.loadEventEnd - navigation.domContentLoadedEventEnd,
      // 总加载时间
      totalLoad: navigation.loadEventEnd - navigation.navigationStart
    }
  }
  
  // 获取资源加载指标
  getResourceMetrics() {
    const resources = performance.getEntriesByType('resource')
    
    const metrics = {
      totalResources: resources.length,
      totalSize: 0,
      slowResources: [],
      resourceTypes: {}
    }
    
    resources.forEach((resource: any) => {
      const duration = resource.responseEnd - resource.startTime
      
      // 记录慢资源
      if (duration > 1000) {
        metrics.slowResources.push({
          name: resource.name,
          duration,
          size: resource.transferSize
        })
      }
      
      // 统计资源类型
      const type = this.getResourceType(resource.name)
      metrics.resourceTypes[type] = (metrics.resourceTypes[type] || 0) + 1
      
      // 累计大小
      metrics.totalSize += resource.transferSize || 0
    })
    
    return metrics
  }
  
  private getResourceType(url: string): string {
    if (url.match(/\.(js|jsx|ts|tsx)$/)) return 'script'
    if (url.match(/\.(css|scss|sass)$/)) return 'stylesheet'
    if (url.match(/\.(jpg|jpeg|png|gif|webp|svg)$/)) return 'image'
    if (url.match(/\.(woff|woff2|ttf|otf)$/)) return 'font'
    return 'other'
  }
  
  // 发送性能数据
  sendMetrics() {
    const data = {
      coreMetrics: Object.fromEntries(this.metrics),
      pageLoad: this.getPageLoadMetrics(),
      resources: this.getResourceMetrics(),
      userAgent: navigator.userAgent,
      timestamp: Date.now()
    }
    
    // 发送到分析服务
    if (navigator.sendBeacon) {
      navigator.sendBeacon('/api/performance', JSON.stringify(data))
    } else {
      fetch('/api/performance', {
        method: 'POST',
        body: JSON.stringify(data),
        headers: { 'Content-Type': 'application/json' }
      }).catch(() => {
        // 静默处理发送失败
      })
    }
  }
  
  // 清理观察器
  cleanup() {
    this.observers.forEach(observer => observer.disconnect())
    this.observers = []
  }
}

// Vue 3 Composable封装
import { ref, onMounted, onUnmounted } from 'vue'

export function usePerformanceMonitor() {
  const monitorRef = ref<PerformanceMonitor>()
  
  onMounted(() => {
    monitorRef.value = new PerformanceMonitor()
    
    // 页面卸载时发送数据
    const handleBeforeUnload = () => {
      monitorRef.value?.sendMetrics()
    }
    
    window.addEventListener('beforeunload', handleBeforeUnload)
    
    onUnmounted(() => {
      window.removeEventListener('beforeunload', handleBeforeUnload)
      monitorRef.value?.cleanup()
    })
  })
  
  const measureCustomMetric = (name: string, fn: () => void) => {
    monitorRef.value?.measureCustomMetric(name, fn)
  }
  
  return { measureCustomMetric }
}
```
