## 🚀 快速开始示例

### 基础表格
```vue
<template>
  <fan-table
    :columns="columns"
    :table-data="tableData"
    row-key-field-name="id"
  />
</template>

<script>
export default {
  data() {
    return {
      columns: [
        { field: 'name', key: 'name', title: '姓名', width: 100 },
        { field: 'age', key: 'age', title: '年龄', width: 80 },
        { field: 'address', key: 'address', title: '地址' }
      ],
      tableData: [
        { id: 1, name: '张三', age: 25, address: '北京市朝阳区' },
        { id: 2, name: '李四', age: 30, address: '上海市浦东新区' }
      ]
    }
  }
}
</script>
```

### 高级功能组合
```vue
<template>
  <fan-table
    :columns="columns"
    :table-data="tableData"
    :virtual-scroll-option="virtualScrollOption"
    :checkbox-option="checkboxOption"
    :expand-option="expandOption"
    :cell-autofill-option="cellAutofillOption"
    :clipboard-option="clipboardOption"
    row-key-field-name="id"
    :max-height="500"
  />
</template>

<script>
export default {
  data() {
    return {
      // 虚拟滚动 - 大数据性能优化
      virtualScrollOption: {
        enable: true,
        minRowHeight: 40
      },
      // 行多选功能
      checkboxOption: {
        selectedRowChange: ({ selectedRowKeys }) => {
          console.log('选中行:', selectedRowKeys)
        }
      },
      // 行展开功能
      expandOption: {
        render: ({ row }) => {
          return <div>详细信息: {row.name} - {row.address}</div>
        }
      },
      // 单元格自动填充
      cellAutofillOption: {
        directionX: true,
        directionY: true
      },
      // 剪贴板操作
      clipboardOption: {
        copy: true,
        paste: true,
        cut: true,
        delete: true
      }
    }
  }
}
</script>
```
