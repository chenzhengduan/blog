class UserManager {
  constructor() {
    this.users = [];
    this.nextId = 1;
  }

  addUser(userData) {
    // 验证用户数据
    if (!userData.name || !userData.email) {
      throw new Error('Name and email are required');
    }

    if (!validateEmail(userData.email)) {
      throw new Error('Invalid email format');
    }

    // 检查邮箱是否已存在
    if (this.users.find(user => user.email === userData.email)) {
      throw new Error('Email already exists');
    }

    const user = {
      id: this.nextId++,
      name: userData.name,
      email: userData.email,
      createdAt: new Date(),
      updatedAt: new Date()
    };

    this.users.push(user);
    return user;
  }

  getUser(id) {
    const user = this.users.find(u => u.id === id);
    if (!user) {
      throw new Error('User not found');
    }
    return user;
  }

  updateUser(id, updates) {
    const userIndex = this.users.findIndex(u => u.id === id);
    if (userIndex === -1) {
      throw new Error('User not found');
    }

    // 验证更新数据
    if (updates.email && !validateEmail(updates.email)) {
      throw new Error('Invalid email format');

    // 检查邮箱冲突
    if (updates.email && updates.email !== this.users[userIndex].email) {
      if (this.users.find(u => u.email === updates.email && u.id !== id)) {
        throw new Error('Email already exists');
      }
    }

    this.users[userIndex] = {
      ...this.users[userIndex],
      ...updates,
      updatedAt: new Date()
    };

    return this.users[userIndex];
  }

    const userIndex = this.users.findIndex(u => u.id === id);
    if (userIndex === -1) {

        }

        return userValue === filterValue;

/**
 * AI提示：实现快速排序算法
 * 要求：包含详细的注释，时间复杂度分析
 */
class QuickSort {
  constructor() {
    this.comparisons = 0;
    this.swaps = 0;
  }

  sort(array) {
    this.comparisons = 0;
    this.swaps = 0;

    if (array.length <= 1) {
      return array;
    }

    const sorted = this.quickSort(array);

    console.log(`排序完成，比较次数: ${this.comparisons}, 交换次数: ${this.swaps}`);
    return sorted;
  }

  quickSort(array) {
    if (array.length <= 1) {
      return array;
    }

    const pivot = this.partition(array);
    const left = array.slice(0, pivot);
    const right = array.slice(pivot + 1);

    return [
      ...this.quickSort(left),
      array[pivot],
      ...this.quickSort(right)
    ];
  }

  partition(array) {
    const pivot = array[0];
    let i = 1;

    for (let j = 1; j < array.length; j++) {
      this.comparisons++;
      if (array[j] < pivot) {
        if (i !== j) {
          this.swap(array, i, j);
          this.swaps++;
        }
        i++;
      }
    }

    if (i > 1) {
      this.swap(array, 0, i - 1);
      this.swaps++;
    }

    return i - 1;
  }

  swap(array, i, j) {
    const temp = array[i];
    array[i] = array[j];
    array[j] = temp;
  }

  // 时间复杂度分析
  getTimeComplexity() {
    return {
      best: 'O(n log n)',  // 每次都能将数组均匀分割
      average: 'O(n log n)',  // 平均情况下
      worst: 'O(n²)',  // 数组已经排序或逆序
      space: 'O(log n)'  // 递归调用栈空间
    };
  }
}

// 使用示例

console.log('排序结果:', sortedArray);

```javascript
/**
 * AI提示：实现图的深度优先搜索和广度优先搜索
 * 要求：支持加权图，路径记录，最短路径查找
 */
class Graph {
  constructor() {
    this.adjacencyList = new Map();
  }

  addVertex(vertex) {
    if (!this.adjacencyList.has(vertex)) {
      this.adjacencyList.set(vertex, []);
    }
  }

  addEdge(vertex1, vertex2, weight = 1) {
    this.addVertex(vertex1);
    this.addVertex(vertex2);

    this.adjacencyList.get(vertex1).push({ node: vertex2, weight });
    this.adjacencyList.get(vertex2).push({ node: vertex1, weight });
  }

  // 深度优先搜索
  dfs(startVertex) {
    const visited = new Set();
    const path = [];
    const traversalOrder = [];

    const dfsRecursive = (vertex) => {
      visited.add(vertex);
      traversalOrder.push(vertex);

      const neighbors = this.adjacencyList.get(vertex);
      for (const neighbor of neighbors) {
        if (!visited.has(neighbor.node)) {
          path.push(neighbor.node);
          dfsRecursive(neighbor.node);
          path.pop();
        }
      }
    };

    dfsRecursive(startVertex);

    return {
      traversalOrder,
      path: [...path, startVertex],
      visited: Array.from(visited)
    };
  }

  // 广度优先搜索
  bfs(startVertex) {
    const visited = new Set();
    const queue = [startVertex];
    const traversalOrder = [];
    const distances = new Map();
    const previous = new Map();

    visited.add(startVertex);
    distances.set(startVertex, 0);
    previous.set(startVertex, null);

    while (queue.length > 0) {
      const current = queue.shift();
      traversalOrder.push(current);

      const neighbors = this.adjacencyList.get(current);
      for (const neighbor of neighbors) {
        if (!visited.has(neighbor.node)) {
          visited.add(neighbor.node);
          queue.push(neighbor.node);
          distances.set(neighbor.node, distances.get(current) + neighbor.weight);
          previous.set(neighbor.node, current);
        }
      }
    }

    return {
      traversalOrder,
      distances,
      previous
    };
  }

  // Dijkstra最短路径算法
  dijkstra(startVertex, endVertex) {
    const distances = new Map();
    const previous = new Map();
    const unvisited = new Set();

    // 初始化距离
    for (const vertex of this.adjacencyList.keys()) {
      distances.set(vertex, vertex === startVertex ? 0 : Infinity);
      previous.set(vertex, null);
      unvisited.add(vertex);
    }

    while (unvisited.size > 0) {
      // 找到未访问的最近顶点
      let current = null;
      let minDistance = Infinity;

      for (const vertex of unvisited) {
        if (distances.get(vertex) < minDistance) {
          current = vertex;
          minDistance = distances.get(vertex);
        }
      }

      if (current === null || minDistance === Infinity) {
        break;
      }

      unvisited.delete(current);

      // 更新邻居距离
      const neighbors = this.adjacencyList.get(current);
      for (const neighbor of neighbors) {
        if (unvisited.has(neighbor.node)) {
          const distance = distances.get(current) + neighbor.weight;
          if (distance < distances.get(neighbor.node)) {
            distances.set(neighbor.node, distance);
            previous.set(neighbor.node, current);
          }
        }
      }
    }

    // 构建路径
    const path = [];
    let current = endVertex;

    while (current !== null) {
      path.unshift(current);
      current = previous.get(current);
    }

    return {
      distance: distances.get(endVertex),
      path: path[0] === startVertex ? path : [],
      distances
    };
  }
}

// 使用示例
const graph = new Graph();
graph.addEdge('A', 'B', 4);
graph.addEdge('A', 'C', 2);
graph.addEdge('B', 'C', 1);
graph.addEdge('B', 'D', 5);
graph.addEdge('C', 'D', 8);
graph.addEdge('C', 'E', 10);
graph.addEdge('D', 'E', 2);

console.log('DFS遍历:', graph.dfs('A'));
console.log('BFS遍历:', graph.bfs('A'));
console.log('最短路径:', graph.dijkstra('A', 'E'));
```

    for (let i = 0; i < arr.length; i++) {

          duplicates.push(arr[i]);

        seen.add(item);
      }
    }

      }
    }
    return -1;
  }

  // 优化后：二分搜索
  binarySearchOptimized(arr, target) {
    let left = 0;
    let right = arr.length - 1;

    while (left <= right) {
      const mid = Math.floor((left + right) / 2);

      if (arr[mid] === target) {
        return mid;
      } else if (arr[mid] < target) {

        resolve(this.expensiveOperation(item));

    for (const { role, permissions } of roleHierarchy) {
      if (user.permissions.some(perm => permissions.includes(perm))) {
        return role;
      }
    }

    return 'guest';
  }

  // 重构前：低效的数组操作
  filterActiveUsersBefore(users) {
    const activeUsers = [];
    for (let i = 0; i < users.length; i++) {
      if (users[i].status === 'active' && users[i].lastLogin > Date.now() - 30 * 24 * 60 * 60 * 1000) {
        activeUsers.push(users[i]);
      }
    }
    return activeUsers;
  }

  // 重构后：函数式编程
  filterActiveUsersAfter(users) {
    const thirtyDaysAgo = Date.now() - 30 * 24 * 60 * 60 * 1000;

    return users.filter(user =>
      user.status === 'active' && user.lastLogin > thirtyDaysAgo
    );
  }

  // 重构前：深层嵌套的回调
  fetchUserDataBefore(userId, callback) {
    setTimeout(() => {
      const user = { id: userId, name: 'User ' + userId };

      setTimeout(() => {
        const profile = { userId, age: 25, city: 'New York' };

        setTimeout(() => {
          const preferences = { userId, theme: 'dark', language: 'en' };
          callback({ user, profile, preferences });
        }, 100);
      }, 100);
    }, 100);
  }

  // 重构后：Promise
  fetchUserDataAfter(userId) {
    return new Promise((resolve) => {
      setTimeout(() => {
        const user = { id: userId, name: 'User ' + userId };
        resolve(user);
      }, 100);
    })
    .then(user => {
      return new Promise((resolve) => {
        setTimeout(() => {
          const profile = { userId: user.id, age: 25, city: 'New York' };
          resolve({ user, profile });
        }, 100);
      });
    })
    .then(({ user, profile }) => {
      return new Promise((resolve) => {
        setTimeout(() => {
          const preferences = { userId: user.id, theme: 'dark', language: 'en' };
          resolve({ user, profile, preferences });
        }, 100);
      });
    });
  }

  // 重构后：async/await
  async fetchUserDataAsync(userId) {
    const user = await this.fetchUser(userId);
    const profile = await this.fetchProfile(user.id);
    const preferences = await this.fetchPreferences(user.id);

    return { user, profile, preferences };
  }

  async fetchUser(userId) {
    return new Promise((resolve) => {
      setTimeout(() => {
        resolve({ id: userId, name: 'User ' + userId });
      }, 100);
    });
  }

  async fetchProfile(userId) {
    return new Promise((resolve) => {
      setTimeout(() => {
        resolve({ userId, age: 25, city: 'New York' });
      }, 100);
    });
  }

  async fetchPreferences(userId) {
    return new Promise((resolve) => {
      setTimeout(() => {
        resolve({ userId, theme: 'dark', language: 'en' });
      }, 100);
    });
  }
}
