// 手动捕获异常示例 function processPayment(paymentData) { try { const result = await paymentAPI.createOrder(paymentData); return result; } catch (error) { // 为错误添加业务上下文 Sentry.withScope(scope => { scope.setTag('operation', 'payment'); scope.setLevel('error'); scope.setContext('paymentData', { amount: paymentData.amount, method: paymentData.method, // 不记录敏感信息如卡号 }); Sentry.captureException(error); }); throw error; } } // 记录业务警告 function logBusinessWarning(message, context) { Sentry.withScope(scope => { scope.setLevel('warning'); scope.setContext('business', context); Sentry.captureMessage(message); }); } // 使用示例 logBusinessWarning('库存不足警告', { productId: 'P123', requestedQty: 5, availableQty: 2 });

// 传统调试方式 - 问题重重 // 1. 只能在开发者工具中看到 console.log('支付参数:', { orderId: 'ORD123', amount: 299.99, method: 'alipay' }); // 2. 用户环境无法获取日志 fetch('/api/payment', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(paymentData) }) .then(res => { console.log('响应状态:', res.status); if (!res.ok) { // 3. 错误信息不完整 console.error('支付接口失败'); } return res.json(); }) .catch(err => { // 4. 缺少用户上下文 console.error('网络错误:', err.message); alert('支付失败，请重试'); // 用户体验差 });

✅ Sentry 方式

  • 实时错误告警
  • 完整请求参数
  • 用户操作路径
  • 设备和环境信息
  • 耗时：10-30 分钟
// Sentry 智能监控 - 全面覆盖 // 1. 全局拦截器自动捕获 axios.interceptors.response.use( response => response, error => { Sentry.withScope(scope => { // 2. 丰富的错误上下文 scope.setTag('api_type', 'payment'); scope.setTag('error_type', error.response?.status || 'network'); scope.setLevel('error'); // 3. 完整的请求信息 scope.setContext('request_details', { url: error.config.url, method: error.config.method, headers: error.config.headers, timeout: error.config.timeout, // 脱敏后的数据 dataSize: JSON.stringify(error.config.data).length }); // 4. 响应信息 if (error.response) { scope.setContext('response_details', { status: error.response.status, statusText: error.response.statusText, data: error.response.data }); } // 5. 用户操作路径 scope.addBreadcrumb({ message: '支付接口调用失败', category: 'api', level: 'error', data: { orderId: error.config.data?.orderId, amount: error.config.data?.amount } }); Sentry.captureException(error); }); // 6. 用户友好的错误处理 showUserFriendlyError(error); return Promise.reject(error); } ); // 7. 智能错误提示 function showUserFriendlyError(error) { const status = error.response?.status; let message = '支付遇到问题，请稍后重试'; if (status === 400) message = '支付信息有误，请检查后重试'; if (status === 500) message = '服务暂时不可用，请稍后重试'; // 同时记录用户看到的错误信息 Sentry.addBreadcrumb({ message: `用户看到错误提示: ${message}`, level: 'info' }); }

🔄 版本升级指南

v6 → v7 升级

主要变更：

  • 包名变更：@sentry/browser
  • 初始化方式改变
  • API 重构
// v6 → v7 升级详细指南 // === 第一步：依赖升级 === // package.json 变更 { "dependencies": { // 旧版本 "@sentry/browser": "^6.19.7", "@sentry/tracing": "^6.19.7" // 新版本 "@sentry/browser": "^7.0.0", "@sentry/tracing": "^7.0.0" } } // === 第二步：导入方式变更 === // ❌ v6 旧写法 import * as Sentry from '@sentry/browser'; import { Integrations } from '@sentry/tracing'; Sentry.init({ dsn: 'https://your-dsn@sentry.io/project-id', integrations: [ new Integrations.BrowserTracing({ tracingOrigins: ['localhost', 'api.example.com'] }) ], tracesSampleRate: 1.0, beforeSend(event) { // v6 只有 event 参数 if (event.exception) { console.log('捕获到异常:', event.exception); } return event; } }); // ✅ v7 新写法 import { init, captureException, addBreadcrumb } from '@sentry/browser'; import { BrowserTracing } from '@sentry/tracing'; init({ dsn: 'https://your-dsn@sentry.io/project-id', integrations: [ // 注意：不再需要 Integrations 命名空间 new BrowserTracing({ tracingOrigins: ['localhost', 'api.example.com'], // v7 新增配置项 startTransactionOnLocationChange: true, startTransactionOnPageLoad: true }) ], tracesSampleRate: 0.1, // 生产环境建议降低采样率 // v7 新增 hint 参数 beforeSend(event, hint) { // hint 包含原始异常信息 if (hint.originalException) { console.log('原始异常:', hint.originalException); } // v7 改进的事件过滤 if (event.exception) { const error = hint.originalException; // 过滤掉网络错误 if (error && error.message?.includes('Network Error')) { return null; } } return event; }, // v7 新增配置项 environment: process.env.NODE_ENV, release: process.env.REACT_APP_VERSION, // 改进的错误过滤 ignoreErrors: [ 'ResizeObserver loop limit exceeded', 'Script error.', 'Non-Error promise rejection captured' ] });

📋 团队使用规范

🏷️ 标签规范

  • module: 业务模块 (order, payment, user)
  • feature: 功能特性 (checkout, login)
  • page: 页面标识 (OrderPage, HomePage)
  • api: 接口标识 (getUserInfo, createOrder)

🔄 处理流程

  1. 错误分配给相关开发者
  2. 24小时内响应和评估
  3. 根据优先级制定修复计划
  4. 修复后验证和关闭
  5. 定期回顾和总结

⚠️ 告警策略

  • 紧急：支付、登录等核心功能
  • 重要：主要业务流程异常
  • 一般：非核心功能问题
  • 忽略：第三方脚本、网络超时

📊 监控指标

  • 错误率 < 1%
  • 平均响应时间 < 24h
  • 解决率 > 95%
  • 用户满意度 > 90%
