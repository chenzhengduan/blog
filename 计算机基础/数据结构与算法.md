## 1. 概述

数据结构与算法是计算机科学的核心基础，它们决定了程序的效率和可维护性。数据结构是组织和存储数据的方式，而算法是解决问题的步骤和方法。

### 1.1 数据结构的重要性

- **效率优化**：合适的数据结构能显著提高程序性能
- **内存管理**：有效利用内存空间
- **代码可读性**：清晰的数据组织方式
- **问题抽象**：将复杂问题简化为数据操作

### 1.2 算法的重要性

- **问题求解**：提供系统化的解决方案
- **性能保证**：确保程序在合理时间内完成
- **资源优化**：最小化时间和空间消耗
- **可扩展性**：支持大规模数据处理

## 2. 线性数据结构

线性数据结构中的元素存在一对一的关系，每个元素最多有一个前驱和一个后继。这类数据结构在内存中可以是连续的（如数组）或非连续的（如链表）。

### 2.1 数组（Array）

数组是最基本的数据结构，元素在内存中连续存储，支持随机访问。

#### 2.1.1 数组的特点

- **连续存储**：元素在内存中连续分布
- **随机访问**：通过索引可以O(1)时间访问任意元素
- **固定大小**：静态数组大小在编译时确定
- **类型统一**：所有元素必须是同一类型

#### 2.1.2 数组操作实现

```c
#include <stdio.h>
#include <stdlib.h>

// 动态数组结构
typedef struct {
    int *data;      // 数据指针
    int size;       // 当前元素个数
    int capacity;   // 数组容量
} DynamicArray;

// 创建动态数组
DynamicArray* createArray(int initial_capacity) {
    DynamicArray *arr = (DynamicArray*)malloc(sizeof(DynamicArray));
    arr->data = (int*)malloc(initial_capacity * sizeof(int));
    arr->size = 0;
    arr->capacity = initial_capacity;
    return arr;
}

// 扩容操作
void resize(DynamicArray *arr) {
    arr->capacity *= 2;
    arr->data = (int*)realloc(arr->data, arr->capacity * sizeof(int));
}

// 插入元素
void insert(DynamicArray *arr, int index, int value) {
    if (arr->size >= arr->capacity) {
        resize(arr);
    }
    
    // 移动元素
    for (int i = arr->size; i > index; i--) {
        arr->data[i] = arr->data[i-1];
    }
    
    arr->data[index] = value;
    arr->size++;
}

// 删除元素
void delete(DynamicArray *arr, int index) {
    if (index < 0 || index >= arr->size) return;
    
    // 移动元素
    for (int i = index; i < arr->size - 1; i++) {
        arr->data[i] = arr->data[i+1];
    }
    
    arr->size--;
}

// 查找元素
int search(DynamicArray *arr, int value) {
    for (int i = 0; i < arr->size; i++) {
        if (arr->data[i] == value) {
            return i;
        }
    }
    return -1; // 未找到
}

// 打印数组
void printArray(DynamicArray *arr) {
    printf("Array: [");
    for (int i = 0; i < arr->size; i++) {
        printf("%d", arr->data[i]);
        if (i < arr->size - 1) printf(", ");
    }
    printf("]\n");
}
```

#### 2.1.3 数组复杂度分析

| 操作 | 时间复杂度 | 说明 |
|------|------------|------|
| 访问 | O(1) | 通过索引直接访问 |
| 查找 | O(n) | 需要遍历数组 |
| 插入 | O(n) | 需要移动元素 |
| 删除 | O(n) | 需要移动元素 |
| 尾部插入 | O(1) | 摊销复杂度 |

### 2.2 链表（Linked List）

链表通过指针连接元素，支持动态内存分配，是一种非连续存储的线性数据结构。

#### 2.2.1 单向链表

```c
#include <stdio.h>
#include <stdlib.h>

// 链表节点结构
typedef struct ListNode {
    int data;
    struct ListNode *next;
} ListNode;

// 链表结构
typedef struct {
    ListNode *head;
    int size;
} LinkedList;

// 创建链表
LinkedList* createList() {
    LinkedList *list = (LinkedList*)malloc(sizeof(LinkedList));
    list->head = NULL;
    list->size = 0;
    return list;
}

// 创建新节点
ListNode* createNode(int data) {
    ListNode *node = (ListNode*)malloc(sizeof(ListNode));
    node->data = data;
    node->next = NULL;
    return node;
}

// 头部插入
void insertAtHead(LinkedList *list, int data) {
    ListNode *newNode = createNode(data);
    newNode->next = list->head;
    list->head = newNode;
    list->size++;
}

// 尾部插入
void insertAtTail(LinkedList *list, int data) {
    ListNode *newNode = createNode(data);
    
    if (list->head == NULL) {
        list->head = newNode;
    } else {
        ListNode *current = list->head;
        while (current->next != NULL) {
            current = current->next;
        }
        current->next = newNode;
    }
    list->size++;
}

// 指定位置插入
void insertAtIndex(LinkedList *list, int index, int data) {
    if (index < 0 || index > list->size) return;
    
    if (index == 0) {
        insertAtHead(list, data);
        return;
    }
    
    ListNode *newNode = createNode(data);
    ListNode *current = list->head;
    
    for (int i = 0; i < index - 1; i++) {
        current = current->next;
    }
    
    newNode->next = current->next;
    current->next = newNode;
    list->size++;
}

// 删除节点
void deleteNode(LinkedList *list, int data) {
    if (list->head == NULL) return;
    
    // 删除头节点
    if (list->head->data == data) {
        ListNode *temp = list->head;
        list->head = list->head->next;
        free(temp);
        list->size--;
        return;
    }
    
    ListNode *current = list->head;
    while (current->next != NULL && current->next->data != data) {
        current = current->next;
    }
    
    if (current->next != NULL) {
        ListNode *temp = current->next;
        current->next = current->next->next;
        free(temp);
        list->size--;
    }
}

// 查找节点
ListNode* search(LinkedList *list, int data) {
    ListNode *current = list->head;
    while (current != NULL) {
        if (current->data == data) {
            return current;
        }
        current = current->next;
    }
    return NULL;
}

// 反转链表
void reverse(LinkedList *list) {
    ListNode *prev = NULL;
    ListNode *current = list->head;
    ListNode *next = NULL;
    
    while (current != NULL) {
        next = current->next;
        current->next = prev;
        prev = current;
        current = next;
    }
    
    list->head = prev;
}

// 打印链表
void printList(LinkedList *list) {
    printf("List: ");
    ListNode *current = list->head;
    while (current != NULL) {
        printf("%d", current->data);
        if (current->next != NULL) printf(" -> ");
        current = current->next;
    }
    printf(" -> NULL\n");
}
```

### 2.3 栈（Stack）

栈是后进先出（LIFO）的数据结构，只能在一端（栈顶）进行插入和删除操作。

#### 2.3.1 数组实现栈

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

#define MAX_SIZE 100

// 数组栈结构
typedef struct {
    int data[MAX_SIZE];
    int top;
} ArrayStack;

// 创建栈
ArrayStack* createArrayStack() {
    ArrayStack *stack = (ArrayStack*)malloc(sizeof(ArrayStack));
    stack->top = -1;
    return stack;
}

// 判断栈是否为空
bool isEmpty(ArrayStack *stack) {
    return stack->top == -1;
}

// 判断栈是否已满
bool isFull(ArrayStack *stack) {
    return stack->top == MAX_SIZE - 1;
}

// 入栈
void push(ArrayStack *stack, int data) {
    if (isFull(stack)) {
        printf("Stack overflow!\n");
        return;
    }
    stack->data[++stack->top] = data;
}

// 出栈
int pop(ArrayStack *stack) {
    if (isEmpty(stack)) {
        printf("Stack underflow!\n");
        return -1;
    }
    return stack->data[stack->top--];
}

// 查看栈顶元素
int peek(ArrayStack *stack) {
    if (isEmpty(stack)) {
        printf("Stack is empty!\n");
        return -1;
    }
    return stack->data[stack->top];
}

// 获取栈大小
int size(ArrayStack *stack) {
    return stack->top + 1;
}
```

#### 2.3.2 链表实现栈

```c
// 链表栈节点
typedef struct StackNode {
    int data;
    struct StackNode *next;
} StackNode;

// 链表栈结构
typedef struct {
    StackNode *top;
    int size;
} LinkedStack;

// 创建链表栈
LinkedStack* createLinkedStack() {
    LinkedStack *stack = (LinkedStack*)malloc(sizeof(LinkedStack));
    stack->top = NULL;
    stack->size = 0;
    return stack;
}

// 入栈
void pushLinked(LinkedStack *stack, int data) {
    StackNode *newNode = (StackNode*)malloc(sizeof(StackNode));
    newNode->data = data;
    newNode->next = stack->top;
    stack->top = newNode;
    stack->size++;
}

// 出栈
int popLinked(LinkedStack *stack) {
    if (stack->top == NULL) {
        printf("Stack underflow!\n");
        return -1;
    }
    
    StackNode *temp = stack->top;
    int data = temp->data;
    stack->top = stack->top->next;
    free(temp);
    stack->size--;
    return data;
}
```

#### 2.3.3 栈的应用

**1. 表达式求值**

```c
// 中缀表达式转后缀表达式
void infixToPostfix(char *infix, char *postfix) {
    ArrayStack *stack = createArrayStack();
    int i = 0, j = 0;
    
    while (infix[i] != '\0') {
        char ch = infix[i];
        
        if (isalnum(ch)) {
            postfix[j++] = ch;
        } else if (ch == '(') {
            push(stack, ch);
        } else if (ch == ')') {
            while (!isEmpty(stack) && peek(stack) != '(') {
                postfix[j++] = pop(stack);
            }
            pop(stack); // 弹出 '('
        } else { // 操作符
            while (!isEmpty(stack) && precedence(peek(stack)) >= precedence(ch)) {
                postfix[j++] = pop(stack);
            }
            push(stack, ch);
        }
        i++;
    }
    
    while (!isEmpty(stack)) {
        postfix[j++] = pop(stack);
    }
    postfix[j] = '\0';
}

// 操作符优先级
int precedence(char op) {
    switch (op) {
        case '+': case '-': return 1;
        case '*': case '/': return 2;
        case '^': return 3;
        default: return 0;
    }
}
```

**2. 括号匹配**

```c
// 检查括号是否匹配
bool isValidParentheses(char *str) {
    ArrayStack *stack = createArrayStack();
    
    for (int i = 0; str[i] != '\0'; i++) {
        char ch = str[i];
        
        if (ch == '(' || ch == '[' || ch == '{') {
            push(stack, ch);
        } else if (ch == ')' || ch == ']' || ch == '}') {
            if (isEmpty(stack)) return false;
            
            char top = pop(stack);
            if ((ch == ')' && top != '(') ||
                (ch == ']' && top != '[') ||
                (ch == '}' && top != '{')) {
                return false;
            }
        }
    }
    
    return isEmpty(stack);
}
```

### 2.4 队列（Queue）

队列是先进先出（FIFO）的数据结构，在一端（队尾）插入元素，在另一端（队头）删除元素。

#### 2.4.1 数组实现队列

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

#define MAX_SIZE 100

// 循环队列结构
typedef struct {
    int data[MAX_SIZE];
    int front;
    int rear;
    int size;
} CircularQueue;

// 创建循环队列
CircularQueue* createCircularQueue() {
    CircularQueue *queue = (CircularQueue*)malloc(sizeof(CircularQueue));
    queue->front = 0;
    queue->rear = 0;
    queue->size = 0;
    return queue;
}

// 判断队列是否为空
bool isQueueEmpty(CircularQueue *queue) {
    return queue->size == 0;
}

// 判断队列是否已满
bool isQueueFull(CircularQueue *queue) {
    return queue->size == MAX_SIZE;
}

// 入队
void enqueue(CircularQueue *queue, int data) {
    if (isQueueFull(queue)) {
        printf("Queue overflow!\n");
        return;
    }
    
    queue->data[queue->rear] = data;
    queue->rear = (queue->rear + 1) % MAX_SIZE;
    queue->size++;
}

// 出队
int dequeue(CircularQueue *queue) {
    if (isQueueEmpty(queue)) {
        printf("Queue underflow!\n");
        return -1;
    }
    
    int data = queue->data[queue->front];
    queue->front = (queue->front + 1) % MAX_SIZE;
    queue->size--;
    return data;
}

// 查看队头元素
int front(CircularQueue *queue) {
    if (isQueueEmpty(queue)) {
        printf("Queue is empty!\n");
        return -1;
    }
    return queue->data[queue->front];
}
```

#### 2.4.2 链表实现队列

```c
// 队列节点
typedef struct QueueNode {
    int data;
    struct QueueNode *next;
} QueueNode;

// 链表队列结构
typedef struct {
    QueueNode *front;
    QueueNode *rear;
    int size;
} LinkedQueue;

// 创建链表队列
LinkedQueue* createLinkedQueue() {
    LinkedQueue *queue = (LinkedQueue*)malloc(sizeof(LinkedQueue));
    queue->front = NULL;
    queue->rear = NULL;
    queue->size = 0;
    return queue;
}

// 入队
void enqueueLinked(LinkedQueue *queue, int data) {
    QueueNode *newNode = (QueueNode*)malloc(sizeof(QueueNode));
    newNode->data = data;
    newNode->next = NULL;
    
    if (queue->rear == NULL) {
        queue->front = queue->rear = newNode;
    } else {
        queue->rear->next = newNode;
        queue->rear = newNode;
    }
    queue->size++;
}

// 出队
int dequeueLinked(LinkedQueue *queue) {
    if (queue->front == NULL) {
        printf("Queue underflow!\n");
        return -1;
    }
    
    QueueNode *temp = queue->front;
    int data = temp->data;
    queue->front = queue->front->next;
    
    if (queue->front == NULL) {
        queue->rear = NULL;
    }
    
    free(temp);
    queue->size--;
    return data;
}
```

#### 2.4.3 优先队列

```c
// 优先队列元素
typedef struct {
    int data;
    int priority;
} PriorityElement;

// 优先队列结构
typedef struct {
    PriorityElement elements[MAX_SIZE];
    int size;
} PriorityQueue;

// 创建优先队列
PriorityQueue* createPriorityQueue() {
    PriorityQueue *pq = (PriorityQueue*)malloc(sizeof(PriorityQueue));
    pq->size = 0;
    return pq;
}

// 入队（按优先级插入）
void enqueuePriority(PriorityQueue *pq, int data, int priority) {
    if (pq->size >= MAX_SIZE) {
        printf("Priority queue overflow!\n");
        return;
    }
    
    int i = pq->size - 1;
    
    // 找到插入位置
    while (i >= 0 && pq->elements[i].priority < priority) {
        pq->elements[i + 1] = pq->elements[i];
        i--;
    }
    
    pq->elements[i + 1].data = data;
    pq->elements[i + 1].priority = priority;
    pq->size++;
}

// 出队（移除最高优先级元素）
int dequeuePriority(PriorityQueue *pq) {
    if (pq->size == 0) {
        printf("Priority queue underflow!\n");
        return -1;
    }
    
    int data = pq->elements[0].data;
    
    for (int i = 1; i < pq->size; i++) {
        pq->elements[i - 1] = pq->elements[i];
    }
    
    pq->size--;
    return data;
}
```

#### 2.4.4 队列的应用

**1. 广度优先搜索（BFS）**

```c
// BFS遍历图
void BFS(int graph[][MAX_VERTICES], int vertices, int start) {
    bool visited[MAX_VERTICES] = {false};
    CircularQueue *queue = createCircularQueue();
    
    visited[start] = true;
    enqueue(queue, start);
    
    printf("BFS traversal: ");
    
    while (!isQueueEmpty(queue)) {
        int vertex = dequeue(queue);
        printf("%d ", vertex);
        
        for (int i = 0; i < vertices; i++) {
            if (graph[vertex][i] == 1 && !visited[i]) {
                visited[i] = true;
                enqueue(queue, i);
            }
        }
    }
    printf("\n");
}
```

**2. 任务调度**

```c
// 任务结构
typedef struct {
    int id;
    int arrival_time;
    int burst_time;
    int remaining_time;
} Task;

// 时间片轮转调度
void roundRobinScheduling(Task tasks[], int n, int time_quantum) {
    LinkedQueue *queue = createLinkedQueue();
    int current_time = 0;
    
    // 将所有任务加入队列
    for (int i = 0; i < n; i++) {
        enqueueLinked(queue, i);
    }
    
    while (!isQueueEmpty(queue)) {
        int task_index = dequeueLinked(queue);
        Task *current_task = &tasks[task_index];
        
        int execution_time = (current_task->remaining_time > time_quantum) 
                           ? time_quantum : current_task->remaining_time;
        
        current_task->remaining_time -= execution_time;
        current_time += execution_time;
        
        printf("Task %d executed for %d units, remaining: %d\n", 
               current_task->id, execution_time, current_task->remaining_time);
        
        if (current_task->remaining_time > 0) {
            enqueueLinked(queue, task_index);
        }
    }
}
```

## 3. 非线性数据结构

非线性数据结构中的元素存在一对多或多对多的关系，能够更好地表示现实世界中的复杂关系。

### 3.1 树（Tree）

树是层次化的数据结构，具有根节点和子树，广泛应用于文件系统、数据库索引、编译器等领域。

#### 3.1.1 二叉树基本概念

```c
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>

// 二叉树节点结构
typedef struct TreeNode {
    int data;
    struct TreeNode *left;
    struct TreeNode *right;
} TreeNode;

// 创建新节点
TreeNode* createTreeNode(int data) {
    TreeNode *node = (TreeNode*)malloc(sizeof(TreeNode));
    node->data = data;
    node->left = NULL;
    node->right = NULL;
    return node;
}

// 插入节点（二叉搜索树）
TreeNode* insert(TreeNode *root, int data) {
    if (root == NULL) {
        return createTreeNode(data);
    }
    
    if (data < root->data) {
        root->left = insert(root->left, data);
    } else if (data > root->data) {
        root->right = insert(root->right, data);
    }
    
    return root;
}

// 查找节点
TreeNode* search(TreeNode *root, int data) {
    if (root == NULL || root->data == data) {
        return root;
    }
    
    if (data < root->data) {
        return search(root->left, data);
    }
    
    return search(root->right, data);
}

// 找到最小值节点
TreeNode* findMin(TreeNode *root) {
    while (root && root->left) {
        root = root->left;
    }
    return root;
}

// 删除节点
TreeNode* deleteNode(TreeNode *root, int data) {
    if (root == NULL) return root;
    
    if (data < root->data) {
        root->left = deleteNode(root->left, data);
    } else if (data > root->data) {
        root->right = deleteNode(root->right, data);
    } else {
        // 找到要删除的节点
        if (root->left == NULL) {
            TreeNode *temp = root->right;
            free(root);
            return temp;
        } else if (root->right == NULL) {
            TreeNode *temp = root->left;
            free(root);
            return temp;
        }
        
        // 有两个子节点的情况
        TreeNode *temp = findMin(root->right);
        root->data = temp->data;
        root->right = deleteNode(root->right, temp->data);
    }
    return root;
}
```

#### 3.1.3 AVL树（自平衡二叉搜索树）

```c
// AVL树节点
typedef struct AVLNode {
    int data;
    int height;
    struct AVLNode *left;
    struct AVLNode *right;
} AVLNode;

// 获取节点高度
int getHeight(AVLNode *node) {
    if (node == NULL) return 0;
    return node->height;
}

// 获取平衡因子
int getBalance(AVLNode *node) {
    if (node == NULL) return 0;
    return getHeight(node->left) - getHeight(node->right);
}

// 更新节点高度
void updateHeight(AVLNode *node) {
    if (node != NULL) {
        int leftHeight = getHeight(node->left);
        int rightHeight = getHeight(node->right);
        node->height = 1 + (leftHeight > rightHeight ? leftHeight : rightHeight);
    }
}

// 右旋转
AVLNode* rightRotate(AVLNode *y) {
    AVLNode *x = y->left;
    AVLNode *T2 = x->right;
    
    // 执行旋转
    x->right = y;
    y->left = T2;
    
    // 更新高度
    updateHeight(y);
    updateHeight(x);
    
    return x;
}

// 左旋转
AVLNode* leftRotate(AVLNode *x) {
    AVLNode *y = x->right;
    AVLNode *T2 = y->left;
    
    // 执行旋转
    y->left = x;
    x->right = T2;
    
    // 更新高度
    updateHeight(x);
    updateHeight(y);
    
    return y;
}

// AVL插入
AVLNode* avlInsert(AVLNode *node, int data) {
    // 1. 执行标准BST插入
    if (node == NULL) {
        AVLNode *newNode = (AVLNode*)malloc(sizeof(AVLNode));
        newNode->data = data;
        newNode->height = 1;
        newNode->left = NULL;
        newNode->right = NULL;
        return newNode;
    }
    
    if (data < node->data) {
        node->left = avlInsert(node->left, data);
    } else if (data > node->data) {
        node->right = avlInsert(node->right, data);
    } else {
        return node; // 重复值不插入
    }
    
    // 2. 更新当前节点高度
    updateHeight(node);
    
    // 3. 获取平衡因子
    int balance = getBalance(node);
    
    // 4. 如果不平衡，进行旋转
    // Left Left Case
    if (balance > 1 && data < node->left->data) {
        return rightRotate(node);
    }
    
    // Right Right Case
    if (balance < -1 && data > node->right->data) {
        return leftRotate(node);
    }
    
    // Left Right Case
    if (balance > 1 && data > node->left->data) {
        node->left = leftRotate(node->left);
        return rightRotate(node);
    }
    
    // Right Left Case
    if (balance < -1 && data < node->right->data) {
        node->right = rightRotate(node->right);
        return leftRotate(node);
    }
    
    return node;
}
```

### 3.3 堆（Heap）

堆是完全二叉树，满足堆性质（父节点的值总是大于等于或小于等于子节点的值）。

#### 3.3.1 最大堆实现

```c
#define MAX_HEAP_SIZE 100

// 最大堆结构
typedef struct {
    int data[MAX_HEAP_SIZE];
    int size;
} MaxHeap;

// 创建最大堆
MaxHeap* createMaxHeap() {
    MaxHeap *heap = (MaxHeap*)malloc(sizeof(MaxHeap));
    heap->size = 0;
    return heap;
}

// 获取父节点索引
int parent(int i) {
    return (i - 1) / 2;
}

// 获取左子节点索引
int leftChild(int i) {
    return 2 * i + 1;
}

// 获取右子节点索引
int rightChild(int i) {
    return 2 * i + 2;
}

// 交换两个元素
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

// 向上调整（用于插入）
void heapifyUp(MaxHeap *heap, int index) {
    while (index > 0 && heap->data[parent(index)] < heap->data[index]) {
        swap(&heap->data[parent(index)], &heap->data[index]);
        index = parent(index);
    }
}

// 向下调整（用于删除）
void heapifyDown(MaxHeap *heap, int index) {
    int largest = index;
    int left = leftChild(index);
    int right = rightChild(index);
    
    if (left < heap->size && heap->data[left] > heap->data[largest]) {
        largest = left;
    }
    
    if (right < heap->size && heap->data[right] > heap->data[largest]) {
        largest = right;
    }
    
    if (largest != index) {
        swap(&heap->data[index], &heap->data[largest]);
        heapifyDown(heap, largest);
    }
}

// 插入元素
void insertHeap(MaxHeap *heap, int value) {
    if (heap->size >= MAX_HEAP_SIZE) {
        printf("Heap overflow!\n");
        return;
    }
    
    heap->data[heap->size] = value;
    heapifyUp(heap, heap->size);
    heap->size++;
}

// 删除最大元素
int extractMax(MaxHeap *heap) {
    if (heap->size <= 0) {
        printf("Heap underflow!\n");
        return -1;
    }
    
    int max = heap->data[0];
    heap->data[0] = heap->data[heap->size - 1];
    heap->size--;
    heapifyDown(heap, 0);
    
    return max;
}

// 获取最大元素
int getMax(MaxHeap *heap) {
    if (heap->size <= 0) {
        printf("Heap is empty!\n");
        return -1;
    }
    return heap->data[0];
}

// 构建堆（从数组）
void buildMaxHeap(MaxHeap *heap, int arr[], int n) {
    heap->size = n;
    for (int i = 0; i < n; i++) {
        heap->data[i] = arr[i];
    }
    
    // 从最后一个非叶子节点开始向下调整
    for (int i = (n / 2) - 1; i >= 0; i--) {
        heapifyDown(heap, i);
    }
}
```

### 3.4 哈希表（Hash Table）

哈希表通过哈希函数实现快速查找，平均时间复杂度为O(1)。

#### 3.4.1 链地址法解决冲突

```c

    table->size = 0;
    
    for (int i = 0; i < HASH_TABLE_SIZE; i++) {
        table->buckets[i] = NULL;
    }
    
    return table;
}

// 哈希函数
int hash(int key) {
    return abs(key) % HASH_TABLE_SIZE;
}

// 创建哈希节点
HashNode* createHashNode(int key, int value) {
    HashNode *node = (HashNode*)malloc(sizeof(HashNode));
    node->key = key;
    node->value = value;
    node->next = NULL;
    return node;
}

// 插入键值对
void insertHash(HashTable *table, int key, int value) {
    int index = hash(key);
    HashNode *newNode = createHashNode(key, value);
    
    // 如果桶为空，直接插入
    if (table->buckets[index] == NULL) {
        table->buckets[index] = newNode;
        table->size++;
        return;
    }
    
    // 检查是否已存在该键
    HashNode *current = table->buckets[index];
    while (current != NULL) {
        if (current->key == key) {
            current->value = value; // 更新值
            free(newNode);
            return;
        }
        current = current->next;
    }
    
    // 插入到链表头部
    newNode->next = table->buckets[index];
    table->buckets[index] = newNode;
    table->size++;
}

// 查找值
int searchHash(HashTable *table, int key) {
    int index = hash(key);
    HashNode *current = table->buckets[index];
    
    while (current != NULL) {

            return current->value;
        }
        current = current->next;

}

        if (current->key == key) {
            if (prev == NULL) {
                table->buckets[index] = current->next;

                prev->next = current->next;
            }

            return true;
        }

        current = current->next;
```

## 4. 排序算法

排序算法是将一组数据按照特定顺序排列的算法，是计算机科学中最基础和重要的算法之一。

### 4.1 冒泡排序（Bubble Sort）

冒泡排序通过相邻元素的比较和交换来排序，较大的元素会像气泡一样"冒"到数组的末尾。

```c
// 冒泡排序实现
void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        bool swapped = false;
        
        // 每轮将最大元素"冒泡"到末尾
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // 交换相邻元素
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = true;
            }
        }
        
        // 如果没有发生交换，说明数组已经有序
        if (!swapped) {
            break;
        }
    }
}

// 优化版本：双向冒泡排序（鸡尾酒排序）
void cocktailSort(int arr[], int n) {
    int left = 0, right = n - 1;
    
    while (left < right) {
        // 从左到右冒泡最大值
        for (int i = left; i < right; i++) {
            if (arr[i] > arr[i + 1]) {
                int temp = arr[i];
                arr[i] = arr[i + 1];
                arr[i + 1] = temp;
            }
        }
        right--;
        
        // 从右到左冒泡最小值
        for (int i = right; i > left; i--) {
            if (arr[i] < arr[i - 1]) {
                int temp = arr[i];
                arr[i] = arr[i - 1];
                arr[i - 1] = temp;
            }
        }
        left++;
    }
}
```

**时间复杂度：** O(n²)  
**空间复杂度：** O(1)  
**稳定性：** 稳定

### 4.2 选择排序（Selection Sort）

选择排序每次从未排序部分选择最小（或最大）元素，放到已排序部分的末尾。

```c
// 选择排序实现
void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int minIndex = i;
        
        // 在未排序部分找到最小元素的索引
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        
        // 将最小元素交换到当前位置
        if (minIndex != i) {
            int temp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = temp;
        }
    }
}

// 双向选择排序
void doubleSelectionSort(int arr[], int n) {
    int left = 0, right = n - 1;
    
    while (left < right) {
        int minIndex = left, maxIndex = left;
        
        // 同时找到最小值和最大值
        for (int i = left; i <= right; i++) {
            if (arr[i] < arr[minIndex]) {
                minIndex = i;
            }
            if (arr[i] > arr[maxIndex]) {
                maxIndex = i;
            }
        }
        
        // 将最小值放到左端
        if (minIndex != left) {
            int temp = arr[left];
            arr[left] = arr[minIndex];
            arr[minIndex] = temp;
        }
        
        // 如果最大值在left位置，需要更新maxIndex
        if (maxIndex == left) {
            maxIndex = minIndex;
        }
        
        // 将最大值放到右端
        if (maxIndex != right) {
            int temp = arr[right];
            arr[right] = arr[maxIndex];
            arr[maxIndex] = temp;
        }
        
        left++;
        right--;
    }
}
```

**时间复杂度：** O(n²)  
**空间复杂度：** O(1)  
**稳定性：** 不稳定

### 4.3 插入排序（Insertion Sort）

插入排序将元素逐个插入到已排序部分的正确位置，类似于整理扑克牌。

```c
// 插入排序实现
void insertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i];
        int j = i - 1;
        
        // 将大于key的元素向右移动
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        
        // 插入key到正确位置
        arr[j + 1] = key;
    }
}

// 二分插入排序（优化查找位置）
void binaryInsertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i];
        int left = 0, right = i;
        
        // 二分查找插入位置
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (arr[mid] > key) {
                right = mid;
            } else {
                left = mid + 1;
            }
        }
        
        // 移动元素
        for (int j = i; j > left; j--) {
            arr[j] = arr[j - 1];
        }
        
        // 插入元素
        arr[left] = key;
    }
}

// 希尔排序（插入排序的改进版）
void shellSort(int arr[], int n) {
    // 使用Knuth序列：h = 3*h + 1
    int h = 1;
    while (h < n / 3) {
        h = 3 * h + 1;
    }
    
    while (h >= 1) {
        // 对每个子序列进行插入排序
        for (int i = h; i < n; i++) {
            int key = arr[i];
            int j = i;
            
            while (j >= h && arr[j - h] > key) {
                arr[j] = arr[j - h];
                j -= h;
            }
            
            arr[j] = key;
        }
        
        h = h / 3;
    }
}
```

**时间复杂度：** O(n²)，最好情况O(n)  
**空间复杂度：** O(1)  
**稳定性：** 稳定

### 4.4 快速排序（Quick Sort）

快速排序采用分治策略，选择基准元素进行分区，是实际应用中最常用的排序算法之一。

```c
// 分区函数（Lomuto分区方案）
int partition(int arr[], int low, int high) {
    int pivot = arr[high]; // 选择最后一个元素作为基准
    int i = low - 1; // 小于基准的元素的索引
    
    for (int j = low; j < high; j++) {
        if (arr[j] <= pivot) {
            i++;
            // 交换arr[i]和arr[j]
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
    
    // 将基准元素放到正确位置
    int temp = arr[i + 1];
    arr[i + 1] = arr[high];
    arr[high] = temp;
    
    return i + 1;
}

// 快速排序主函数
void quickSort(int arr[], int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        
        // 递归排序基准元素左右两部分
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

// Hoare分区方案（更高效）
int hoarePartition(int arr[], int low, int high) {
    int pivot = arr[low];
    int i = low - 1, j = high + 1;
    
    while (true) {
        do {
            i++;
        } while (arr[i] < pivot);
        
        do {
            j--;
        } while (arr[j] > pivot);
        
        if (i >= j) {
            return j;
        }
        
        // 交换arr[i]和arr[j]
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }
}

// 三数取中法选择基准
int medianOfThree(int arr[], int low, int high) {
    int mid = low + (high - low) / 2;
    
    if (arr[mid] < arr[low]) {
        int temp = arr[mid];
        arr[mid] = arr[low];
        arr[low] = temp;
    }
    
    if (arr[high] < arr[low]) {
        int temp = arr[high];
        arr[high] = arr[low];
        arr[low] = temp;
    }
    
    if (arr[high] < arr[mid]) {
        int temp = arr[high];
        arr[high] = arr[mid];
        arr[mid] = temp;
    }
    
    // 将中位数放到末尾
    int temp = arr[mid];
    arr[mid] = arr[high];
    arr[high] = temp;
    
    return high;
}

// 优化的快速排序
void optimizedQuickSort(int arr[], int low, int high) {
    while (low < high) {
        // 对于小数组使用插入排序
        if (high - low < 10) {
            for (int i = low + 1; i <= high; i++) {
                int key = arr[i];
                int j = i - 1;
                while (j >= low && arr[j] > key) {
                    arr[j + 1] = arr[j];
                    j--;
                }
                arr[j + 1] = key;
            }
            break;
        }
        
        // 三数取中选择基准
        medianOfThree(arr, low, high);
        int pi = partition(arr, low, high);
        
        // 优化递归：先处理较小的部分
        if (pi - low < high - pi) {
            quickSort(arr, low, pi - 1);
            low = pi + 1;
        } else {
            quickSort(arr, pi + 1, high);
            high = pi - 1;
        }
    }
}
```

**时间复杂度：** 平均O(n log n)，最坏O(n²)  
**空间复杂度：** O(log n)  
**稳定性：** 不稳定

### 4.5 归并排序（Merge Sort）

归并排序采用分治策略，将数组分割成小部分后合并，保证稳定性。

```c
// 合并两个有序数组
void merge(int arr[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    
    // 创建临时数组
    int *leftArr = (int*)malloc(n1 * sizeof(int));
    int *rightArr = (int*)malloc(n2 * sizeof(int));
    
    // 复制数据到临时数组
    for (int i = 0; i < n1; i++) {
        leftArr[i] = arr[left + i];
    }
    for (int j = 0; j < n2; j++) {
        rightArr[j] = arr[mid + 1 + j];
    }
    
    // 合并临时数组回到原数组
    int i = 0, j = 0, k = left;
    
    while (i < n1 && j < n2) {
        if (leftArr[i] <= rightArr[j]) {
            arr[k] = leftArr[i];
            i++;
        } else {
            arr[k] = rightArr[j];
            j++;
        }
        k++;
    }
    
    // 复制剩余元素
    while (i < n1) {
        arr[k] = leftArr[i];
        i++;
        k++;
    }
    
    while (j < n2) {
        arr[k] = rightArr[j];
        j++;
        k++;
    }
    
    free(leftArr);
    free(rightArr);
}

// 归并排序主函数
void mergeSort(int arr[], int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        
        // 递归排序左右两部分
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);
        
        // 合并已排序的部分
        merge(arr, left, mid, right);
    }
}

// 自底向上的归并排序（迭代版本）
void mergeSortIterative(int arr[], int n) {
    int *temp = (int*)malloc(n * sizeof(int));
    
    for (int size = 1; size < n; size *= 2) {
        for (int left = 0; left < n - size; left += 2 * size) {
            int mid = left + size - 1;
            int right = (left + 2 * size - 1 < n - 1) ? left + 2 * size - 1 : n - 1;
            
            // 合并arr[left...mid]和arr[mid+1...right]
            int i = left, j = mid + 1, k = left;
            
            while (i <= mid && j <= right) {
                if (arr[i] <= arr[j]) {
                    temp[k++] = arr[i++];
                } else {
                    temp[k++] = arr[j++];
                }
            }
            
            while (i <= mid) {
                temp[k++] = arr[i++];
            }
            
            while (j <= right) {
                temp[k++] = arr[j++];
            }
            
            // 复制回原数组
            for (int i = left; i <= right; i++) {
                arr[i] = temp[i];
            }
        }
    }
    
    free(temp);
}
```

**时间复杂度：** O(n log n)  
**空间复杂度：** O(n)  
**稳定性：** 稳定

### 4.6 堆排序（Heap Sort）

堆排序利用堆的性质进行排序，首先构建最大堆，然后逐个取出最大元素。

```c
// 调整堆（向下调整）
void heapify(int arr[], int n, int i) {
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    
    // 找到最大值的索引
    if (left < n && arr[left] > arr[largest]) {
        largest = left;
    }
    
    if (right < n && arr[right] > arr[largest]) {
        largest = right;
    }
    
    // 如果最大值不是根节点，交换并继续调整
    if (largest != i) {
        int temp = arr[i];
        arr[i] = arr[largest];
        arr[largest] = temp;
        
        heapify(arr, n, largest);
    }
}

// 堆排序主函数
void heapSort(int arr[], int n) {
    // 构建最大堆
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
    
    // 逐个取出最大元素
    for (int i = n - 1; i > 0; i--) {
        // 将最大元素移到末尾
        int temp = arr[0];
        arr[0] = arr[i];
        arr[i] = temp;
        
        // 重新调整堆
        heapify(arr, i, 0);
    }
}
```

**时间复杂度：** O(n log n)  
**空间复杂度：** O(1)  
**稳定性：** 不稳定

### 4.7 计数排序（Counting Sort）

计数排序是非比较排序算法，适用于整数且范围不大的情况。

```c
// 计数排序实现
void countingSort(int arr[], int n, int maxVal) {
    // 创建计数数组
    int *count = (int*)calloc(maxVal + 1, sizeof(int));
    int *output = (int*)malloc(n * sizeof(int));
    
    // 统计每个元素的出现次数
    for (int i = 0; i < n; i++) {
        count[arr[i]]++;
    }
    
    // 计算累积计数
    for (int i = 1; i <= maxVal; i++) {
        count[i] += count[i - 1];
    }
    
    // 构建输出数组
    for (int i = n - 1; i >= 0; i--) {
        output[count[arr[i]] - 1] = arr[i];
        count[arr[i]]--;
    }
    
    // 复制回原数组
    for (int i = 0; i < n; i++) {
        arr[i] = output[i];
    }
    
    free(count);
    free(output);
}
```

**时间复杂度：** O(n + k)，k为数据范围  
**空间复杂度：** O(k)  
**稳定性：** 稳定

### 4.8 基数排序（Radix Sort）

基数排序按位数进行排序，从最低位到最高位。

```c
// 获取数组中的最大值
int getMax(int arr[], int n) {
    int max = arr[0];
    for (int i = 1; i < n; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

// 按指定位数进行计数排序
void countingSortByDigit(int arr[], int n, int exp) {
    int output[n];
    int count[10] = {0};
    
    // 统计每个数字的出现次数
    for (int i = 0; i < n; i++) {
        count[(arr[i] / exp) % 10]++;
    }
    
    // 计算累积计数
    for (int i = 1; i < 10; i++) {
        count[i] += count[i - 1];
    }
    
    // 构建输出数组
    for (int i = n - 1; i >= 0; i--) {
        output[count[(arr[i] / exp) % 10] - 1] = arr[i];
        count[(arr[i] / exp) % 10]--;
    }
    
    // 复制回原数组
    for (int i = 0; i < n; i++) {
        arr[i] = output[i];
    }
}

// 基数排序主函数
void radixSort(int arr[], int n) {
    int max = getMax(arr, n);
    
    // 对每一位进行计数排序
    for (int exp = 1; max / exp > 0; exp *= 10) {
        countingSortByDigit(arr, n, exp);
    }
}
```

**时间复杂度：** O(d × (n + k))，d为位数  
**空间复杂度：** O(n + k)  
**稳定性：** 稳定

## 5. 查找算法

查找算法用于在数据集合中寻找特定元素，是数据处理中的基本操作。

### 5.1 线性查找（Linear Search）

线性查找逐个检查每个元素直到找到目标，适用于无序数据。

```c
// 线性查找实现
int linearSearch(int arr[], int n, int target) {
    for (int i = 0; i < n; i++) {
        if (arr[i] == target) {
            return i; // 返回找到的索引
        }
    }
    return -1; // 未找到
}

// 线性查找（查找所有匹配项）
int* linearSearchAll(int arr[], int n, int target, int *count) {
    int *result = (int*)malloc(n * sizeof(int));
    *count = 0;
    
    for (int i = 0; i < n; i++) {
        if (arr[i] == target) {
            result[*count] = i;
            (*count)++;
        }
    }
    
    if (*count == 0) {
        free(result);
        return NULL;
    }
    
    // 重新分配内存以节省空间
    result = (int*)realloc(result, (*count) * sizeof(int));
    return result;
}

// 哨兵线性查找（减少边界检查）
int sentinelLinearSearch(int arr[], int n, int target) {
    int last = arr[n - 1];
    arr[n - 1] = target;
    
    int i = 0;
    while (arr[i] != target) {
        i++;
    }
    
    arr[n - 1] = last; // 恢复原值
    
    if (i < n - 1 || arr[n - 1] == target) {
        return i;
    }
    
    return -1;
}
```

**时间复杂度：** O(n)  
**空间复杂度：** O(1)  
**适用场景：** 无序数据、小数据集

### 5.2 二分查找（Binary Search）

二分查找在有序数组中通过比较中间元素来查找，效率很高。

```c
// 递归二分查找
int binarySearchRecursive(int arr[], int left, int right, int target) {
    if (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) {
            return mid;
        }
        
        if (arr[mid] > target) {
            return binarySearchRecursive(arr, left, mid - 1, target);
        }
        
        return binarySearchRecursive(arr, mid + 1, right, target);
    }
    
    return -1;
}

// 迭代二分查找
int binarySearchIterative(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) {
            return mid;
        }
        
        if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return -1;
}

// 查找第一个出现的位置
int findFirst(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    int result = -1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) {
            result = mid;
            right = mid - 1; // 继续在左半部分查找
        } else if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return result;
}

// 查找最后一个出现的位置
int findLast(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    int result = -1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) {
            result = mid;
            left = mid + 1; // 继续在右半部分查找
        } else if (arr[mid] < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    
    return result;
}
```

**时间复杂度：** O(log n)  
**空间复杂度：** O(1)（迭代），O(log n)（递归）  
**适用场景：** 有序数据

### 5.3 哈希查找（Hash Search）

哈希查找通过哈希函数直接定位元素位置，实现常数时间查找。

```c
// 使用前面定义的哈希表结构进行查找
bool hashSearch(HashTable *table, int key) {
    return searchHash(table, key) != -1;
}

// 布隆过滤器（概率性数据结构）
#define BLOOM_SIZE 1000
#define HASH_FUNCTIONS 3

typedef struct {
    bool bits[BLOOM_SIZE];
} BloomFilter;

// 创建布隆过滤器
BloomFilter* createBloomFilter() {
    BloomFilter *filter = (BloomFilter*)malloc(sizeof(BloomFilter));
    for (int i = 0; i < BLOOM_SIZE; i++) {
        filter->bits[i] = false;
    }
    return filter;
}

// 哈希函数1
int hash1(int key) {
    return key % BLOOM_SIZE;
}

// 哈希函数2
int hash2(int key) {
    return (key * 31) % BLOOM_SIZE;
}

// 哈希函数3
int hash3(int key) {
    return (key * 37 + 17) % BLOOM_SIZE;
}

// 添加元素到布隆过滤器
void bloomAdd(BloomFilter *filter, int key) {
    filter->bits[hash1(key)] = true;
    filter->bits[hash2(key)] = true;
    filter->bits[hash3(key)] = true;
}

// 检查元素是否可能存在
bool bloomMightContain(BloomFilter *filter, int key) {
    return filter->bits[hash1(key)] && 
           filter->bits[hash2(key)] && 
           filter->bits[hash3(key)];
}
```

**时间复杂度：** 平均O(1)，最坏O(n)  
**空间复杂度：** O(n)  
**适用场景：** 快速查找，允许哈希冲突

### 5.4 树查找（Tree Search）

在树结构中进行查找，包括二叉搜索树、AVL树等。

```c
// 二叉搜索树查找（使用前面定义的TreeNode结构）
```

### 5.5 插值查找（Interpolation Search）

插值查找是二分查找的改进版，根据数据分布估算查找位置。

```c
// 插值查找实现
int interpolationSearch(int arr[], int n, int target) {
    int left = 0, right = n - 1;
    
    while (left <= right && target >= arr[left] && target <= arr[right]) {
        // 如果数组只有一个元素
        if (left == right) {
            if (arr[left] == target) {
                return left;
            }
            return -1;
        }
        
        // 计算插值位置
        int pos = left + ((double)(target - arr[left]) / 
                         (arr[right] - arr[left])) * (right - left);
        
        if (arr[pos] == target) {
            return pos;
        }
        
        if (arr[pos] < target) {
            left = pos + 1;
        } else {
            right = pos - 1;
        }
    }
    
    return -1;
}
```

**时间复杂度：** 平均O(log log n)，最坏O(n)  
**空间复杂度：** O(1)  
**适用场景：** 均匀分布的有序数据

#### 6.2.1 Dijkstra算法（单源最短路径）

```c
#define INF 99999

void dijkstra(int graph[][MAX_VERTICES], int numVertices, int start) {
    int dist[MAX_VERTICES];
    bool visited[MAX_VERTICES];
    int parent[MAX_VERTICES];
    
    // 初始化
    for (int i = 0; i < numVertices; i++) {
        dist[i] = INF;
        visited[i] = false;
        parent[i] = -1;
    }
    
    dist[start] = 0;
    
    for (int count = 0; count < numVertices - 1; count++) {
        // 找到最小距离的未访问顶点
        int u = -1;
        for (int v = 0; v < numVertices; v++) {
            if (!visited[v] && (u == -1 || dist[v] < dist[u])) {
                u = v;
            }
        }
        
        visited[u] = true;
        
        // 更新邻接顶点的距离
        for (int v = 0; v < numVertices; v++) {
            if (!visited[v] && graph[u][v] != 0 && 
                dist[u] != INF && dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
                parent[v] = u;
            }
        }
    }
    
    // 打印结果
    printf("顶点\t距离\t路径\n");
    for (int i = 0; i < numVertices; i++) {
        printf("%d\t%d\t", i, dist[i]);
        printDijkstraPath(parent, i);
        printf("\n");
    }
}

void printDijkstraPath(int parent[], int vertex) {
    if (parent[vertex] == -1) {
        printf("%d", vertex);
        return;
    }
    printDijkstraPath(parent, parent[vertex]);
    printf(" -> %d", vertex);
}
```

#### 6.2.2 Floyd-Warshall算法（所有顶点对最短路径）

```c
void floydWarshall(int graph[][MAX_VERTICES], int numVertices) {
    int dist[MAX_VERTICES][MAX_VERTICES];
    
    // 初始化距离矩阵
    for (int i = 0; i < numVertices; i++) {
        for (int j = 0; j < numVertices; j++) {
            if (i == j) {
                dist[i][j] = 0;
            } else if (graph[i][j] != 0) {
                dist[i][j] = graph[i][j];
            } else {
                dist[i][j] = INF;
            }
        }
    }
    
    // 通过所有顶点作为中间点更新最短路径
    for (int k = 0; k < numVertices; k++) {
        for (int i = 0; i < numVertices; i++) {
            for (int j = 0; j < numVertices; j++) {
                if (dist[i][k] != INF && dist[k][j] != INF && 
                    dist[i][k] + dist[k][j] < dist[i][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                }
            }
        }
    }
    
    // 打印最短距离矩阵
    printf("最短距离矩阵:\n");
    for (int i = 0; i < numVertices; i++) {
        for (int j = 0; j < numVertices; j++) {
            if (dist[i][j] == INF) {
                printf("INF\t");
            } else {
                printf("%d\t", dist[i][j]);
            }
        }
        printf("\n");
    }
}
```

### 6.3 最小生成树算法

#### 6.3.1 Kruskal算法

```c
// 边结构
typedef struct {
    int src, dest, weight;
} Edge;

// 并查集结构
typedef struct {
    int parent, rank;
} Subset;

// 查找操作（路径压缩）
int find(Subset subsets[], int i) {
    if (subsets[i].parent != i) {
        subsets[i].parent = find(subsets, subsets[i].parent);
    }
    return subsets[i].parent;
}

// 合并操作（按秩合并）
void unionSets(Subset subsets[], int x, int y) {
    int xroot = find(subsets, x);
    int yroot = find(subsets, y);
    
    if (subsets[xroot].rank < subsets[yroot].rank) {
        subsets[xroot].parent = yroot;
    } else if (subsets[xroot].rank > subsets[yroot].rank) {
        subsets[yroot].parent = xroot;
    } else {
        subsets[yroot].parent = xroot;
        subsets[xroot].rank++;
    }
}

// Kruskal算法实现
void kruskalMST(Edge edges[], int numEdges, int numVertices) {
    Edge result[numVertices - 1];
    int resultIndex = 0;
    
    // 按权重排序所有边
    qsort(edges, numEdges, sizeof(Edge), compareEdges);
    
    // 创建并查集
    Subset *subsets = (Subset*)malloc(numVertices * sizeof(Subset));
    for (int v = 0; v < numVertices; v++) {
        subsets[v].parent = v;
        subsets[v].rank = 0;
    }
    
    int edgeIndex = 0;
    
    while (resultIndex < numVertices - 1 && edgeIndex < numEdges) {
        Edge nextEdge = edges[edgeIndex++];
        
        int x = find(subsets, nextEdge.src);
        int y = find(subsets, nextEdge.dest);
        
        if (x != y) {
            result[resultIndex++] = nextEdge;
            unionSets(subsets, x, y);
        }
    }
    
    printf("最小生成树的边:\n");
    for (int i = 0; i < resultIndex; i++) {
        printf("%d -- %d: %d\n", result[i].src, result[i].dest, result[i].weight);
    }
    
    free(subsets);
}
```

```c
void kahnTopologicalSort(Graph *graph) {
```

## 7. 动态规划

动态规划通过分解子问题来解决复杂问题，避免重复计算，是解决优化问题的重要方法。

### 7.1 基本概念

动态规划的核心思想是将复杂问题分解为子问题，通过存储子问题的解来避免重复计算。

**基本要素：**
1. **最优子结构**：问题的最优解包含子问题的最优解
2. **重叠子问题**：递归算法会反复求解相同的子问题
3. **状态转移方程**：描述问题状态之间的关系
4. **边界条件**：递归的终止条件

```c
// 动态规划基本框架
int dp[MAX_SIZE];

int dynamicProgramming(int n) {
    // 初始化边界条件
    dp[0] = base_case;
    
    // 填充DP表
    for (int i = 1; i <= n; i++) {
        // 状态转移方程
        dp[i] = transition_function(dp[i-1], dp[i-2], ...);
    }
    
    return dp[n];
}
```

```c
// 0-1背包问题
int knapsack01(int weights[], int values[], int n, int capacity) {
    int dp[n + 1][capacity + 1];
    
    // 初始化
    for (int i = 0; i <= n; i++) {
        for (int w = 0; w <= capacity; w++) {
            if (i == 0 || w == 0) {
                dp[i][w] = 0;
            } else if (weights[i - 1] <= w) {
                // 选择物品i或不选择的最大值
                int include = values[i - 1] + dp[i - 1][w - weights[i - 1]];
                int exclude = dp[i - 1][w];
                dp[i][w] = (include > exclude) ? include : exclude;
            } else {
                dp[i][w] = dp[i - 1][w];
            }
        }
    }
    
    return dp[n][capacity];
}

// 空间优化的0-1背包
int knapsack01Optimized(int weights[], int values[], int n, int capacity) {
    int dp[capacity + 1];
    
    // 初始化
    for (int i = 0; i <= capacity; i++) {
        dp[i] = 0;
    }
    
    // 处理每个物品
    for (int i = 0; i < n; i++) {
        // 从后往前更新，避免重复使用
        for (int w = capacity; w >= weights[i]; w--) {

    }

}
```

#### 7.3.3 状态压缩

```c
// 旅行商问题（TSP）的状态压缩DP
#define MAX_N 16
#define INF 1e9

int tsp(int dist[][MAX_N], int n) {
    int dp[1 << n][n];
    
    // 初始化
    for (int mask = 0; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            dp[mask][i] = INF;
        }
    }
    
    dp[1][0] = 0; // 从城市0开始
    
    for (int mask = 0; mask < (1 << n); mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u)) || dp[mask][u] == INF) {
                continue;
            }
            
            for (int v = 0; v < n; v++) {
                if (mask & (1 << v)) {
                    continue;
                }
                
                int newMask = mask | (1 << v);
                if (dp[newMask][v] > dp[mask][u] + dist[u][v]) {
                    dp[newMask][v] = dp[mask][u] + dist[u][v];
                }
            }
        }
    }
    
    int result = INF;
    for (int i = 1; i < n; i++) {
        if (dp[(1 << n) - 1][i] + dist[i][0] < result) {
            result = dp[(1 << n) - 1][i] + dist[i][0];
        }
    }
    
    return result;
}
```

**动态规划复杂度分析：**
- **时间复杂度：** 通常为O(状态数 × 转移复杂度)
- **空间复杂度：** O(状态数)，可通过滚动数组等技巧优化
- **应用领域：** 最优化问题、计数问题、决策问题

## 8. 贪心算法

贪心算法在每一步都做出当前看起来最优的选择，希望通过局部最优选择达到全局最优解。

### 8.1 基本思想

贪心算法的核心思想是在每个决策点选择当前最优的方案，不考虑全局情况。

**贪心选择性质：**
- 每一步的选择都是当前状态下的最优选择
- 不会回溯修改已做的选择
- 局部最优选择能导致全局最优解

```c
// 贪心算法基本框架
int greedyAlgorithm(Problem problem) {
    Solution solution = initializeSolution();
    
    while (!isComplete(solution, problem)) {
        // 贪心选择：选择当前最优的元素
        Element best = selectBest(problem, solution);
        
        // 将选择加入解中
        addToSolution(solution, best);
        
        // 更新问题状态
        updateProblem(problem, best);
    }
    
    return solution;
}
```

```c
// 哈夫曼树节点
typedef struct HuffmanNode {
    char character;
    int frequency;
    struct HuffmanNode *left, *right;
} HuffmanNode;

// 优先队列（最小堆）
typedef struct {
    HuffmanNode **nodes;
    int size;
    int capacity;
} MinHeap;

// 创建新节点
HuffmanNode* createNode(char character, int frequency) {
    HuffmanNode *node = (HuffmanNode*)malloc(sizeof(HuffmanNode));
    node->character = character;
    node->frequency = frequency;
    node->left = node->right = NULL;
    return node;
}

// 创建最小堆
MinHeap* createMinHeap(int capacity) {
    MinHeap *heap = (MinHeap*)malloc(sizeof(MinHeap));
    heap->nodes = (HuffmanNode**)malloc(capacity * sizeof(HuffmanNode*));
    heap->size = 0;
    heap->capacity = capacity;
    return heap;
}

// 交换两个节点
void swapNodes(HuffmanNode **a, HuffmanNode **b) {
    HuffmanNode *temp = *a;
    *a = *b;
    *b = temp;
}

// 最小堆化
void minHeapify(MinHeap *heap, int idx) {
    int smallest = idx;
    int left = 2 * idx + 1;
    int right = 2 * idx + 2;
    
    if (left < heap->size && 
        heap->nodes[left]->frequency < heap->nodes[smallest]->frequency) {
        smallest = left;
    }
    
    if (right < heap->size && 
        heap->nodes[right]->frequency < heap->nodes[smallest]->frequency) {
        smallest = right;
    }
    
    if (smallest != idx) {
        swapNodes(&heap->nodes[smallest], &heap->nodes[idx]);
        minHeapify(heap, smallest);
    }
}

// 提取最小元素
HuffmanNode* extractMin(MinHeap *heap) {
    HuffmanNode *temp = heap->nodes[0];
    heap->nodes[0] = heap->nodes[heap->size - 1];
    heap->size--;
    minHeapify(heap, 0);
    return temp;
}

// 插入节点
void insertMinHeap(MinHeap *heap, HuffmanNode *node) {
    heap->size++;
    int i = heap->size - 1;
    
    while (i && node->frequency < heap->nodes[(i - 1) / 2]->frequency) {
        heap->nodes[i] = heap->nodes[(i - 1) / 2];
        i = (i - 1) / 2;
    }
    
    heap->nodes[i] = node;
}

// 构建哈夫曼树
HuffmanNode* buildHuffmanTree(char characters[], int frequencies[], int n) {
    MinHeap *heap = createMinHeap(n);
    
    // 创建叶子节点并插入堆
    for (int i = 0; i < n; i++) {
        HuffmanNode *node = createNode(characters[i], frequencies[i]);
        insertMinHeap(heap, node);
    }
    
    // 构建哈夫曼树
    while (heap->size > 1) {
        HuffmanNode *left = extractMin(heap);
        HuffmanNode *right = extractMin(heap);
        
        HuffmanNode *merged = createNode('\0', left->frequency + right->frequency);
        merged->left = left;
        merged->right = right;
        
        insertMinHeap(heap, merged);
    }
    
    return extractMin(heap);
}

// 打印哈夫曼编码
void printHuffmanCodes(HuffmanNode *root, char code[], int top) {
    if (root->left) {
        code[top] = '0';
        printHuffmanCodes(root->left, code, top + 1);
    }
    
    if (root->right) {
        code[top] = '1';
        printHuffmanCodes(root->right, code, top + 1);
    }
    
    if (!root->left && !root->right) {
        printf("%c: ", root->character);
        for (int i = 0; i < top; i++) {
            printf("%c", code[i]);
        }
        printf("\n");
    }
}

    HuffmanNode *root = buildHuffmanTree(characters, frequencies, n);
    
    char code[100];
    printf("哈夫曼编码:\n");
    printHuffmanCodes(root, code, 0);
}
```

```c
                u = v;
```

- **时间复杂度：** 通常为O(n log n)（主要是排序的复杂度）
- **空间复杂度：** O(1)到O(n)
- **优点：** 简单高效，易于实现
- **缺点：** 不能保证所有问题都能得到最优解
- **应用领域：** 调度问题、图论问题、编码问题

## 9. 分治算法

分治算法采用"分而治之"的策略，将复杂问题分解为若干个规模较小的相同子问题，递归求解这些子问题，然后合并子问题的解得到原问题的解。

### 9.1 基本思想

分治算法的核心思想包含三个步骤：

1. **分解（Divide）**：将原问题分解为若干个规模较小的子问题
2. **解决（Conquer）**：递归地求解各个子问题
3. **合并（Combine）**：将子问题的解合并为原问题的解

```c
// 分治算法基本框架
ResultType divideAndConquer(ProblemType problem) {
    // 基本情况：问题规模足够小，直接求解
    if (isBaseCase(problem)) {
        return solveDirect(problem);
    }
    
    // 分解：将问题分解为子问题
    SubProblem subProblems[] = divide(problem);
    
    // 解决：递归求解子问题
    ResultType subResults[numSubProblems];
    for (int i = 0; i < numSubProblems; i++) {
        subResults[i] = divideAndConquer(subProblems[i]);
    }
    
    // 合并：合并子问题的解
    return combine(subResults, numSubProblems);
}
```

**设计要点：**
- **确定分解方式**：如何将问题分解为子问题
- **确定基本情况**：递归的终止条件
- **设计合并策略**：如何合并子问题的解
- **分析时间复杂度**：使用主定理分析复杂度

### 9.2 经典应用

#### 9.2.1 归并排序

归并排序是分治算法的经典应用，时间复杂度为O(n log n)。

```c
// 合并两个有序数组
void merge(int arr[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    
    // 创建临时数组
    int *leftArr = (int*)malloc(n1 * sizeof(int));
    int *rightArr = (int*)malloc(n2 * sizeof(int));
    
    // 复制数据到临时数组
    for (int i = 0; i < n1; i++) {
        leftArr[i] = arr[left + i];
    }
    for (int j = 0; j < n2; j++) {
        rightArr[j] = arr[mid + 1 + j];
    }
    
    // 合并临时数组回到原数组
    int i = 0, j = 0, k = left;
    
    while (i < n1 && j < n2) {
        if (leftArr[i] <= rightArr[j]) {
            arr[k] = leftArr[i];
            i++;
        } else {
            arr[k] = rightArr[j];
            j++;
        }
        k++;
    }
    
    // 复制剩余元素
    while (i < n1) {
        arr[k] = leftArr[i];
        i++;
        k++;
    }
    
    while (j < n2) {
        arr[k] = rightArr[j];
        j++;
        k++;
    }
    
    free(leftArr);
    free(rightArr);
}

// 归并排序主函数
void mergeSort(int arr[], int left, int right) {
    if (left < right) {
        // 分解：找到中点
        int mid = left + (right - left) / 2;
        
        // 递归排序左半部分
        mergeSort(arr, left, mid);
        
        // 递归排序右半部分
        mergeSort(arr, mid + 1, right);
        
        // 合并两个有序部分
        merge(arr, left, mid, right);
    }
}

// 带详细输出的归并排序
void mergeSortVerbose(int arr[], int left, int right, int depth) {
    // 打印缩进
    for (int i = 0; i < depth; i++) {
        printf("  ");
    }
    
    printf("排序区间 [%d, %d]: ", left, right);
    for (int i = left; i <= right; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    
    if (left < right) {
        int mid = left + (right - left) / 2;
        
        // 分解阶段
        for (int i = 0; i < depth; i++) {
            printf("  ");
        }
        printf("分解为: [%d, %d] 和 [%d, %d]\n", left, mid, mid + 1, right);
        
        mergeSortVerbose(arr, left, mid, depth + 1);
        mergeSortVerbose(arr, mid + 1, right, depth + 1);
        
        // 合并阶段

        for (int i = 0; i < depth; i++) {
            printf("  ");
        }
        printf("合并结果: ");

        }
        printf("\n");
    }
```

#### 9.2.2 快速排序

快速排序也是分治算法的经典应用，平均时间复杂度为O(n log n)。

```c
// 分区函数（Lomuto分区方案）
int partition(int arr[], int low, int high) {
    int pivot = arr[high]; // 选择最后一个元素作为基准
    int i = low - 1; // 小于基准的元素的索引
    
    for (int j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            // 交换元素
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
    
    // 将基准放到正确位置
    int temp = arr[i + 1];
    arr[i + 1] = arr[high];
    arr[high] = temp;
    
    return i + 1;
}

// 快速排序主函数
void quickSort(int arr[], int low, int high) {
    if (low < high) {
        // 分区：找到基准的正确位置
        int pi = partition(arr, low, high);
        
        // 递归排序基准左边的元素
        quickSort(arr, low, pi - 1);
        
        // 递归排序基准右边的元素
        quickSort(arr, pi + 1, high);
    }
}

// 随机化快速排序
int randomizedPartition(int arr[], int low, int high) {
    // 随机选择基准
    int randomIndex = low + rand() % (high - low + 1);
    
    // 将随机选择的元素与最后一个元素交换
    int temp = arr[randomIndex];
    arr[randomIndex] = arr[high];
    arr[high] = temp;
    
    return partition(arr, low, high);
}

void randomizedQuickSort(int arr[], int low, int high) {
    if (low < high) {
        int pi = randomizedPartition(arr, low, high);
        randomizedQuickSort(arr, low, pi - 1);
        randomizedQuickSort(arr, pi + 1, high);
    }
}
```

#### 9.2.3 最大子数组问题

使用分治法解决最大子数组和问题。

```c
// 跨越中点的最大子数组
typedef struct {
    int sum;
    int left;
    int right;
} MaxSubarray;

MaxSubarray findMaxCrossingSubarray(int arr[], int low, int mid, int high) {
    int leftSum = INT_MIN;
    int sum = 0;
    int maxLeft = mid;
    
    // 从中点向左找最大和
    for (int i = mid; i >= low; i--) {
        sum += arr[i];
        if (sum > leftSum) {
            leftSum = sum;
            maxLeft = i;
        }
    }
    
    int rightSum = INT_MIN;
    sum = 0;
    int maxRight = mid + 1;
    
    // 从中点向右找最大和
    for (int i = mid + 1; i <= high; i++) {
        sum += arr[i];
        if (sum > rightSum) {
            rightSum = sum;
            maxRight = i;
        }
    }
    
    MaxSubarray result;
    result.sum = leftSum + rightSum;
    result.left = maxLeft;
    result.right = maxRight;
    
    return result;
}

// 分治法求最大子数组
MaxSubarray findMaxSubarray(int arr[], int low, int high) {
    MaxSubarray result;
    
    // 基本情况：只有一个元素
    if (low == high) {
        result.sum = arr[low];
        result.left = low;
        result.right = high;
        return result;
    }
    
    int mid = (low + high) / 2;
    
    // 递归求解左半部分
    MaxSubarray leftResult = findMaxSubarray(arr, low, mid);
    
    // 递归求解右半部分
    MaxSubarray rightResult = findMaxSubarray(arr, mid + 1, high);
    
    // 求解跨越中点的最大子数组
    MaxSubarray crossResult = findMaxCrossingSubarray(arr, low, mid, high);
    
    // 返回三者中的最大值
    if (leftResult.sum >= rightResult.sum && leftResult.sum >= crossResult.sum) {
        return leftResult;
    } else if (rightResult.sum >= leftResult.sum && rightResult.sum >= crossResult.sum) {
        return rightResult;
    } else {
        return crossResult;
    }
}

// 打印最大子数组信息
void printMaxSubarray(int arr[], MaxSubarray result) {
    printf("最大子数组: [");
    for (int i = result.left; i <= result.right; i++) {
        printf("%d", arr[i]);
        if (i < result.right) printf(", ");
    }
    printf("]\n");
    printf("位置: [%d, %d]\n", result.left, result.right);
    printf("最大和: %d\n", result.sum);
}
```

#### 9.2.4 大整数乘法（Karatsuba算法）

```c
// 计算10的幂
long long power10(int exp) {
    long long result = 1;
    for (int i = 0; i < exp; i++) {
        result *= 10;
    }
    return result;
}

// 获取数字的位数
int getDigits(long long num) {
    if (num == 0) return 1;
    int count = 0;
    while (num > 0) {
        count++;
        num /= 10;
    }
    return count;
}

// Karatsuba大整数乘法
long long karatsuba(long long x, long long y) {
    // 基本情况：小数字直接相乘
    if (x < 10 || y < 10) {
        return x * y;
    }
    
    // 计算位数
    int n = fmax(getDigits(x), getDigits(y));
    int half = n / 2;
    
    long long powerOf10 = power10(half);
    
    // 分解数字
    long long a = x / powerOf10;  // 高位
    long long b = x % powerOf10;  // 低位
    long long c = y / powerOf10;  // 高位
    long long d = y % powerOf10;  // 低位
    
    // 递归计算三个乘积
    long long ac = karatsuba(a, c);
    long long bd = karatsuba(b, d);
    long long ad_bc = karatsuba(a + b, c + d) - ac - bd;
    
    // 合并结果
    return ac * power10(2 * half) + ad_bc * power10(half) + bd;
}
```

### 9.3 复杂度分析

分治算法的复杂度分析通常使用**主定理（Master Theorem）**：

**主定理形式：** T(n) = aT(n/b) + f(n)

其中：
- a ≥ 1：子问题的数量
- b > 1：每个子问题的规模缩减因子
- f(n)：分解和合并的时间复杂度

**主定理的三种情况：**

1. **情况1：** 如果 f(n) = O(n^(log_b(a) - ε))，其中 ε > 0，则 T(n) = Θ(n^log_b(a))
2. **情况2：** 如果 f(n) = Θ(n^log_b(a))，则 T(n) = Θ(n^log_b(a) * log n)
3. **情况3：** 如果 f(n) = Ω(n^(log_b(a) + ε))，其中 ε > 0，且对足够大的n有af(n/b) ≤ cf(n)（其中c < 1），则 T(n) = Θ(f(n))

```c

    printf("   a=1, b=2, f(n)=O(1)\n");
    printf("   log_2(1) = 0, f(n) = Θ(n^0)\n");

    printf("3. Karatsuba乘法:\n");

    printf("   a=3, b=2, f(n)=O(n)\n");
    printf("   log_2(3) ≈ 1.585, f(n) = O(n^1)\n");
```

**分治算法的优缺点：**

**优点：**
- 能够有效处理大规模问题
- 算法结构清晰，易于理解和实现
- 通常能达到较好的时间复杂度
- 天然支持并行化

**缺点：**
- 递归调用有额外的空间开销
- 可能存在重复计算的子问题
- 对于小规模问题，分治的开销可能超过收益

**应用领域：**
- 排序算法（归并排序、快速排序）
- 查找算法（二分查找）
- 数值计算（大整数乘法、矩阵乘法）
- 图像处理（图像分割、特征提取）
- 计算几何（最近点对、凸包）

```c
// 1. 原地算法（In-place Algorithm）
void bubbleSortInPlace(int arr[], int n) {
    // 空间复杂度：O(1)
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // 原地交换，不需要额外空间
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

// 2. 滚动数组优化
int fibonacciOptimized(int n) {
    // 空间复杂度：O(1)，而不是O(n)
    if (n <= 1) return n;
    
    int prev2 = 0, prev1 = 1, current;
    
    for (int i = 2; i <= n; i++) {
        current = prev1 + prev2;
        prev2 = prev1;
        prev1 = current;
    }
    
    return current;
}

// 3. 位运算优化
void setBitOptimized(int *bitset, int pos) {
    // 使用位运算减少空间使用
    int index = pos / 32;
    int bit = pos % 32;
    bitset[index] |= (1 << bit);
}

bool getBitOptimized(int *bitset, int pos) {
    int index = pos / 32;
    int bit = pos % 32;
    return (bitset[index] & (1 << bit)) != 0;
}

// 4. 递归转迭代优化
int factorialIterative(int n) {
    // 空间复杂度：O(1)，而不是递归的O(n)
    int result = 1;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}
```

```c
// 快速排序的三种情况分析
int partitionAnalysis(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = low - 1;
    
    for (int j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            // 交换
            int temp = arr[i];
            arr[i] = arr[j];
            arr[j] = temp;
        }
    }
    
    // 将基准放到正确位置
    int temp = arr[i + 1];
    arr[i + 1] = arr[high];
    arr[high] = temp;
    
    return i + 1;
}

void quickSortAnalysis(int arr[], int low, int high) {
    if (low < high) {
        int pi = partitionAnalysis(arr, low, high);
        quickSortAnalysis(arr, low, pi - 1);
        quickSortAnalysis(arr, pi + 1, high);
    }
}
```

```c
// 进阶练习示例：动态规划
int longestCommonSubsequence(char* text1, char* text2) {
    int m = strlen(text1), n = strlen(text2);
    int dp[m + 1][n + 1];
    
    // 初始化
    for (int i = 0; i <= m; i++) {
        for (int j = 0; j <= n; j++) {
            if (i == 0 || j == 0) {
                dp[i][j] = 0;
            } else if (text1[i - 1] == text2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            } else {
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
            }
        }
    }
    
    return dp[m][n];
}

void dijkstraAlgorithm(int graph[][MAX_VERTICES], int src, int V) {
    int dist[MAX_VERTICES];
    bool visited[MAX_VERTICES];
    
    // 初始化
    for (int i = 0; i < V; i++) {
        dist[i] = INT_MAX;
        visited[i] = false;
    }
    dist[src] = 0;
    
    for (int count = 0; count < V - 1; count++) {
        int u = minDistance(dist, visited, V);
        visited[u] = true;
        
        for (int v = 0; v < V; v++) {
            if (!visited[v] && graph[u][v] && 
                dist[u] != INT_MAX && 
                dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
            }
        }
    }

}
```
