## 1. 数据库基本概念

### 1.1 数据库系统组成

数据库系统由以下几个部分组成：

- **数据库（Database）**：存储数据的仓库
- **数据库管理系统（DBMS）**：管理数据库的软件
- **数据库应用程序**：使用数据库的应用
- **数据库管理员（DBA）**：管理数据库的人员

### 1.2 数据模型

#### 1.2.1 层次模型

```
学校
├── 计算机学院
│   ├── 计算机科学系
│   └── 软件工程系
└── 数学学院
    ├── 数学系
    └── 统计系
```

#### 1.2.2 网状模型

```
学生 ←→ 课程
  ↓      ↓
成绩 ←→ 教师
```

#### 1.2.3 关系模型

关系模型是目前最主要的数据模型，基于数学中的关系理论。

```sql
-- 学生表
CREATE TABLE students (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    age INT,
    major VARCHAR(50)
);

-- 课程表
CREATE TABLE courses (
    course_id INT PRIMARY KEY,
    course_name VARCHAR(100),
    credits INT
);

-- 选课表
CREATE TABLE enrollments (
    student_id INT,
    course_id INT,
    grade DECIMAL(3,1),
    PRIMARY KEY (student_id, course_id),
    FOREIGN KEY (student_id) REFERENCES students(student_id),
    FOREIGN KEY (course_id) REFERENCES courses(course_id)
);
```

### 1.3 数据库系统架构

#### 1.3.1 三级模式结构

```
外模式（用户视图）
     ↓
概念模式（逻辑结构）
     ↓
内模式（物理结构）
```

#### 1.3.2 数据独立性

- **物理独立性**：概念模式不依赖于内模式
- **逻辑独立性**：外模式不依赖于概念模式

## 2. 关系数据库理论

### 2.1 关系代数

#### 2.1.1 基本操作

```sql
-- 选择（Selection）σ
SELECT * FROM students WHERE age > 20;

-- 投影（Projection）π
SELECT name, major FROM students;

-- 连接（Join）⋈
SELECT s.name, c.course_name, e.grade
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c ON e.course_id = c.course_id;

-- 并（Union）∪
SELECT name FROM students WHERE major = 'Computer Science'
UNION
SELECT name FROM students WHERE major = 'Mathematics';

-- 差（Difference）-
SELECT name FROM students WHERE major = 'Computer Science'
EXCEPT
SELECT name FROM students WHERE age < 18;

-- 笛卡尔积（Cartesian Product）×
SELECT * FROM students CROSS JOIN courses;
```

#### 2.1.2 扩展操作

```sql
-- 外连接（Outer Join）
-- 左外连接
SELECT s.name, e.grade
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id;

-- 右外连接
SELECT s.name, e.grade
FROM students s
RIGHT JOIN enrollments e ON s.student_id = e.student_id;

-- 全外连接
SELECT s.name, e.grade
FROM students s
FULL OUTER JOIN enrollments e ON s.student_id = e.student_id;

-- 半连接（Semi Join）
SELECT * FROM students s
WHERE EXISTS (
    SELECT 1 FROM enrollments e 
    WHERE e.student_id = s.student_id
);

-- 反连接（Anti Join）
SELECT * FROM students s
WHERE NOT EXISTS (
    SELECT 1 FROM enrollments e 
    WHERE e.student_id = s.student_id
);
```

### 2.2 函数依赖

#### 2.2.1 函数依赖定义

如果在关系R中，属性集X的值能够唯一确定属性集Y的值，则称Y函数依赖于X，记作X→Y。

```sql
-- 示例：学生表中
-- student_id → name, age, major
-- (student_id, course_id) → grade

CREATE TABLE student_courses (
    student_id INT,
    student_name VARCHAR(50),
    course_id INT,
    course_name VARCHAR(100),
    grade DECIMAL(3,1),
    -- student_id → student_name
    -- course_id → course_name
    -- (student_id, course_id) → grade
);
```

#### 2.2.2 函数依赖类型

```sql
-- 完全函数依赖
-- (student_id, course_id) → grade
-- grade完全依赖于(student_id, course_id)

-- 部分函数依赖
-- (student_id, course_id) → student_name
-- student_name只依赖于student_id，不依赖于course_id

-- 传递函数依赖
-- student_id → department_id → department_name
-- student_id → department_name（传递依赖）

CREATE TABLE students_with_dept (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    department_id INT,
    department_name VARCHAR(50)
    -- 存在传递依赖：student_id → department_id → department_name
);
```

### 2.3 范式理论

#### 2.3.1 第一范式（1NF）

要求每个属性都是原子的，不可再分。

```sql
-- 违反1NF的设计
CREATE TABLE students_bad (
    student_id INT,
    name VARCHAR(50),
    courses VARCHAR(200)  -- 存储多个课程，违反1NF
);

-- 符合1NF的设计
CREATE TABLE students (
    student_id INT,
    name VARCHAR(50)
);

CREATE TABLE student_courses (
    student_id INT,
    course_id INT,
    PRIMARY KEY (student_id, course_id)
);
```

#### 2.3.2 第二范式（2NF）

在1NF基础上，消除部分函数依赖。

```sql
-- 违反2NF的设计
CREATE TABLE enrollments_bad (
    student_id INT,
    course_id INT,
    student_name VARCHAR(50),  -- 部分依赖于student_id
    course_name VARCHAR(100),  -- 部分依赖于course_id
    grade DECIMAL(3,1),
    PRIMARY KEY (student_id, course_id)
);

-- 符合2NF的设计
CREATE TABLE students (
    student_id INT PRIMARY KEY,
    student_name VARCHAR(50)
);

CREATE TABLE courses (
    course_id INT PRIMARY KEY,
    course_name VARCHAR(100)
);

CREATE TABLE enrollments (
    student_id INT,
    course_id INT,
    grade DECIMAL(3,1),
    PRIMARY KEY (student_id, course_id),
    FOREIGN KEY (student_id) REFERENCES students(student_id),
    FOREIGN KEY (course_id) REFERENCES courses(course_id)
);
```

#### 2.3.3 第三范式（3NF）

在2NF基础上，消除传递函数依赖。

```sql
-- 违反3NF的设计
CREATE TABLE students_bad (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    department_id INT,
    department_name VARCHAR(50),  -- 传递依赖
    department_location VARCHAR(100)  -- 传递依赖
);

-- 符合3NF的设计
CREATE TABLE departments (
    department_id INT PRIMARY KEY,
    department_name VARCHAR(50),
    location VARCHAR(100)
);

CREATE TABLE students (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    department_id INT,
    FOREIGN KEY (department_id) REFERENCES departments(department_id)
);
```

#### 2.3.4 BCNF（Boyce-Codd范式）

在3NF基础上，要求每个决定因素都是候选键。

```sql
-- 违反BCNF的设计
CREATE TABLE course_instructor (
    course_id INT,
    instructor_id INT,
    instructor_name VARCHAR(50),
    room VARCHAR(20),
    PRIMARY KEY (course_id, instructor_id)
    -- 假设：instructor_id → instructor_name
    -- 但instructor_id不是候选键
);

-- 符合BCNF的设计
CREATE TABLE instructors (
    instructor_id INT PRIMARY KEY,
    instructor_name VARCHAR(50)
);

CREATE TABLE course_assignments (
    course_id INT,
    instructor_id INT,
    room VARCHAR(20),
    PRIMARY KEY (course_id, instructor_id),
    FOREIGN KEY (instructor_id) REFERENCES instructors(instructor_id)
);
```

## 3. SQL语言基础

### 3.1 数据定义语言（DDL）

#### 3.1.1 创建数据库和表

```sql
-- 创建数据库
CREATE DATABASE university;
USE university;

-- 创建表
CREATE TABLE students (
    student_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE,
    age INT CHECK (age >= 16 AND age <= 100),
    major VARCHAR(50),
    enrollment_date DATE DEFAULT CURRENT_DATE,
    gpa DECIMAL(3,2) DEFAULT 0.00,
    INDEX idx_major (major),
    INDEX idx_gpa (gpa)
);

-- 创建带约束的表
CREATE TABLE courses (
    course_id INT AUTO_INCREMENT PRIMARY KEY,
    course_code VARCHAR(10) NOT NULL UNIQUE,
    course_name VARCHAR(100) NOT NULL,
    credits INT NOT NULL CHECK (credits > 0),
    department VARCHAR(50),
    prerequisite_id INT,
    FOREIGN KEY (prerequisite_id) REFERENCES courses(course_id)
);

-- 创建选课表
CREATE TABLE enrollments (
    enrollment_id INT AUTO_INCREMENT PRIMARY KEY,
    student_id INT NOT NULL,
    course_id INT NOT NULL,
    semester VARCHAR(20) NOT NULL,
    grade DECIMAL(3,1) CHECK (grade >= 0 AND grade <= 100),
    enrollment_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY unique_enrollment (student_id, course_id, semester),
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE
);
```

#### 3.1.2 修改表结构

```sql
-- 添加列
ALTER TABLE students 
ADD COLUMN phone VARCHAR(15),
ADD COLUMN address TEXT;

-- 修改列
ALTER TABLE students 
MODIFY COLUMN name VARCHAR(100) NOT NULL;

-- 删除列
ALTER TABLE students 
DROP COLUMN address;

-- 添加索引
ALTER TABLE students 
ADD INDEX idx_name (name),
ADD INDEX idx_email_major (email, major);

-- 添加约束
ALTER TABLE students 
ADD CONSTRAINT chk_gpa CHECK (gpa >= 0.0 AND gpa <= 4.0);

-- 删除约束
ALTER TABLE students 
DROP CONSTRAINT chk_gpa;
```

### 3.2 数据操作语言（DML）

#### 3.2.1 插入数据

```sql
-- 基本插入
INSERT INTO students (name, email, age, major, gpa) 
VALUES ('张三', 'zhangsan@email.com', 20, 'Computer Science', 3.5);

-- 批量插入
INSERT INTO students (name, email, age, major, gpa) VALUES
('李四', 'lisi@email.com', 21, 'Mathematics', 3.7),
('王五', 'wangwu@email.com', 19, 'Physics', 3.2),
('赵六', 'zhaoliu@email.com', 22, 'Computer Science', 3.9);

-- 从查询结果插入
INSERT INTO honor_students (student_id, name, gpa)
SELECT student_id, name, gpa 
FROM students 
WHERE gpa >= 3.5;

-- 插入或更新（MySQL）
INSERT INTO students (student_id, name, email, gpa) 
VALUES (1, '张三', 'zhangsan_new@email.com', 3.6)
ON DUPLICATE KEY UPDATE 
    email = VALUES(email),
    gpa = VALUES(gpa);
```

#### 3.2.2 更新数据

```sql
-- 基本更新
UPDATE students 
SET gpa = 3.8 
WHERE student_id = 1;

-- 多列更新
UPDATE students 
SET gpa = gpa + 0.1, 
    major = 'Computer Engineering' 
WHERE major = 'Computer Science' AND gpa < 3.0;

-- 基于连接的更新
UPDATE students s
JOIN (
    SELECT student_id, AVG(grade/25) as avg_gpa
    FROM enrollments 
    WHERE grade IS NOT NULL
    GROUP BY student_id
) e ON s.student_id = e.student_id
SET s.gpa = e.avg_gpa;

-- 条件更新
UPDATE students 
SET gpa = CASE 
    WHEN gpa >= 3.5 THEN gpa + 0.1
    WHEN gpa >= 3.0 THEN gpa + 0.05
    ELSE gpa
END
WHERE major = 'Computer Science';
```

#### 3.2.3 删除数据

```sql
-- 基本删除
DELETE FROM students 
WHERE gpa < 2.0;

-- 基于连接的删除
DELETE s FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
WHERE e.student_id IS NULL;

-- 删除重复记录（保留最新的）
DELETE s1 FROM students s1
INNER JOIN students s2 
WHERE s1.email = s2.email 
  AND s1.student_id < s2.student_id;

-- 清空表
TRUNCATE TABLE temp_students;
```

### 3.3 数据查询语言（DQL）

#### 3.3.1 基本查询

```sql
-- 简单查询
SELECT name, major, gpa 
FROM students 
WHERE gpa > 3.0
ORDER BY gpa DESC;

-- 使用别名
SELECT 
    s.name AS student_name,
    s.major AS field_of_study,
    s.gpa AS grade_point_average
FROM students s
WHERE s.age BETWEEN 18 AND 25;

-- 去重查询
SELECT DISTINCT major 
FROM students 
ORDER BY major;

-- 限制结果数量
SELECT name, gpa 
FROM students 
ORDER BY gpa DESC 
LIMIT 10;

-- 分页查询
SELECT name, gpa 
FROM students 
ORDER BY name 
LIMIT 20 OFFSET 40;  -- 第3页，每页20条
```

#### 3.3.2 条件查询

```sql
-- 多条件查询
SELECT * FROM students 
WHERE major = 'Computer Science' 
  AND gpa >= 3.5 
  AND age < 25;

-- 模糊查询
SELECT * FROM students 
WHERE name LIKE '张%'  -- 姓张的学生
   OR email LIKE '%@gmail.com';  -- Gmail邮箱

-- 范围查询
SELECT * FROM students 
WHERE gpa BETWEEN 3.0 AND 3.5
  AND age IN (19, 20, 21);

-- 空值查询
SELECT * FROM students 
WHERE phone IS NULL 
   OR phone = '';

-- 正则表达式查询（MySQL）
SELECT * FROM students 
WHERE email REGEXP '^[a-zA-Z0-9]+@[a-zA-Z0-9]+\.[a-zA-Z]{2,}$';
```

#### 3.3.3 聚合查询

```sql
-- 基本聚合函数
SELECT 
    COUNT(*) as total_students,
    COUNT(DISTINCT major) as total_majors,
    AVG(gpa) as average_gpa,
    MAX(gpa) as highest_gpa,
    MIN(gpa) as lowest_gpa,
    SUM(CASE WHEN gpa >= 3.5 THEN 1 ELSE 0 END) as honor_students
FROM students;

-- 分组聚合
SELECT 
    major,
    COUNT(*) as student_count,
    AVG(gpa) as avg_gpa,
    MAX(gpa) as max_gpa,
    MIN(gpa) as min_gpa
FROM students 
GROUP BY major
HAVING COUNT(*) >= 5  -- 至少5个学生的专业
ORDER BY avg_gpa DESC;

-- 多级分组
SELECT 
    major,
    CASE 
        WHEN age < 20 THEN 'Under 20'
        WHEN age < 25 THEN '20-24'
        ELSE '25+'
    END as age_group,
    COUNT(*) as count,
    AVG(gpa) as avg_gpa
FROM students 
GROUP BY major, age_group
ORDER BY major, age_group;

-- 窗口函数（MySQL 8.0+）
SELECT 
    name,
    major,
    gpa,
    ROW_NUMBER() OVER (PARTITION BY major ORDER BY gpa DESC) as rank_in_major,
    RANK() OVER (ORDER BY gpa DESC) as overall_rank,
    DENSE_RANK() OVER (ORDER BY gpa DESC) as dense_rank,
    LAG(gpa) OVER (PARTITION BY major ORDER BY gpa DESC) as prev_gpa,
    LEAD(gpa) OVER (PARTITION BY major ORDER BY gpa DESC) as next_gpa
FROM students;

-- 累计统计
SELECT 
    name,
    gpa,
    SUM(1) OVER (ORDER BY gpa DESC) as cumulative_count,
    AVG(gpa) OVER (ORDER BY gpa DESC ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) as running_avg
FROM students
ORDER BY gpa DESC;
```

#### 3.3.4 连接查询

```sql
-- 内连接
SELECT 
    s.name,
    s.major,
    c.course_name,
    e.grade
FROM students s
INNER JOIN enrollments e ON s.student_id = e.student_id
INNER JOIN courses c ON e.course_id = c.course_id
WHERE e.grade >= 85;

-- 左外连接（显示所有学生，包括未选课的）
SELECT 
    s.name,
    s.major,
    COUNT(e.course_id) as courses_taken,
    AVG(e.grade) as avg_grade
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.student_id, s.name, s.major
ORDER BY courses_taken DESC;

-- 右外连接（显示所有课程，包括无人选修的）
SELECT 
    c.course_name,
    c.credits,
    COUNT(e.student_id) as enrollment_count
FROM enrollments e
RIGHT JOIN courses c ON e.course_id = c.course_id
GROUP BY c.course_id, c.course_name, c.credits
ORDER BY enrollment_count DESC;

-- 全外连接
SELECT 
    COALESCE(s.name, 'No Student') as student_name,
    COALESCE(c.course_name, 'No Course') as course_name,
    e.grade
FROM students s
FULL OUTER JOIN enrollments e ON s.student_id = e.student_id
FULL OUTER JOIN courses c ON e.course_id = c.course_id;

-- 自连接（查找同专业的学生）
SELECT 
    s1.name as student1,
    s2.name as student2,
    s1.major
FROM students s1
JOIN students s2 ON s1.major = s2.major AND s1.student_id < s2.student_id
ORDER BY s1.major, s1.name;

-- 交叉连接（笛卡尔积）
SELECT 
    s.name,
    c.course_name
FROM students s
CROSS JOIN courses c
WHERE s.major = 'Computer Science' 
  AND c.department = 'Computer Science';
```

#### 3.3.5 子查询

```sql
-- 标量子查询
SELECT 
    name,
    gpa,
    (SELECT AVG(gpa) FROM students) as avg_gpa,
    gpa - (SELECT AVG(gpa) FROM students) as gpa_diff
FROM students
WHERE gpa > (SELECT AVG(gpa) FROM students);

-- 列子查询（IN/NOT IN）
SELECT name, major 
FROM students 
WHERE student_id IN (
    SELECT student_id 
    FROM enrollments 
    WHERE grade >= 90
);

-- 行子查询
SELECT name, major, gpa
FROM students
WHERE (major, gpa) IN (
    SELECT major, MAX(gpa)
    FROM students
    GROUP BY major
);

-- 表子查询
SELECT 
    top_students.major,
    top_students.max_gpa,
    COUNT(*) as student_count
FROM (
    SELECT major, MAX(gpa) as max_gpa
    FROM students
    GROUP BY major
    HAVING MAX(gpa) >= 3.5
) top_students
JOIN students s ON top_students.major = s.major
GROUP BY top_students.major, top_students.max_gpa;

-- 相关子查询（EXISTS）
SELECT name, major
FROM students s
WHERE EXISTS (
    SELECT 1
    FROM enrollments e
    WHERE e.student_id = s.student_id
      AND e.grade >= 95
);

-- 相关子查询（比较）
SELECT name, major, gpa
FROM students s1
WHERE gpa > (
    SELECT AVG(gpa)
    FROM students s2
    WHERE s2.major = s1.major
);
```

#### 3.3.6 集合操作

```sql
-- 并集（UNION）
SELECT name, 'High GPA' as category
FROM students
WHERE gpa >= 3.5
UNION
SELECT name, 'Honor Roll' as category
FROM students
WHERE student_id IN (
    SELECT student_id
    FROM enrollments
    WHERE grade >= 90
);

-- 交集（INTERSECT）- MySQL不直接支持，用JOIN实现
SELECT DISTINCT s.name
FROM students s
JOIN enrollments e1 ON s.student_id = e1.student_id
JOIN courses c1 ON e1.course_id = c1.course_id AND c1.course_name = 'Database Systems'
JOIN enrollments e2 ON s.student_id = e2.student_id
JOIN courses c2 ON e2.course_id = c2.course_id AND c2.course_name = 'Data Structures';

-- 差集（EXCEPT）- MySQL不直接支持，用NOT EXISTS实现
SELECT name
FROM students s
WHERE major = 'Computer Science'
  AND NOT EXISTS (
    SELECT 1
    FROM enrollments e
    JOIN courses c ON e.course_id = c.course_id
    WHERE e.student_id = s.student_id
      AND c.course_name = 'Database Systems'
  );
```

#### 3.4.2 权限管理

```sql
-- 授予权限
GRANT SELECT, INSERT, UPDATE ON university.students TO 'student_user'@'localhost';
GRANT ALL PRIVILEGES ON university.* TO 'teacher_user'@'%';
GRANT SELECT ON university.enrollments TO 'readonly_user'@'%';

-- 授予特定列的权限
GRANT SELECT (name, major), UPDATE (gpa) ON university.students TO 'advisor'@'localhost';

-- 撤销权限
REVOKE INSERT, UPDATE ON university.students FROM 'student_user'@'localhost';
REVOKE ALL PRIVILEGES ON university.* FROM 'old_teacher'@'%';

-- 查看权限
SHOW GRANTS FOR 'student_user'@'localhost';
SHOW GRANTS FOR CURRENT_USER();

-- 刷新权限
FLUSH PRIVILEGES;
```

#### 3.4.3 角色管理（MySQL 8.0+）

```sql
-- 创建角色
CREATE ROLE 'student_role', 'teacher_role', 'admin_role';

-- 给角色授权
GRANT SELECT ON university.* TO 'student_role';
GRANT SELECT, INSERT, UPDATE, DELETE ON university.* TO 'teacher_role';
GRANT ALL PRIVILEGES ON university.* TO 'admin_role';

-- 将角色分配给用户
GRANT 'student_role' TO 'john'@'localhost';
GRANT 'teacher_role' TO 'mary'@'localhost';
GRANT 'admin_role' TO 'admin'@'localhost';

-- 设置默认角色
SET DEFAULT ROLE 'student_role' TO 'john'@'localhost';

-- 激活角色
SET ROLE 'teacher_role';

-- 查看角色
SELECT * FROM mysql.role_edges;
SHOW GRANTS FOR 'student_role';
```

### 4.1 需求分析

#### 4.1.1 需求收集

```sql
-- 示例：学生管理系统需求分析
/*
功能需求：
1. 学生信息管理（增删改查）
```

```sql
DELIMITER //
CREATE PROCEDURE UpdateCourseStatistics(IN p_course_id VARCHAR(20))
BEGIN
    DECLARE v_total_enrollments INT DEFAULT 0;
    DECLARE v_current_enrollments INT DEFAULT 0;
    DECLARE v_average_grade DECIMAL(5,2) DEFAULT 0;
    DECLARE v_pass_rate DECIMAL(5,2) DEFAULT 0;
    DECLARE v_passed_students INT DEFAULT 0;
    DECLARE v_graded_students INT DEFAULT 0;
    
    -- 计算总选课人数
    SELECT COUNT(*) INTO v_total_enrollments
    FROM enrollments
    WHERE course_id = p_course_id;
    
    -- 计算当前选课人数
    SELECT COUNT(*) INTO v_current_enrollments
    FROM enrollments
    WHERE course_id = p_course_id AND status = 'Enrolled';
    
    -- 计算平均成绩
    SELECT AVG(grade) INTO v_average_grade
    FROM enrollments
    WHERE course_id = p_course_id AND grade IS NOT NULL;
    
    -- 计算通过率
    SELECT 
        COUNT(CASE WHEN grade >= 60 THEN 1 END),
        COUNT(CASE WHEN grade IS NOT NULL THEN 1 END)
    INTO v_passed_students, v_graded_students
    FROM enrollments
    WHERE course_id = p_course_id;
    
    IF v_graded_students > 0 THEN
        SET v_pass_rate = (v_passed_students / v_graded_students) * 100;
    END IF;
    
    -- 更新统计表
    INSERT INTO course_statistics (
        course_id, course_name, total_enrollments, 
        current_enrollments, average_grade, pass_rate
    )
    SELECT 
        p_course_id, c.course_name, v_total_enrollments,
        v_current_enrollments, v_average_grade, v_pass_rate
    FROM courses c
    WHERE c.course_id = p_course_id
    ON DUPLICATE KEY UPDATE
        total_enrollments = v_total_enrollments,
        current_enrollments = v_current_enrollments,
        average_grade = v_average_grade,
        pass_rate = v_pass_rate,
        last_updated = CURRENT_TIMESTAMP;
END//
```

```sql
DELIMITER //
CREATE PROCEDURE CalculateGPAForAllStudents()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE v_student_id VARCHAR(20);
    DECLARE v_gpa DECIMAL(3,2);
    
    -- 声明游标
    DECLARE student_cursor CURSOR FOR 
        SELECT student_id FROM students WHERE status = 'Active';
    
    -- 声明异常处理
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;
    
    -- 打开游标
    OPEN student_cursor;
    
    -- 循环处理
    student_loop: LOOP
        FETCH student_cursor INTO v_student_id;
        
        IF done THEN
            LEAVE student_loop;
        END IF;
        
        -- 计算GPA
        SELECT AVG(grade) INTO v_gpa
        FROM enrollments
        WHERE student_id = v_student_id 
          AND grade IS NOT NULL 
          AND status = 'Completed';
        
        -- 更新学生GPA
        UPDATE students 
        SET gpa = COALESCE(v_gpa, 0)
        WHERE student_id = v_student_id;
        
    END LOOP;
    
    -- 关闭游标
    CLOSE student_cursor;
END//
DELIMITER ;
```

```sql
DELIMITER //
CREATE PROCEDURE EnrollStudent(
    IN p_student_id VARCHAR(20),
    IN p_course_id VARCHAR(20),
    IN p_teacher_id VARCHAR(20),
    IN p_semester VARCHAR(20),
    IN p_academic_year VARCHAR(10)

    DECLARE v_max_students INT;
    DECLARE v_current_enrollments INT;
    
    -- 声明异常处理
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        GET DIAGNOSTICS CONDITION 1
            v_error_msg = MESSAGE_TEXT;
        SELECT CONCAT('Error: ', v_error_msg) as error_message;
    END;
    
    -- 开始事务
    START TRANSACTION;
    
    -- 检查课程容量
    SELECT max_students INTO v_max_students
    FROM teaching_assignments
    WHERE teacher_id = p_teacher_id 
      AND course_id = p_course_id 
      AND semester = p_semester 
      AND academic_year = p_academic_year;
    
    SELECT COUNT(*) INTO v_current_enrollments
    FROM enrollments
    WHERE course_id = p_course_id 
      AND semester = p_semester 
      AND academic_year = p_academic_year
      AND status = 'Enrolled';
    
    -- 检查是否超过容量
    IF v_current_enrollments >= v_max_students THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Course is full';
    END IF;
    
    -- 插入选课记录
    INSERT INTO enrollments (
        student_id, course_id, teacher_id, 
        semester, academic_year, enrollment_date
    ) VALUES (
        p_student_id, p_course_id, p_teacher_id,
        p_semester, p_academic_year, CURDATE()
    );
    
    -- 提交事务
    COMMIT;
    
    SELECT 'Student enrolled successfully' as message;
END//
DELIMITER ;
```

```sql
DELIMITER //
CREATE TRIGGER update_gpa_after_grade_change
AFTER UPDATE ON enrollments
FOR EACH ROW
BEGIN
    DECLARE v_new_gpa DECIMAL(3,2);
    DECLARE v_total_credits INT;
    
    -- 只有成绩发生变化时才更新
    IF OLD.grade != NEW.grade OR OLD.status != NEW.status THEN
        
        -- 计算新的GPA
        SELECT 
            COALESCE(AVG(e.grade), 0),
            COALESCE(SUM(c.credits), 0)
        INTO v_new_gpa, v_total_credits
        FROM enrollments e
        JOIN courses c ON e.course_id = c.course_id
        WHERE e.student_id = NEW.student_id 
          AND e.status = 'Completed'
          AND e.grade IS NOT NULL
          AND e.grade >= 60;
        
        -- 更新学生表
        UPDATE students 
        SET 
            gpa = v_new_gpa,
            total_credits = v_total_credits,
            updated_at = NOW()
        WHERE student_id = NEW.student_id;
        
        -- 更新汇总表
        UPDATE student_course_summary
        SET 
            gpa = v_new_gpa,
            total_credits = v_total_credits,
            completed_courses = (
                SELECT COUNT(*)
                FROM enrollments
                WHERE student_id = NEW.student_id 
                  AND status = 'Completed'
                  AND grade >= 60
            ),
            last_updated = NOW()
        WHERE student_id = NEW.student_id;
```

### 6.4 游标

#### 6.4.1 游标基本操作

```sql
-- 使用游标处理批量数据
DELIMITER //
CREATE PROCEDURE ProcessLowGPAStudents()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE v_student_id VARCHAR(20);
    DECLARE v_name VARCHAR(50);
    DECLARE v_gpa DECIMAL(3,2);
    DECLARE v_major VARCHAR(50);
    
    -- 声明游标
    DECLARE low_gpa_cursor CURSOR FOR
        SELECT student_id, name, gpa, major
        FROM students
        WHERE gpa < 2.0 AND status = 'Active'
        ORDER BY gpa ASC;
    
    -- 声明异常处理
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;
    
    -- 创建临时表存储结果
    CREATE TEMPORARY TABLE IF NOT EXISTS temp_warnings (
        student_id VARCHAR(20),
        name VARCHAR(50),
        gpa DECIMAL(3,2),
        major VARCHAR(50),
        warning_level VARCHAR(20),
        action_required VARCHAR(100)
    );
    
    -- 打开游标
    OPEN low_gpa_cursor;
    
    -- 循环处理
    process_loop: LOOP
        FETCH low_gpa_cursor INTO v_student_id, v_name, v_gpa, v_major;
        
        IF done THEN
            LEAVE process_loop;
        END IF;
        
        -- 根据GPA确定警告级别
        IF v_gpa < 1.0 THEN
            INSERT INTO temp_warnings VALUES (
                v_student_id, v_name, v_gpa, v_major,
                'CRITICAL', 'Academic probation - immediate counseling required'
            );
        ELSEIF v_gpa < 1.5 THEN
            INSERT INTO temp_warnings VALUES (
                v_student_id, v_name, v_gpa, v_major,
                'HIGH', 'Academic warning - counseling recommended'
            );
        ELSE
            INSERT INTO temp_warnings VALUES (
                v_student_id, v_name, v_gpa, v_major,
                'MEDIUM', 'Academic support available'
            );
        END IF;
        
    END LOOP;
    
    -- 关闭游标
    CLOSE low_gpa_cursor;
    
    -- 返回结果
    SELECT * FROM temp_warnings ORDER BY gpa ASC;
    
    -- 清理临时表
    DROP TEMPORARY TABLE temp_warnings;
END//
DELIMITER ;

-- 调用存储过程
CALL ProcessLowGPAStudents();
```

#### 9.1.3 聚合操作

```javascript
// 基本聚合
db.students.aggregate([
    {$match: {major: "Computer Science"}},
    {$group: {
        _id: "$major",
        avgGPA: {$avg: "$gpa"},
        count: {$sum: 1},
        maxGPA: {$max: "$gpa"}
    }}
]);

// 复杂聚合管道
db.students.aggregate([
    // 展开courses数组
    {$unwind: "$courses"},
    
    // 匹配高分课程
    {$match: {"courses.grade": {$gte: 85}}},
    
    // 按专业分组
    {$group: {
        _id: "$major",
        students: {$addToSet: "$name"},
        avgGrade: {$avg: "$courses.grade"},
        totalCourses: {$sum: 1}
    }},
    
    // 排序
    {$sort: {avgGrade: -1}},
    
    // 投影
    {$project: {
        major: "$_id",
        studentCount: {$size: "$students"},
        avgGrade: {$round: ["$avgGrade", 2]},
        totalCourses: 1,
        _id: 0
    }}
]);

// 查找操作（类似SQL的JOIN）
db.students.aggregate([
    {
        $lookup: {
            from: "courses",
            localField: "courses.course_id",
            foreignField: "course_id",
            as: "course_details"
        }
    },
    {
        $project: {
            name: 1,
            major: 1,
            course_details: {$slice: ["$course_details", 3]}
        }
    }
]);
```

SELECT 
    o.order_id,
    u.username,
    o.order_date,
    SUM(oi.quantity * oi.price) as total_amount,
    COUNT(oi.product_id) as item_count
FROM orders o
JOIN users u ON o.user_id = u.user_id
JOIN order_items oi ON o.order_id = oi.order_id
WHERE o.order_date >= DATE_SUB(NOW(), INTERVAL 30 DAY)
GROUP BY o.order_id, u.username, o.order_date
